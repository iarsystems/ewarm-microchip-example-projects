unsigned char DMAD_IsFinished(unsigned char channel)
{
    SANITY_CHECK(channel <= DMA_CHANNEL_NUM);
    if (dmad.transfers[channel].transferSize > 0) {
        return 0;
    }
    else {
        DMA_DisableChannel(channel);
        return 1;
    }
}
