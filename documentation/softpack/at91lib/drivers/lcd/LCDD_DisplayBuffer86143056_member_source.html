void * LCDD_DisplayBuffer(void *pBuffer)
{
    void *pOldBuffer;

    pOldBuffer = LCD_SetFrameBufferAddress(pBuffer);

    // Enable LCD & DMA if needed
    if ((AT91C_BASE_LCDC->LCDC_DMACON & AT91C_LCDC_DMAEN) != AT91C_LCDC_DMAEN) {

        LCD_EnableDma();
        LCD_Enable(0x0C);
    }

    return pOldBuffer;
}
