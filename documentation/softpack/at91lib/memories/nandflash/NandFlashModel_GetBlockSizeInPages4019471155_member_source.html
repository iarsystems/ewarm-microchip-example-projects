unsigned short NandFlashModel_GetBlockSizeInPages(const struct NandFlashModel *model)
{
    #if !defined(OP_BOOTSTRAP_on)
    return model->blockSizeInKBytes * 1024 / model->pageSizeInBytes;
    #else
    unsigned int pow;
    pow = CALPOW((model->blockSizeInKBytes * 1024), model->pageSizeInBytes);
    return (0x1 << pow);
    #endif
}
