void USART_PutChar(AT91S_USART *usart, unsigned char c)
{
    // Wait for the transmitter to be ready  
    while ((usart->US_CSR & AT91C_US_TXEMPTY) == 0);

    // Send character    
    usart->US_THR = c;
    
    // Wait for the transfer to complete    
    while ((usart->US_CSR & AT91C_US_TXEMPTY) == 0);    
}
