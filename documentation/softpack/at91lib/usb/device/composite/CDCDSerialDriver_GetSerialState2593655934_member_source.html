unsigned short CDCDSerialDriver_GetSerialState(unsigned char port)
{
    if (port > cdcFunDriver.numPorts)
        return USBD_STATUS_INVALID_PARAMETER;

    return cdcFunDriver.pCdcPorts[port].serialState;
}
