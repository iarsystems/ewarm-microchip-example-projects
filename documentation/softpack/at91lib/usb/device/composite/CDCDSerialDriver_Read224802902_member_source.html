unsigned char CDCDSerialDriver_Read(unsigned char port, void *data, unsigned int size, TransferCallback callback, void *argument)
{
    unsigned char ep = cdcFunDriver.pCdcPorts[port].bulkOutEndpoint;
    
    if (port > cdcFunDriver.numPorts)
        return USBD_STATUS_INVALID_PARAMETER;

    return USBD_Read(ep,
                     data,
                     size,
                     callback,
                     argument);
}
