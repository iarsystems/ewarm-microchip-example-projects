void TestLCD_BGR_240_320_24bpp(unsigned char *pBufferLCD)
{
    unsigned int i;
    unsigned int j;

    j=0;
    while( j<(240*3*320) ) {
        // 24 bpp
        i=0;
        while(i<240) { // Red in BGR
            pBufferLCD[j+i] = 0xFF;
            i++;
            pBufferLCD[j+i] = 0x00;
            i++;
            pBufferLCD[j+i] = 0x00;
            i++;
        }
        j+=240;
        i=0;
        while(i<240) {  // Green
            pBufferLCD[j+i] = 0x00;
            i++;
            pBufferLCD[j+i] = 0xFF;
            i++;
            pBufferLCD[j+i] = 0x00;
            i++;
        }
        j+=240;
        i=0;
        while(i<240) {  // Blue
            pBufferLCD[j+i] = 0x00;
            i++;
            pBufferLCD[j+i] = 0x00;
            i++;
            pBufferLCD[j+i] = 0xFF;
            i++;
        }
        j+=240;
    }

}
