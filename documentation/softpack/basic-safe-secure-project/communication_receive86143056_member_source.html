unsigned int communication_receive(void *vAddress)
{
    unsigned int bytes;
    unsigned int remaining;
    unsigned int endOfFirmware;
    unsigned int i;

    // Restart transfer
    us0_writeChar(XON);

    // Wait for data, monitoring activity to detect the end of the firmware
    endOfFirmware = 0;
    bytes = 0;
    while((endOfFirmware == 0) && (bytes < MEDIA_BOOT_PAGE_SIZE)) {
        bytes = us0_poll();
        endOfFirmware = 1;
        for (i=0; i < XON_XOFF_WAIT_COUNT; i++) {
            if (bytes != us0_poll()) {
                endOfFirmware = 0;
                break;
            }
        }
    }
    if (bytes != 0) {
        // Read page
        if (bytes > MEDIA_BOOT_PAGE_SIZE) {
            bytes = MEDIA_BOOT_PAGE_SIZE;
        }
        bytes = us0_read(vAddress, bytes);

        // Send XOFF control character and wait for remaining characters
        us0_writeChar(XOFF);
        remaining = us0_poll();
    
        for (i=0; i < XON_XOFF_WAIT_COUNT; i++) {
            if (remaining != us0_poll()) {
                remaining = us0_poll();
                i = 0;
            }
        }
    }
    TRACE_DEBUG("bytes: %d\n\r", bytes);

    return bytes;
}
