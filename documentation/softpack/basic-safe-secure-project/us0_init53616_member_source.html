void us0_init(void)
{
    unsigned int i;

    TRACE_DEBUG("USART0: Initializing ...\n\r");

    // Configure USART
    PIO_Configure(pinsUsart, PIO_LISTSIZE(pinsUsart));
    PMC_EnablePeripheral(AT91C_ID_US0);
    AT91C_BASE_US0->US_IDR = 0xFFFFFFFF;

    // Configure parameters
    USART_Configure(AT91C_BASE_US0,
                    US0_MODE,
                    US0_BAUDRATE,
                    BOARD_MCK);
    USART_SetTransmitterEnabled(AT91C_BASE_US0, 1);
    USART_SetReceiverEnabled(AT91C_BASE_US0, 1);

    // Initialize cyclic buffer
    cyclicDataStart = 0;
    cyclicDataEnd = 0;
    cyclicDataSize = 0;
    for( i=0; i<US0_BUFFER_SIZE; i++ ) {
        cyclicBuffer[i] = 0;
    }
    // Configure rxReady interrupt
    IRQ_ConfigureIT(AT91C_ID_US0,
                    0x7,
                    us0_interruptHandler);

    IRQ_EnableIT(AT91C_ID_US0);
    TRACE_DEBUG("USART0: Initialization done.\n\r");
    // Enable IT
    AT91C_BASE_US0->US_IER = AT91C_US_RXRDY;
}
