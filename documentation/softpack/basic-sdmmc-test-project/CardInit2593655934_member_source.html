void CardInit(unsigned char slot)
{
    unsigned int mciSpeed;
    unsigned int mciDiv;

    DumpSeperator();

    if (!SDInitExec())
        return;

    DumpCardInfo(slot);

    mciSpeed = 10000000;
    ChangeMciSpeed(slot, mciSpeed);

    MCI_GetSpeed(&mciDrv, &mciDiv);
    printf("-I- MCK %dK Hz, MCI Speed %dK, divisor %d. \r\n",
           BOARD_MCK/1000, mciSpeed/1000, mciDiv);

}
