void DumpBlock(unsigned char *pData, unsigned int block)
{
    unsigned int i;
    //printf("-I- Block %d: %s .. %s..", block, pData, &pData[8]);
    printf("-I- Block %d: %c .. %c .. %c .. %c..",
            block, pData[0], pData[3], pData[8], pData[8+5]);
    for (i = 0; i < 512; i ++) {
        if((i % 16) == 0) printf("\n\r%3x:", i);
        printf(" %02X", pData[i]);
    }
    printf("\n\r");
}
