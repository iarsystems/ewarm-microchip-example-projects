unsigned char KBD_Code2Ascii(unsigned char code)
{
    // Alphabetic
    if ((code >= 'A') && (code <= 'Z')) {

        return code;
    }
    // Numeric
    else if (KBD_Code2Digit(code) >= 0) {

        return code;
    }
    // Special keys
    else {

        switch (code) {
        case S7LEKKBM_PLUS: return '+';
        case S7LEKKBM_MINUS: return '-';
        case S7LEKKBM_EQUAL: return '=';
        default: return 0;
        }
    }
}
