unsigned char KBD_GetKeyCode(unsigned char key)
{
    // Natural key
    if ((!g2ndF && !g2ndFLock) || (g2ndF && g2ndFLock)) {

        return gpKeyboardMatrix[key];
    }
    // Alternate key
    else {

        return gpKeyboardMatrixAlt[key];
    }
}
