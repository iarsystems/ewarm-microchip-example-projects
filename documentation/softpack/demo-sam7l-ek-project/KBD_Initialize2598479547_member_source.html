void KBD_Initialize(KeyPressedCallback callback)
{
    unsigned int div, tcclks;

    TRACE_DEBUG("KBD_Initialize()\n\r");
    SANITY_CHECK(callback);

    // Set callback function
    lCallback = callback;

    // Keyboard is not idle
    idle = 0;

    // Configure keyboard matrix pins
    TRACE_DEBUG("Configuring PIOs\n\r");
    PIO_Configure(lPinsRow, PIO_LISTSIZE(lPinsRow));
    PIO_Configure(lPinsCol, PIO_LISTSIZE(lPinsCol));

    // Configure TC0
    TRACE_DEBUG("Configuring TC0\n\r");
    AT91C_BASE_PMC->PMC_PCER = 1 << AT91C_ID_TC0;
    ASSERT(TC_FindMckDivisor(SAMPLE_RATE, BOARD_MCK, &div, &tcclks),
           "-F- KBD_Initialize: Could not generate the desired timer frequency\n\r");
    TC_Configure(AT91C_BASE_TC0, tcclks | AT91C_TC_WAVE | AT91C_TC_WAVESEL_UP_AUTO);
    AT91C_BASE_TC0->TC_RC = BOARD_MCK / (SAMPLE_RATE * div);

     // Configure TC0 interrupt
    IRQ_DisableIT(AT91C_ID_TC0);
    IRQ_ConfigureIT(AT91C_ID_TC0, 0, ISR_Tc0);
    AT91C_BASE_TC0->TC_IER = AT91C_TC_CPCS;
    IRQ_EnableIT(AT91C_ID_TC0);

    // Initialize keyboard
    TRACE_DEBUG("Configuring keyboard matrix driver\n\r");
    KBMATRIX_Initialize(&lKbMatrix,
                        lPinsRow,
                        PIO_LISTSIZE(lPinsRow),
                        lPinsCol,
                        PIO_LISTSIZE(lPinsCol),
                        KeyStateChanged);

    // Start timer
    TC_Start(AT91C_BASE_TC0);
}
