void * MODE_BoardName(signed char key, unsigned char refresh)
{
    unsigned char code = KBD_GetKeyCode(key);

    // Display the board name
    if (refresh == MODE_REFRESH_RESTART) {

        TRACE_INFO("Board name mode restart\n\r");

        memcpy(gpNewBoardName, gpBoardName, S7LEKLCD_MAX_CHARS);
        gEdit = 0;
    }
    else if (refresh == MODE_REFRESH_DISPLAY) {

        TRACE_INFO("Board name mode refresh\n\r");
    }

    // Process key events
    if (key >= 0) {

        // Escape key pressed
        if (code == S7LEKKBM_ESCAPE) {

            return MODE_Menu;
        }
        // Left arrow key pressed
        else if (code == S7LEKKBM_LEFTARROW) {

            if (gEdit > 0) {

                gEdit--;
            }
        }
        // Right arrow key pressed
        else if (code == S7LEKKBM_RIGHTARROW) {

            if (gEdit < (S7LEKLCD_MAX_CHARS-1)) {

                gEdit++;
            }
        }
        // Delete key pressed
        else if (code == S7LEKKBM_DELETE) {

            // Delete current character without moving the cursor
            gpNewBoardName[gEdit] = ' ';
        }
        // Ok key pressed
        else if (KBD_GetNaturalKeyCode(key) == S7LEKKBM_OK) {

            memcpy(gpBoardName, gpNewBoardName, S7LEKLCD_MAX_CHARS);
            return MODE_Menu;
        }
        // Other keys
        else {

            if (KBD_Code2Ascii(code)) {

                code = KBD_Code2Ascii(code);
                gpNewBoardName[gEdit] = code;
                if (gEdit < (S7LEKLCD_MAX_CHARS-1)) {

                    gEdit++;
                }
            }
        }
    }

    // Refresh string
    SLCD_PutString(gpNewBoardName, 0);
    SLCD_SetBlinking(1 << gEdit);

    return MODE_BoardName;
}
