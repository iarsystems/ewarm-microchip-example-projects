void * MODE_Calc(signed char key, unsigned char refresh)
{
    unsigned char code = KBD_GetNaturalKeyCode(key);
    char pString[2 * MAX_OPERAND_SIZE];
    unsigned int num = 0;

    // Restart
    if (refresh == MODE_REFRESH_RESTART) {

        TRACE_DEBUG("Calculator mode refresh\n\r");

        SLCD_PutSymbol(S7LEKLCD_CALC, 1);
    }

    // Process key events
    if (key >= 0) {

        // Return to menu mode
        if (code == S7LEKKBM_MODE) {

            SLCD_PutSymbol(S7LEKLCD_CALC, 0);
            return MODE_Menu;
        }
        // Return to default mode
        else if (code == S7LEKKBM_ESCAPE) {

            SLCD_PutSymbol(S7LEKLCD_CALC, 0);
            return MODE_Default;
        }
        // Clear
        else if (code == S7LEKKBM_DELETE) {

            gCalcState = STATE_FIRSTOPERAND;
            gCalcOperand1 = 0;
            gCalcOperand2 = 0;
        }
        // State machine
        else {

            // Display result
            if (gCalcState == STATE_RESULT_DISPLAY) {

                // Operator
                if ((code == S7LEKKBM_PLUS) || (code == S7LEKKBM_MINUS)
                    || (code == S7LEKKBM_MULTIPLY) || (code == S7LEKKBM_DIVIDE)) {

                    gCalcOperand1 = gCalcResult;
                    gCalcOperand2 = 0;
                    gCalcState = STATE_OPERATOR;
                }
                // Number
                else if ((code >= '0') && (code <= '9')) {

                    gCalcOperand1 = 0;
                    gCalcOperator = OPERATOR_NONE;
                    gCalcOperand2 = 0;
                    gCalcState = STATE_FIRSTOPERAND;
                }
                // Sign
                else if (code == S7LEKKBM_SIGN) {

                    gCalcOperand1 = gCalcResult;
                    gCalcOperator = OPERATOR_NONE;
                    gCalcOperand2 = 0;
                    gCalcState = STATE_FIRSTOPERAND;
                }
                // Equal
                else if (code == S7LEKKBM_EQUAL) {

                    gCalcOperand1 = gCalcResult;
                    gCalcState = STATE_RESULT_COMPUTE;
                }
            }

            // First operand integer part
            if (gCalcState == STATE_FIRSTOPERAND) {

                // Compute current size of operand
                num = sprintf(pString, "%d", gCalcOperand1);

                // Number
                if ((code >= '0') && (code <= '9')) {
                    if (num < MAX_OPERAND_SIZE) {

                        if (gCalcOperand1 >= 0) {

                            gCalcOperand1 = gCalcOperand1 * 10 + code - '0';
                        }
                        else {

                            gCalcOperand1 = gCalcOperand1 * 10 - code + '0';
                        }
                    }
                }
                // Sign
                else if (code == S7LEKKBM_SIGN) {

                    gCalcOperand1 = -gCalcOperand1;
                }
                // Operator
                else if ((code == S7LEKKBM_PLUS) || (code == S7LEKKBM_MINUS)
                         || (code == S7LEKKBM_MULTIPLY) || (code == S7LEKKBM_DIVIDE)) {
    
                    gCalcState = STATE_OPERATOR;
                }
                // Result
                else if (code == S7LEKKBM_EQUAL) {

                    gCalcState = STATE_RESULT_COMPUTE;
                }
            }

            // Operator
            if (gCalcState == STATE_OPERATOR) {

                // Plus
                if (code == S7LEKKBM_PLUS) {

                    gCalcOperator = OPERATOR_PLUS;
                }
                // Minus
                else if (code == S7LEKKBM_MINUS) {

                    gCalcOperator = OPERATOR_MINUS;
                }
                // Multiply
                else if (code == S7LEKKBM_MULTIPLY) {

                    gCalcOperator = OPERATOR_MULTIPLY;
                }
                // Divide
                else if (code == S7LEKKBM_DIVIDE) {

                    gCalcOperator = OPERATOR_DIVIDE;
                }
                // Number
                if ((code >= '0') && (code <= '9')) {
                    
                    gCalcState = STATE_SECONDOPERAND;
                }
            }

            // Second operand integer part
            if (gCalcState == STATE_SECONDOPERAND) {

                // Compute current size of operand
                num = sprintf(pString, "%d", gCalcOperand2);

                // Number
                if ((code >= '0') && (code <= '9')) {
                    if (num < MAX_OPERAND_SIZE) {

                        if (gCalcOperand2 >= 0) {
                            
                            gCalcOperand2 = gCalcOperand2 * 10 + code - '0';
                        }
                        else {

                            gCalcOperand2 = gCalcOperand2 * 10 - code + '0';
                        }
                    }
                }
                // Sign
                else if (code == S7LEKKBM_SIGN) {

                    gCalcOperand2 = -gCalcOperand2;
                }
                // Result
                else if (code == S7LEKKBM_EQUAL) {

                    gCalcState = STATE_RESULT_COMPUTE;
                }
            }

            // Compute result
            if (gCalcState == STATE_RESULT_COMPUTE) {

                switch (gCalcOperator) {

                    case OPERATOR_NONE:
                        gCalcResult = gCalcOperand1;
                        gCalcState = STATE_RESULT_DISPLAY;
                        break;

                    case OPERATOR_PLUS:
                        gCalcResult = gCalcOperand1 + gCalcOperand2;
                        num = sprintf(pString, "%d", gCalcResult);
                        if (num > MAX_OPERAND_SIZE) {

                            gCalcState = STATE_OVERFLOW;
                        }
                        else {

                            gCalcState = STATE_RESULT_DISPLAY;
                        }
                        break;

                    case OPERATOR_MINUS:
                        gCalcResult = gCalcOperand1 - gCalcOperand2;
                        num = sprintf(pString, "%d", gCalcResult);
                        if (num > MAX_OPERAND_SIZE) {

                            gCalcState = STATE_OVERFLOW;
                        }
                        else {

                            gCalcState = STATE_RESULT_DISPLAY;
                        }
                        break;

                    case OPERATOR_MULTIPLY:
                        num = sprintf(pString, "%d", gCalcOperand1 * gCalcOperand2);
                        if (num > MAX_OPERAND_SIZE) {

                            gCalcState = STATE_OVERFLOW;
                        }
                        else {

                            gCalcResult = gCalcOperand1 * gCalcOperand2;
                            gCalcState = STATE_RESULT_DISPLAY;
                        }
                        break;

                    case OPERATOR_DIVIDE:

                        if (gCalcOperand2 == 0) {

                            gCalcState = STATE_DIVBY0;
                        }
                        else {

                            gCalcResult = gCalcOperand1 / gCalcOperand2;
                            gCalcState = STATE_RESULT_DISPLAY;
                        }
                        break;
                }
            }
        }
    }

    // Refresh string
    switch (gCalcState) {

        case STATE_FIRSTOPERAND:
        case STATE_OPERATOR:
            num = sprintf(pString, "%d", gCalcOperand1);
            break;
    
        case STATE_SECONDOPERAND:
            num = sprintf(pString, "%d", gCalcOperand2);
            break;

        case STATE_RESULT_DISPLAY:
            num = sprintf(pString, "%d", gCalcResult);
            break;

        case STATE_OVERFLOW:
            num = sprintf(pString, "-OVERFLOW-");
            break;

        case STATE_DIVBY0:
            num = sprintf(pString, "-DIV BY 0-");
            break;
    }
    SLCD_PutString(pString, S7LEKLCD_MAX_CHARS - num - 1);

    return MODE_Calc;
}
