void * MODE_Date(signed char key, unsigned char refresh)
{
    unsigned char code = KBD_GetNaturalKeyCode(key);
    char string[] = "DD-MON-YYYY";
    unsigned char maxDay;
    const char *pMonths[13] = {"JAN", "FEB", "MAR", "APR", "MAY", "JUN",
                               "JUL", "AUG", "SEP", "OCT", "NOV", "DEC"};

    // Refresh display
    if (refresh == MODE_REFRESH_RESTART) {

        TRACE_INFO("Date mode restart\n\r");

        gEdit = 0;
        gNewDay = dateDay;
        gNewMonth = dateMonth;
        gNewYear = dateYear;
        S7LEKLCD_Symbol(S7LEKLCD_DATE, 1);
    }
    else if (refresh == MODE_REFRESH_DISPLAY) {

        TRACE_INFO("Date mode refresh\n\r");

        S7LEKLCD_Symbol(S7LEKLCD_DATE, 1);
    }

    // Process key events
    if (key >= 0) { 

        // Cancel
        if (code == S7LEKKBM_ESCAPE) {

            S7LEKLCD_Symbol(S7LEKLCD_DATE, 0);
            return MODE_Menu;
        }
        // Validate
        else if (code == S7LEKKBM_OK) {

            dateYear = gNewYear;
            dateMonth = gNewMonth;
            dateDay = gNewDay;
            S7LEKLCD_Symbol(S7LEKLCD_DATE, 0);
            return MODE_Menu;
        }
        // Move cursor left
        else if (code == S7LEKKBM_LEFTARROW) {

            if (gEdit > 0) {

                gEdit--;
            }
        }
        // Move cursor right
        else if (code == S7LEKKBM_RIGHTARROW) {

            if (gEdit < 2) {

                gEdit++;
            }
        }
        // Increase
        else if (code == S7LEKKBM_UPARROW) {

            // Day
            if (gEdit == 0) {

                if (gNewDay < 31) {

                    gNewDay++;
                }
            }
            // Month
            else if (gEdit == 1) {

                if (gNewMonth < 12) {

                    gNewMonth++;
                }
            }
            // Year
            else {

                if (gNewYear < 9999) {

                    gNewYear++;
                }
            }
        }
        // Decrease
        else if (code == S7LEKKBM_DOWNARROW) {

            // Day
            if (gEdit == 0) {

                if (gNewDay > 1) {

                    gNewDay--;
                }
            }
            // Month
            else if (gEdit == 1) {

                if (gNewMonth > 1) {

                    gNewMonth--;
                }
            }
            // Year
            else {

                if (gNewYear > 0) {

                    gNewYear--;
                }
            }
        }
    }

    // Enforce day/month/year constraints
    switch (gNewMonth) {
    case 1:
    case 3:
    case 5:
    case 7:
    case 8:
    case 10:
    case 12: maxDay = 31; break;
    case 2:
        if ((((gNewYear % 100) != 0) && ((gNewYear % 4) == 0))
           || ((gNewYear % 400) == 0)) {

            maxDay = 29;
        }
        else {

            maxDay = 28;
        }
        break;
    default: maxDay = 30;
    }

    gNewDay = min(gNewDay, maxDay);

    // Display current date
    string[0] = gNewDay / 10 + '0';
    string[1] = gNewDay % 10 + '0';
    memcpy(&(string[3]), pMonths[gNewMonth-1], 3);
    string[7] = gNewYear / 1000 + '0';
    string[8] = (gNewYear / 100) % 10 + '0';
    string[9] = (gNewYear / 10) % 10 + '0';
    string[10] = gNewYear % 10 + '0';
    SLCD_PutString(string, 0);

    // Enable blinking on current section
    switch (gEdit) {
    case 0: SLCD_SetBlinking(0x0003); break;
    case 1: SLCD_SetBlinking(0x0038); break;
    case 2: SLCD_SetBlinking(0x0780); break;
    }

    return MODE_Date;
}
