void * MODE_LcdContrast(signed char key, unsigned char refresh)
{
    unsigned char code = KBD_GetNaturalKeyCode(key);

    // Display the board name
    if (refresh == MODE_REFRESH_RESTART) {

        TRACE_INFO("LCD contrast mode restart\n\r");

        gNewContrast = gContrast;
    }
    else if (refresh == MODE_REFRESH_DISPLAY) {

        TRACE_INFO("LCD contrast mode refresh\n\r");
    }

    // Process key events
    if (key >= 0) {

        // Cancel
        if (code == S7LEKKBM_ESCAPE) {

            SLCD_SetContrast(gContrast);
            SLCD_SetVolume(0);
            return MODE_Menu;
        }
        // Validate
        else if(code == S7LEKKBM_OK) {

            gContrast = gNewContrast;
            SLCD_SetContrast(gContrast);
            SLCD_SetVolume(0);
            return MODE_Menu;
        }
        // Decrease
        else if(code == S7LEKKBM_LEFTARROW) {

            if (gNewContrast > 0) {

                gNewContrast--;
            }
        }
        // Increase
        else if(code == S7LEKKBM_RIGHTARROW) {

            if (gNewContrast < 4) {

                gNewContrast++;
            }
        }
    }

    // Refresh contrast display
    SLCD_SetContrast(gNewContrast);
    SLCD_SetVolume(gNewContrast+1);

    return MODE_LcdContrast;
}
