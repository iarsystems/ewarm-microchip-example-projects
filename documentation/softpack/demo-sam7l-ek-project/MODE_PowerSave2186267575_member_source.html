void * MODE_PowerSave(signed char key, unsigned char refresh)
{
    char string[8] = "XX MIN.";
    unsigned char code = KBD_GetNaturalKeyCode(key);

    // Refresh display
    if (refresh == MODE_REFRESH_RESTART) {

        TRACE_INFO("Power save mode restart\n\r");
        gNewPowerSaveTimeout = gPowerSaveTimeout;
    }
    else if (refresh == MODE_REFRESH_DISPLAY) {

        TRACE_INFO("Power save mode refresh\n\r");
    }

    // Process key events
    if (key >= 0) {

        // Cancel
        if (code == S7LEKKBM_ESCAPE) {

            return MODE_Menu;
        }
        // Validate
        else if (code == S7LEKKBM_OK) {

            gPowerSaveTimeout = gNewPowerSaveTimeout;
            return MODE_Menu;
        }
        // Increase
        else if (code == S7LEKKBM_UPARROW) {

            if (gNewPowerSaveTimeout < 20) {

                gNewPowerSaveTimeout += 5;
            }
        }
        // Decrease
        else if (code == S7LEKKBM_DOWNARROW) {

            if (gNewPowerSaveTimeout > 0) {

                gNewPowerSaveTimeout -= 5;
            }
        }
    }

    // Display current timeout value
    if (gNewPowerSaveTimeout > 0) {

        string[0] = gNewPowerSaveTimeout / 10 + '0';
        string[1] = gNewPowerSaveTimeout % 10 + '0';
        SLCD_PutString(string, 0);
    }
    else {

        SLCD_PutString("NONE", 0);
    }

    return MODE_PowerSave;
}
