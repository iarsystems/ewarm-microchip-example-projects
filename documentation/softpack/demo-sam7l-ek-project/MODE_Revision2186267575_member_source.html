void * MODE_Revision(signed char key, unsigned char refresh)
{
    // Display the revision
    if (refresh) {

        TRACE_INFO("Revision mode restart\n\r");
        SLCD_PutString(REVISION, 0);
    }

    // Process key events
    if (key >= 0) {

        if (KBD_GetNaturalKeyCode(key) == S7LEKKBM_ESCAPE) {

            return MODE_Menu;
        }
    }

    return MODE_Revision;
