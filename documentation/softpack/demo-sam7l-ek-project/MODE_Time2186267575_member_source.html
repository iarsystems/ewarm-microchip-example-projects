void * MODE_Time(signed char key, unsigned char refresh)
{
    unsigned char code = KBD_GetNaturalKeyCode(key);
    char string[] = "HHMMSS";

    // Refresh display
    if (refresh == MODE_REFRESH_RESTART) {

        TRACE_INFO("Time mode restart\n\r");

        gEdit = 0;
        RTC_GetTime(&gNewHours, &gNewMinutes, &gNewSeconds);

        S7LEKLCD_Symbol(S7LEKLCD_TIME, 1);
        S7LEKLCD_Symbol(S7LEKLCD_COL2, 1);
        S7LEKLCD_Symbol(S7LEKLCD_COL3, 1);
    }
    else if (refresh == MODE_REFRESH_DISPLAY) {

        TRACE_INFO("Time mode refresh\n\r");

        S7LEKLCD_Symbol(S7LEKLCD_TIME, 1);
        S7LEKLCD_Symbol(S7LEKLCD_COL2, 1);
        S7LEKLCD_Symbol(S7LEKLCD_COL3, 1);
    }

    // Process key events
    if (key >= 0) {

        // Cancel
        if (code == S7LEKKBM_ESCAPE) {

            S7LEKLCD_Symbol(S7LEKLCD_TIME, 0);
            S7LEKLCD_Symbol(S7LEKLCD_COL2, 0);
            S7LEKLCD_Symbol(S7LEKLCD_COL3, 0);
            return MODE_Menu;
        }
        // Validate
        else if (code == S7LEKKBM_OK) {

            RTC_SetTime(gNewHours, gNewMinutes, gNewSeconds);
            S7LEKLCD_Symbol(S7LEKLCD_TIME, 0);
            S7LEKLCD_Symbol(S7LEKLCD_COL2, 0);
            S7LEKLCD_Symbol(S7LEKLCD_COL3, 0);
            return MODE_Menu;
        }
        // Increase
        else if (code == S7LEKKBM_UPARROW) {

            // Hours
            if (gEdit == 0) {

                if (gNewHours < 23) {

                    gNewHours++;
                }
            }
            // Minutes
            else if (gEdit == 1) {

                if (gNewMinutes < 59) {

                    gNewMinutes++;
                }
            }
            // Seconds
            else if (gEdit == 2) {

                if (gNewSeconds < 59) {

                    gNewSeconds++;
                }
            }
        }
        // Decrease
        else if (code == S7LEKKBM_DOWNARROW) {

            // Hours
            if (gEdit == 0) {

                if (gNewHours > 0) {

                    gNewHours--;
                }
            }
            // Minutes
            else if (gEdit == 1) {

                if (gNewMinutes > 0) {

                    gNewMinutes--;
                }
            }
            // Seconds
            else if (gEdit == 2) {

                if (gNewSeconds > 0) {

                    gNewSeconds--;
                }
            }
        }
        // Move cursor to the left
        else if (code == S7LEKKBM_LEFTARROW) {

            if (gEdit > 0) {

                gEdit--;
            }
        }
        // Move cursor to the right
        else if (code == S7LEKKBM_RIGHTARROW) {

            if (gEdit < 2) {

                gEdit++;
            }
        }
    }

    // Display current time
    string[0] = gNewHours / 10 + '0';
    string[1] = gNewHours % 10 + '0';
    string[2] = gNewMinutes / 10 + '0';
    string[3] = gNewMinutes % 10 + '0';
    string[4] = gNewSeconds / 10 + '0';
    string[5] = gNewSeconds % 10 + '0';
    SLCD_PutString(string, 6);

    // Update blinking
    switch (gEdit) {
    case 0: SLCD_SetBlinking(0x00C0); break;
    case 1: SLCD_SetBlinking(0x0300); break;
    case 2: SLCD_SetBlinking(0x0C00); break;
    }

    return MODE_Time;
}
