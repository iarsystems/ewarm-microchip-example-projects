void SLCD_Initialize(void)
{
    AT91C_BASE_SUPC->SUPC_MR |= (0xA5 << 24) | AT91C_SUPC_LCDMODE_INTERNAL | 0x8;
    while ((AT91C_BASE_SUPC->SUPC_SR & AT91C_SUPC_LCDS) == 0);
    
    // Disable SLCD
    SLCDC_Disable();
    
    // Define the number of COM and SEG, Buffer driving time; select the bias
    SLCDC_Configure(S7LEKLCD_NUM_COMMONS,
                    S7LEKLCD_NUM_SEGMENTS,
                    AT91C_SLCDC_BIAS_1_3,
                    AT91C_SLCDC_BUFTIME_2_Tsclk);
    
    // Set frame rate 30 Hz
    SLCDC_SetFrameFreq(AT91C_SLCDC_PRESC_SCLK_32, AT91C_SLCDC_DIV_1);

    // Set normal mode
    AT91C_BASE_SLCDC->SLCDC_DR = 0;

    // Enable SLCD
    SLCDC_Enable();

    // Configure SLCDC interrupt
    IRQ_ConfigureIT(AT91C_ID_SLCD, 7, ISR_Slcdc);
    IRQ_EnableIT(AT91C_ID_SLCD);
    SLCDC_EnableInterrupts(AT91C_SLCDC_ENDFRAME);
}
