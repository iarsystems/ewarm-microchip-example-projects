void SLCD_SetBattery(unsigned char level)
{
    TRACE_DEBUG("SLCD_SetBattery()\n\r");
    SANITY_CHECK(level <= 4);

    // If a blinkmask is set, switch to normal mode first
    if (blinkmask > 0) {

        AT91C_BASE_SLCDC->SLCDC_DR = 0;
        SLCDC_EnableInterrupts(AT91C_SLCDC_ENDFRAME);
        while (endframe < 2);
        SLCDC_DisableInterrupts(AT91C_SLCDC_ENDFRAME);
    }

    // Display battery
    S7LEKLCD_Symbol(S7LEKLCD_Q0, 1);
    S7LEKLCD_Symbol(S7LEKLCD_Q1, level >= 1);
    S7LEKLCD_Symbol(S7LEKLCD_Q2, level >= 2);
    S7LEKLCD_Symbol(S7LEKLCD_Q3, level >= 3);
    S7LEKLCD_Symbol(S7LEKLCD_Q4, level >= 4);

    // Display blinked string
    if (blinkmask) {

        SLCD_SetBlinking(blinkmask);
    }
}
