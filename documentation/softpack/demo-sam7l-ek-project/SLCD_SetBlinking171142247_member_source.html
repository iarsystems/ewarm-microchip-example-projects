void SLCD_SetBlinking(unsigned int bitmask)
{
    TRACE_DEBUG("SLCD_SetBlinking()\n\r");
    SANITY_CHECK(bitmask < (1 << (S7LEKLCD_MAX_CHARS + 1)));

    // Save bitmask
    blinkmask = bitmask;

    // Display full string in normal mode first
    AT91C_BASE_SLCDC->SLCDC_DR = 0;
    endframe = 0;
    SLCDC_EnableInterrupts(AT91C_SLCDC_ENDFRAME);
    DisplayString(0);
    while (endframe < 2);
    //SLCDC_DisableInterrupts(AT91C_SLCDC_ENDFRAME);

    // Display blinked string in user buffer only load mode
    AT91C_BASE_SLCDC->SLCDC_DR = 6;
    endframe = 0;
    SLCDC_EnableInterrupts(AT91C_SLCDC_ENDFRAME);
    while (endframe < 1);
    //SLCDC_DisableInterrupts(AT91C_SLCDC_ENDFRAME);
    DisplayString(bitmask);

    // Switch to buffer swap mode
    AT91C_BASE_SLCDC->SLCDC_DR = 7 | (5 << 8);
}
