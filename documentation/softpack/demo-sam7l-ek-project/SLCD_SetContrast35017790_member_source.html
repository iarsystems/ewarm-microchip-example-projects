void SLCD_SetContrast(signed char contrast)
{
    const unsigned char lcdOut[5] = {0, 3, 6, 9, 15};
    unsigned int mode;

    mode = AT91C_BASE_SYS->SUPC_MR;
    mode &= ~AT91C_SUPC_LCDOUT;
    mode |= lcdOut[contrast] | (0x3 << 4) | (0xA5 << 24);
    AT91C_BASE_SYS->SUPC_MR = mode;
}
