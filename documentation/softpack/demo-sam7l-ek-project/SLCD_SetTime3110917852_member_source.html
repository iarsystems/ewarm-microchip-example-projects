void SLCD_SetTime(unsigned char hour, unsigned char minute, unsigned char dot)
{
    static unsigned short pNumbers[] = {
        S7LEKLCD_C0,
        S7LEKLCD_C1, 
        S7LEKLCD_C2,
        S7LEKLCD_C3, 
        S7LEKLCD_C4,
        S7LEKLCD_C5,
        S7LEKLCD_C6,
        S7LEKLCD_C7,
        S7LEKLCD_C8,
        S7LEKLCD_C9};

    TRACE_DEBUG("SLCD_SetTime()\n\r");
    SANITY_CHECK(hour < 24);
    SANITY_CHECK(minute < 60);

    // If a blinkmask is set, switch to normal mode first
    if (blinkmask > 0) {

        AT91C_BASE_SLCDC->SLCDC_DR = 0;
        SLCDC_EnableInterrupts(AT91C_SLCDC_ENDFRAME);
        while (endframe < 2);
        SLCDC_DisableInterrupts(AT91C_SLCDC_ENDFRAME);
    }

    // Display clock
    S7LEKLCD_ClockNumber(pNumbers[hour / 10], 0);
    S7LEKLCD_ClockNumber(pNumbers[hour % 10], 1);
    S7LEKLCD_ClockNumber(pNumbers[minute / 10], 2);
    S7LEKLCD_ClockNumber(pNumbers[minute % 10], 3);
    S7LEKLCD_Symbol(S7LEKLCD_COL4, dot);

    // Display blinked string
    if (blinkmask) {

        SLCD_SetBlinking(blinkmask);
    }
}
