void SLCD_SetVolume(unsigned char bars)
{
    static unsigned short symbols[5] = {S7LEKLCD_V0, S7LEKLCD_V1, S7LEKLCD_V2, S7LEKLCD_V3, S7LEKLCD_V4};
    unsigned int i;

    TRACE_DEBUG("SLCD_SetVolume()\n\r");
    SANITY_CHECK(bars <= 5);

    // If a blinkmask is set, switch to normal mode first
    if (blinkmask > 0) {

        AT91C_BASE_SLCDC->SLCDC_DR = 0;
        SLCDC_EnableInterrupts(AT91C_SLCDC_ENDFRAME);
        while (endframe < 2);
        SLCDC_DisableInterrupts(AT91C_SLCDC_ENDFRAME);
    }

    // Show/hide bars
    for (i=0; i < 5; i++) {

        S7LEKLCD_Symbol(symbols[i], bars > i);
    }

    // Display blinked string
    if (blinkmask) {

        SLCD_SetBlinking(blinkmask);
    }
}
