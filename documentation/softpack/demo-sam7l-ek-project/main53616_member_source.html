int main(void)
{
    unsigned char second;

    PIO_Configure(pins, PIO_LISTSIZE(pins));
    TRACE_CONFIGURE(DBGU_STANDARD, 115200, BOARD_MCK);

    // Back from a backup mode
    if (gBackup) {

        // Deep backup mode
        if ((gLowLevelInitSupcStatus & AT91C_SUPC_FWUPS) == AT91C_SUPC_FWUPS) {

            gInactivity = 0;
        }
        // Normal backup mode
        else {
            
            printf("-I- Restarted from backup mode.\n\r");
        }
    }
    // First start
    else {

        printf("-- Demo SAM7L-EK Project %s --\n\r", SOFTPACK_VERSION);
        printf("-- %s\n\r", BOARD_NAME);
        printf("-- Compiled: %s %s --\n\r", __DATE__, __TIME__);
        printf("-I- Fresh start\n\r");

        // Configure SUPC
        AT91C_BASE_SUPC->SUPC_MR |= ((0xA5 << 24)&AT91C_SUPC_MKEY) 
                                 | AT91C_SUPC_SRAMON | AT91C_SUPC_RTCON;
        AT91C_BASE_SUPC->SUPC_WUIR = 0x0000001F;    
    }

       // Initialize SLCD
    SLCD_Initialize();
    SLCD_SetContrast(gContrast);

    // Configure RTC
    IRQ_ConfigureIT(AT91C_ID_SYS, 0, ISR_Rtc);
    IRQ_EnableIT(AT91C_ID_SYS);
    AT91C_BASE_RTC->RTC_IER = AT91C_RTC_ALARM | AT91C_RTC_SECEV;

    // Refresh starting mode
    if (gCurrentMode == 0) {

        gCurrentMode = MODE_Default;
           ((ModeFunction) gCurrentMode)(-1, MODE_REFRESH_RESTART);
       }
       else {

        ((ModeFunction) gCurrentMode)(-1, MODE_REFRESH_DISPLAY);
    }

    // Configure BOD
    BATT_Configure();
    BATT_CheckLevel();

    // Initialize keyboard
    KBD_Initialize(KeyPressed);

    // Main loop
    while (1) {

        // If no activity is detected, enter backup mode
        if (KBD_IsIdle()) {

            // Configure RTC to wake-up in next second
               RTC_GetTime(0, 0, &second);
            second = (second + 1) % 60;
            RTC_SetTimeAlarm(0, 0, &second);

            // Enter backup mode
            printf("-I- Entering backup mode ...\n\r");
            gBackup = 1;
            KBD_ConfigureWakeUp();
            AT91C_BASE_SUPC->SUPC_WUMR |= AT91C_SUPC_RTCEN;
            AT91C_BASE_SUPC->SUPC_WUMR &= ~AT91C_SUPC_FWUPEN;
            AT91C_BASE_SUPC->SUPC_CR = (0xA5 << 24) | AT91C_SUPC_VROFF;
            while (1);
        }
    }

    return 0;
}
