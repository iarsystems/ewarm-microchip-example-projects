void RTT_IrqHandler()
{
    static unsigned int currentImage = 0;
    volatile unsigned int status;
    unsigned char *tmp;

    status = RTT_GetStatus(AT91C_BASE_RTTC);

    // Check if RTT has been incremented
    if ((status & AT91C_RTTC_RTTINC) != 0) {

        // If there are no images to display, return
        if (numImages == 0) {

            return;
        }

        // Display back buffer (previously decoded image is inside it)
        LCD_SetFrameBufferAddress(backBuffer);
        tmp = backBuffer;
        backBuffer = primaryBuffer;
        primaryBuffer = tmp;

        // Change current image
        currentImage++;
        if (currentImage >= numImages) {

            currentImage -= numImages;
        }

        // Decode new image
        BMP_Decode((void *) images[currentImage],
                   backBuffer,
                   //BOARD_LCD_WIDTH,
                   240, //use a fixed width to meet pic resolution under project dir
                   //BOARD_LCD_HEIGHT,
                   320, //use a fixed width to meet pic resolution under project dir
                   24);
    }
}
