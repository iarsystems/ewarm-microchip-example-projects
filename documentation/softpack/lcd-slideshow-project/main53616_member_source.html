int main()
{   
    unsigned int i;

    TRACE_CONFIGURE(DBGU_STANDARD, 115200, BOARD_MCK);
    printf("-- LCD Slideshow Project %s --\n\r", SOFTPACK_VERSION);
    printf("-- %s\n\r", BOARD_NAME);
    printf("-- Compiled: %s %s --\n\r", __DATE__, __TIME__);

    // If they are present, configure Vbus & Wake-up pins
    PIO_InitializeInterrupts(0);

    // Configure pins
    ConfigurePins();

#if !defined(sdram) && defined(PINS_SDRAM)
    // Initialize SDRAM
    BOARD_ConfigureSdram(BOARD_SDRAM_BUSWIDTH);
#endif
    
#if !defined(ddram) && defined(PINS_DDRAM)
    //configure DDRAM for use
    BOARD_ConfigureDdram(0, BOARD_DDRAM_BUSWIDTH);
#endif

    // Initialize LCD
    memset(primaryBuffer, 0, BOARD_LCD_FRAMESIZE * 4);
    memset(backBuffer, 0, BOARD_LCD_FRAMESIZE * 4);
    InitializeLcd();

    // Reset images
    numImages = 0;
    for (i=0; i < MAXIMAGES; i++) {

        images[i] = 0;
        imageSizes[i] = 0;
    }
    images[0] = FIRSTIMAGEADDRESS;

    // Configure RTT for a 2s tick
    RTT_SetPrescaler(AT91C_BASE_RTTC, 0);
    i = RTT_GetTime(AT91C_BASE_RTTC);
    while(i == RTT_GetTime(AT91C_BASE_RTTC));
    IRQ_ConfigureIT(AT91C_ID_SYS, 0, RTT_IrqHandler);
    IRQ_EnableIT(AT91C_ID_SYS);
    RTT_EnableIT(AT91C_BASE_RTTC, AT91C_RTTC_RTTINCIEN);

    // Initialize the serial driver
    CDCDSerialDriver_Initialize();

    // connect if needed
    VBUS_CONFIGURE();

    while (USBD_GetState() < USBD_STATE_CONFIGURED);

    // Infinite loop
    while (1) {

        // Start receiving data on the USB
        if (numImages < MAXIMAGES) {
        
            ReceiveBmpHeader(numImages);
        }
    }
}
