00047  * \brief  Get the power of input, given a certain result, i.e. input^(power) = result.
00048  *
00049  * \param result  a certain output we want to calculate.

00055 {
00056     uint32_t i=0;
00057 
00058     while(i<32)
00059     {
00060         if(result == (input << i))
00061             return i;
00062         i++;
00063     }
00064 
00065     return 0;
00066 }

00192 /**
00193  * \brief Translates address/size access of a NandFlashModel to block, page and offset values.
00194  *
00195  * \param modelList  List of NandFlashModel instances.
00196  * \param address  Access address.
00197  * \param size  Access size in bytes.
00198  * \param block  Stores the first accessed block number.
00199  * \param page  Stores the first accessed page number inside the first block.
00200  * \param offset  Stores the byte offset inside the first accessed page.
00201  * \return 0 if the access is correct; otherwise returns NandCommon_ERROR_OUTOFBOUNDS.
00202  * \note The values are stored in the provided variables if their pointer is not 0.
00203  */
00204 uint8_t NandFlashModel_TranslateAccess(
00205     const struct NandFlashModel *model,
00206     uint32_t address,
00207     uint32_t size,
00208     uint16_t *block,
00209     uint16_t *page,
00210     uint16_t *offset)
00211 {
00212      /* Check that access is not too big */
00213     #if !defined(OP_BOOTSTRAP_on)
00214     if ((address + size) > NandFlashModel_GetDeviceSizeInBytes(model)) {
00215 
00216         TRACE_DEBUG("NandFlashModel_TranslateAccess: out-of-bounds access.\n\r");
00217         return NandCommon_ERROR_OUTOFBOUNDS;
00218     }
00219     #endif
00220 
00221     /* Get Nand info */
00222     uint32_t blockSize = NandFlashModel_GetBlockSizeInBytes(model);
00223     uint32_t pageSize = NandFlashModel_GetPageDataSize(model);
00224 
00225     /* Translate address*/
00226     #if !defined(OP_BOOTSTRAP_on)
00227     uint16_t tmpBlock = address / blockSize;
00228     address -= tmpBlock * blockSize;
00229     uint16_t tmpPage = address / pageSize;
00230     address -= tmpPage * pageSize;
00231     uint16_t tmpOffset = address;
00232     #else
00233     uint16_t tmpBlock = CALINT(address, blockSize);
00234     address -= tmpBlock * blockSize;
00235     uint16_t tmpPage = CALINT(address, pageSize);
00236     address -= tmpPage * pageSize;
00237     uint16_t tmpOffset= address;
00238     #endif
00239 
00240     /* Save results */
00241     if (block) {
00242 
00243         *block = tmpBlock;
00244     }
00245     if (page) {
00246 
00247         *page = tmpPage;
00248     }
00249     if (offset) {
00250 
00251         *offset = tmpOffset;
00252     }
