00284 /**
00285 *   Calculates the CRC-8-CCITT.
00286 *
00287 *   CRC-8-CCITT is defined to be x^8 + x^2 + x + 1
00288 *
00289 *   To use this function use the following template:
00290 *
00291 *       crc = Crc8( crc, data );
00292 */
00293 #define POLYNOMIAL    (ISP_CFG1_CRC8_POLYNOMIAL << 7)
00294 static U8 Crc8(U8 inCrc, U8 inData)
00295 {
00296     int i;
00297     U16 data;
00298 
00299     data = inCrc ^ inData;
00300     data <<= 8;
00301 
00302     for (i = 0; i < 8; i++) {
00303         if ((data & 0x8000) != 0) {
00304             data = data ^ POLYNOMIAL;
00305         }
00306         data = data << 1;
00307     }
00308     return (unsigned char)(data >> 8);
00309 }

00312 void isp_force_isp(bool force)
00313 {
00314     uint32_t tempo;
00315     U8 crc8 = 0;
00316     int i;
00317 
00318     // 1) Read the config word1 and set the ISP_FORCE bit to force.
00319     tempo = (ISP_CFG1 & ~ISP_CFG1_FORCE_MASK)
00320         | ((force << ISP_CFG1_FORCE_OFFSET) & ISP_CFG1_FORCE_MASK);
00321 
00322     // 2) Compute the CRC8 and update the config word1
00323     for (i = 24; i; i -= 8) // Compute the CRC8 on the 3 upper Bytes of the word.
00324         crc8 = Crc8(crc8, tempo >> i);
00325     tempo = (tempo & ~ISP_CFG1_CRC8_MASK)
00326         | ((crc8 << ISP_CFG1_CRC8_OFFSET) & ISP_CFG1_CRC8_MASK);
00327 
00328     // 3) Write the config word1
00329     mem_user_write(ISP_CFG1_OFFSET, &tempo, 4);
00330 }
