00152     LowLevelInit() ;

00154     /* Initialize the relocate segment */
00155     pSrc = &_etext ;
00156     pDest = &_srelocate ;
00157 
00158     if ( pSrc != pDest )
00159     {
00160         for ( ; pDest < &_erelocate ; )
00161         {
00162             *pDest++ = *pSrc++ ;
00163         }
00164     }
00165 
00166     /* Clear the zero segment */
00167     for ( pDest = &_szero ; pDest < &_ezero ; )
00168     {
00169         *pDest++ = 0;

00171 
00172     /* Set the vector table base address */

00174     SCB->VTOR = ( (uint32_t)pSrc & SCB_VTOR_TBLOFF_Msk ) ;
00175

00177     {
00178         SCB->VTOR |= 1 << SCB_VTOR_TBLBASE_Pos ;
00179     }
