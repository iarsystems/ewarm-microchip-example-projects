00058 /**
00059  * \brief Performs the low-level initialization of the chip.
00060  * This includes EFC and master clock configuration.
00061  * It also enable a low level on the pin NRST triggers a user reset.
00062  */
00063 extern WEAK void LowLevelInit( void )
00064 {
00065     uint32_t dwTimeout;
00066 
00067     /* Set 3 WS for Embedded Flash Access for 84 MHz */
00068     EFC0->EEFC_FMR = EEFC_FMR_FWS( 3 );
00069     EFC1->EEFC_FMR = EEFC_FMR_FWS( 3 );
00070 
00071     /* Initialize main oscillator */
00072     if ( !(PMC->CKGR_MOR & CKGR_MOR_MOSCSEL) )  /* Main Oscillator Selection */
00073     {
00074         /* The Main Crystal Oscillator is enabled */
00075         PMC->CKGR_MOR = CKGR_MOR_KEY(0x37)
00076                       | CKGR_MOR_MOSCXTST(0x8) /* Main Crystal Oscillator Start-up Time: 1.4ms(datasheet) */
00077                       /* Specifies the number of Slow Clock cycles multiplied by 8 for the Main Crystal Oscillator start-up time */
00078                       | CKGR_MOR_MOSCRCEN      /* Main On-Chip RC Oscillator Enable */
00079                       | CKGR_MOR_MOSCXTEN;     /* Main Crystal Oscillator Enable */
00080                       /* MOSCSEL: The Main On-Chip RC Oscillator is selected */
00081                       /* MOSCRCF: The Fast RC Oscillator Frequency is at 4 MHz (default) */
00082                       /* CFDEN:  The Clock Failure Detector is disabled. */
00083         dwTimeout = 0;
00084         while ( !(PMC->PMC_SR & PMC_SR_MOSCXTS) && (dwTimeout++ < CLOCK_TIMEOUT) );
00085     }
00086 
00087     /* Switch to 3-20MHz Xtal oscillator */
00088     /* The Main Crystal Oscillator is enabled */
00089     PMC->CKGR_MOR = CKGR_MOR_KEY(0x37)
00090                   | CKGR_MOR_MOSCXTST(0x8)  /* Main Crystal Oscillator Start-up Time: 1.4ms(datasheet) */
00091                   /* Specifies the number of Slow Clock cycles multiplied by 8 for the Main Crystal Oscillator start-up time */
00092                   | CKGR_MOR_MOSCRCEN       /* Main On-Chip RC Oscillator Enable */
00093                   | CKGR_MOR_MOSCXTEN       /* Main Crystal Oscillator Enable */
00094                   | CKGR_MOR_MOSCSEL;       /* The Main Crystal Oscillator is selected */
00095                   /* MOSCRCF: The Fast RC Oscillator Frequency is at 4 MHz (default) */
00096                   /* CFDEN:  The Clock Failure Detector is disabled. */
00097     dwTimeout = 0;
00098     /* Wait Main XTAL Oscillator Status */
00099     while (!(PMC->PMC_SR & PMC_SR_MOSCXTS) && (dwTimeout++ < CLOCK_TIMEOUT));
00100 
00101     /* configure PLLA to 168 MHz */
00102     PMC->CKGR_PLLAR = CKGR_PLLAR_STUCKTO1
00103                     | CKGR_PLLAR_MULA(13)     /* PLLA Multiplier 12 MHz x (13+1) = 168 MHz */
00104                     | CKGR_PLLAR_PLLACOUNT(2) /* PLLA Counter 200µs(datasheet) */
00105                     | CKGR_PLLAR_DIVA(1);     /* Divider bypassed */
00106     dwTimeout = 0;
00107     /* Wait PLL A Lock Status */
00108     while (!(PMC->PMC_SR & PMC_SR_LOCKA) && (dwTimeout++ < CLOCK_TIMEOUT));
00109 
00110     PMC->PMC_MCKR = PMC_MCKR_PRES_CLK_2        /* Selected clock divided by 2 => 168/2 = 84 MHz  */
00111                   | PMC_MCKR_CSS_MAIN_CLK;     /* Main Clock is selected */
00112     dwTimeout = 0;
00113     /* Wait Master Clock Status */
00114     while (!(PMC->PMC_SR & PMC_SR_MCKRDY) && (dwTimeout++ < CLOCK_TIMEOUT));
00115 
00116     PMC->PMC_MCKR = PMC_MCKR_PRES_CLK_2        /* Selected clock divided by 2 => 168/2 = 84 MHz  */
00117                   | PMC_MCKR_CSS_PLLA_CLK;   /* PLLA Clock is selected */
00118     dwTimeout = 0;
00119     /* Wait Master Clock Status */
00120     while (!(PMC->PMC_SR & PMC_SR_MCKRDY) && (dwTimeout++ < CLOCK_TIMEOUT));
