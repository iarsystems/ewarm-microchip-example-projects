00135 /**
00136  * \brief An accurate one-to-one comparison is necessary between PSRAM and SMC waveforms for
00137  *   a complete SMC configuration.
00138  *  \note The system is running at 48 MHz for the EBI Bus.
00139  *        Please refer to the "AC Characteristics" section of the customer product datasheet.
00140  */
00141 extern void BOARD_ConfigurePSRAM( Smc* pSmc )
00142 {
00143      /* Enable peripheral clock */
00144     PMC_EnablePeripheral( ID_SMC ) ;
00145 
00146     /* Configure SMC, NCS1 is assigned to a external PSRAM */
00147     /**
00148      * PSRAM IS66WV51216BLL
00149      * 55 ns Access time
00150      * tdoe = 25 ns max
00151      * SMC1 (timing SAM3X read mode SMC) = 21 ns of setup
00152      * 21 + 55 = 76 ns => at least 5 cycles at 64 MHz
00153      * Write pulse width minimum = 45 ns (PSRAM)
00154      */
00155     pSmc->SMC_CS_NUMBER[1].SMC_SETUP = SMC_SETUP_NWE_SETUP( 1 )
00156                                      | SMC_SETUP_NCS_WR_SETUP( 0 )
00157                                      | SMC_SETUP_NRD_SETUP( 2 )
00158                                      | SMC_SETUP_NCS_RD_SETUP( 0 ) ;
00159 
00160     pSmc->SMC_CS_NUMBER[1].SMC_PULSE = SMC_PULSE_NWE_PULSE( 3 )
00161                                      | SMC_PULSE_NCS_WR_PULSE( 4 )
00162                                      | SMC_PULSE_NRD_PULSE( 3 )
00163                                      | SMC_PULSE_NCS_RD_PULSE( 5 ) ;
00164 
00165     /* NWE_CYCLE:     The total duration of the write cycle.
00166                       NWE_CYCLE = NWE_SETUP + NWE_PULSE + NWE_HOLD
00167                       = NCS_WR_SETUP + NCS_WR_PULSE + NCS_WR_HOLD
00168                       (tWC) Write Cycle Time min. 70ns
00169        NRD_CYCLE:     The total duration of the read cycle.
00170                       NRD_CYCLE = NRD_SETUP + NRD_PULSE + NRD_HOLD
00171                       = NCS_RD_SETUP + NCS_RD_PULSE + NCS_RD_HOLD
00172                       (tRC) Read Cycle Time min. 70ns. */
00173     pSmc->SMC_CS_NUMBER[1].SMC_CYCLE = SMC_CYCLE_NWE_CYCLE( 4 )
00174                                      | SMC_CYCLE_NRD_CYCLE( 5 ) ;
00175  
00176     pSmc->SMC_CS_NUMBER[1].SMC_MODE  = SMC_MODE_READ_MODE
00177                                      | SMC_MODE_WRITE_MODE
00178                                      | SMC_MODE_DBW_BIT_16 ;
00179 }
