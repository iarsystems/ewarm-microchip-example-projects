00150 //!
00151 //! @brief Looks for mouse events.
00152 //!
00153 Bool is_usb_mouse_event( void )
00154 {
00155    static U8   disp=1;
00156    static U16  count=0;
00157    Bool b_activity;
00158    static Bool old_click_2=FALSE;  // FALSE means released, TRUE means pushed
00159 
00160    u8_report_buttons=0;
00161    u8_report_disp_x=0;
00162    u8_report_disp_y=0;
00163    u8_report_disp_wheel=0;
00164    b_activity=FALSE;
00165 
00166 //   signed int res;
00167 
00168   /* Mouse pointer down */
00169   if (PIO_PinGet(&pinsButtons[PUSHBUTTON_BP1]) == 0)
00170   {
00171     u8_report_disp_y=disp, b_activity=TRUE;
00172   }
00173   /* Mouse pointer up */
00174   if (PIO_PinGet(&pinsButtons[PUSHBUTTON_BP2]) == 0)
00175   {
00176     u8_report_disp_y=-disp, b_activity=TRUE;
00177   }
00178 
00179    if( b_activity )
00180    {
00181       count++;
00182       if( count >= DISP_LIMIT_HIGH )
00183          disp=3;
00184       else if( count >= DISP_LIMIT_LOW )
00185          disp=2;
00186    }
00187    else
00188       count=0, disp=1;
00189 
00190 
00191 
00192    // Look for button activity
00193   if (PIO_PinGet(&pinsButtons[PUSHBUTTON_BP2]) == 0)
00194    {
00195       Set_bits(u8_report_buttons, BUTTON_2_EVENT_PUSH);
00196       if( old_click_2 == FALSE )
00197       {
00198          old_click_2 = TRUE;
00199          b_activity = TRUE;
00200       }
00201    }
00202    else
00203    {
00204       if( old_click_2 == TRUE )
00205       {
00206          old_click_2 = FALSE;
00207          b_activity = TRUE;
00208       }
00209    }
00210 
00211    if( b_activity )  return TRUE;
00212    else              return FALSE;
00213 }
