00113 {
00114   sof_cnt = 0;
00115   data_length = 0;
00116 #ifndef FREERTOS_USED
00117   #if USB_HOST_FEATURE == ENABLED
00118   // If both device and host features are enabled, check if device mode is engaged

00135 //!
00136 //! @brief Entry point of the device applicative task management
00137 //!
00138 //! This function links the device application to the USB bus.
00139 //!
00140 #ifdef FREERTOS_USED
00141 void device_template_task(void *pvParameters)
00142 #else
00143 void device_template_task(void)
00144 #endif
00145 {
00146   static U8 buf[EP_SIZE_TEMP2];
00147 
00148 #ifdef FREERTOS_USED
00149   portTickType xLastWakeTime;
00150 
00151   xLastWakeTime = xTaskGetTickCount();
00152   while (TRUE)
00153   {
00154     vTaskDelayUntil(&xLastWakeTime, configTSK_USB_DTP_PERIOD);
00155 
00156     // First, check the device enumeration state
00157     if (!Is_device_enumerated()) continue;
00158 #else
00159     // First, check the device enumeration state
00160     if (!Is_device_enumerated()) return;
00161 #endif  // FREERTOS_USED
00162 
00163     // HERE STARTS THE USB DEVICE APPLICATIVE CODE
00164     // The example below just performs a loopback transmission/reception.
00165     // All data received with the OUT endpoint is stored in a RAM buffer and
00166     // sent back to the IN endpoint.
00167 
00168 #if BOARD == EVK1100
00169     // For example, display Start-of-Frame counter on LEDs
00170     LED_Display_Field(LED_MONO0_GREEN |
00171                       LED_MONO1_GREEN |
00172                       LED_MONO2_GREEN |
00173                       LED_MONO3_GREEN,
00174                       sof_cnt >> 5);
00175 #elif BOARD == EVK1101 || BOARD == UC3C_EK || BOARD == EVK1104 || BOARD == EVK1105
00176     // For example, display Start-of-Frame counter on LEDs
00177     LED_Display_Field(LED0 |
00178                       LED1,
00179                       sof_cnt >> 5);
00180 #else
00181   #warning The display of the SOFs must be defined here.
00182 #endif
00183 
00184     // If we receive something in the OUT endpoint, just store it in the RAM buffer
00185     if (Is_usb_out_received(EP_TEMP_OUT))
00186     {
00187 #if BOARD != 99
00188         LED_On(LED_APPLI_1);
00189 #endif
00190       Usb_reset_endpoint_fifo_access(EP_TEMP_OUT);
00191       data_length = Usb_byte_count(EP_TEMP_OUT);
00192       usb_read_ep_rxpacket(EP_TEMP_OUT, buf, data_length, NULL);
00193       Usb_ack_out_received_free(EP_TEMP_OUT);
00194 #if BOARD != 99
00195       LED_Off(LED_APPLI_1);
00196 #endif
00197     }
00198 
00199     // Load the IN endpoint with the contents of the RAM buffer
00200     if (data_length && Is_usb_in_ready(EP_TEMP_IN))
00201     {
00202 #if BOARD != 99
00203       LED_On(LED_APPLI_0);
00204 #endif
00205       Usb_reset_endpoint_fifo_access(EP_TEMP_IN);
00206       usb_write_ep_txpacket(EP_TEMP_IN, buf, data_length, NULL);
00207       data_length = 0;
00208       Usb_ack_in_ready_send(EP_TEMP_IN);
00209 #if BOARD != 99
00210       LED_Off(LED_APPLI_0);
00211 #endif
00212     }
00213 #ifdef FREERTOS_USED
00214   }
00215 #endif
00216 }
