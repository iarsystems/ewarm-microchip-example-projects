00030 /** \addtogroup dmad_module

00032  * \section DmaConfig Dma Configuration Usage
00033  *
00034  * To configure a DMA channel, the user has to follow these few steps :
00035  * <ul>
00036  * <li> Initialize a DMA driver instance by DMAD_Initialize().</li>
00037  * <li> choose an available (disabled) channel using DMAD_AllocateChannel().</li>
00038  * <li> After the DMAC selected channel has been programmed, DMAD_PrepareChannel() is to enable 
00039  * clock and dma peripheral of the DMA, and set Configuration register to set up the transfer type 
00040  * (memory or non-memory peripheral for source and destination) and flow control device.</li>
00041  * <li> Configure DMA multi-buffer transfers using DMAD_PrepareMultiTransfer() to set up the chain of Linked List Items,
00042  * single-buffer transfers using DMAD_PrepareSingleTransfer().</li>
00043  * <li> Invoke DMAD_StartTransfer() to start DMA transfer, or DMAD_StopTransfer() to force stop DMA transfer.</li>
00044  * <li> If picture-in-picture mode is enabled, DMAD_ConfigurePIP() helps to configure PIP mode.</li>
00045  * <li> Once the buffer of data is transferred, DMAD_IsTransferDone() checks if DMA transfer is finished.</li>
00046  * <li> DMAD_Handler() handles DMA interrupt, and invoking DMAD_SetCallback() if provided.</li>
00047  * </ul>
00048  *
00049  * Related files:\n
00050  * \ref dmad.h\n
00051  * \ref dmad.c.\n

00072  * \param pDmad  Pointer to DMA driver instance.
00073  * \param ucDmac  DMA controller ID.

00080                                           uint8_t ucDmac,
00081                                           uint8_t ucSrcID, 
00082                                           uint8_t ucDstID)
00083 {
00084     uint32_t i;
00085 
00086     /* Can't support peripheral to peripheral */
00087     if ((( ucSrcID != DMA_TRANSFER_MEMORY ) && ( ucDstID != DMA_TRANSFER_MEMORY ))) 
00088     {
00089         return DMA_ALLOC_FAILED;
00090     }
00091     /* dma transfer from peripheral to memory */

00093     {
00094         if( (!DMAIF_IsValidatedPeripherOnDma(ucDmac, ucSrcID)) )

00096             return DMA_ALLOC_FAILED;
00097         }
00098     }
00099     /* dma transfer from memory to peripheral */
00100     if ( ucSrcID == DMA_TRANSFER_MEMORY )
00101     {
00102         if( (!DMAIF_IsValidatedPeripherOnDma(ucDmac, ucDstID)) )
00103         {
00104             return DMA_ALLOC_FAILED;
00105         }
00106     }
00107     for (i = 0; i < pDmad->ucNumChannels; i ++)
00108     {

00113             /* Get general informations */

00123                 DMAIF_GetChannelNumber(ucDmac, ucDstID, 1);

00211             /* Chained buffer complete */

00217                 
00218             }

00225                 {
00226                     DMAC_EnableChannel( pDmac, ucChannel );

00228             }

00230             {
00231                 bExec = 0;
