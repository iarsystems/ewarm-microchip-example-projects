00239     {
00240         dwClk = EMAC_NCFGR_CLK_MCK_32;
00241     }
00242     else if (dwMck > 20*1000*1000)
00243     {
00244         dwClk = EMAC_NCFGR_CLK_MCK_16;
00245     }
00246     else
00247     {
00248         dwClk = EMAC_NCFGR_CLK_MCK_8;
00249     }
00250 
00251     pEmac->EMAC_NCFGR &= ~EMAC_NCFGR_CLK_Msk;
00252     pEmac->EMAC_NCFGR |= dwClk;
00253 
00254     return 1;
00255 }

00442 /**
00443  * Execute PHY maintanance command
00444  */
00445 void EMAC_PHYMaintain(Emac      *pEmac,
00446                       uint8_t   bPhyAddr,
00447                       uint8_t   bRegAddr,
00448                       uint8_t   bRW,
00449                       uint16_t  wData)
00450 {
00451     /* Wait until bus idle */
00452     while((pEmac->EMAC_NSR & EMAC_NSR_IDLE) == 0);
00453     /* Write maintain register */
00454     pEmac->EMAC_MAN = EMAC_MAN_CODE(10) | EMAC_MAN_SOF(0x1)
00455                     | EMAC_MAN_PHYA(bPhyAddr)
00456                     | EMAC_MAN_REGA(bRegAddr)
00457                     | EMAC_MAN_RW((bRW ? 0x2 : 0x1))
00458                     | EMAC_MAN_DATA(wData)
00459                     ;
00460 }
