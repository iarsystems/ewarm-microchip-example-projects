00030 /**
00031  *  \page adc12 ADC12 Example
00032  *
00033  *  \section Purpose
00034  *
00035  *  The adc12 example demonstrates how to use ADC peripheral with several modes.
00036  *
00037  *  \section Requirements
00038  *
00039  *  This package can be used with SAM3X-EK. To enable full scale measurement
00040  *  of the potentiometer, jumper JP4 has to be closed.
00041  *
00042  *  We use "LEFT CLICK" button for ADTGR, so please connect PA11 (ADTRG) to
00043  *  PE7 (LEFT CLICK) before running the example.
00044  *
00045  *  \section Description
00046  *
00047  *  This application shows how to use the ADC using the several modes:
00048  *  with/without PDC, several types of trigger (Software, ADTRG, Timer, etc.),
00049  *  gain and offset selection, using sequencer. User can select different mode
00050  *  by configuration menu in the terminal.
00051  *
00052  *  \note
00053  *  For early Engineering Samples, there is a known errata about adc:
00054  *  An ADC trigger (both software and hardware trigger) initiates a conversion
00055  *  sequence of one channel and not all activated channels as expected.
00056  *  So, we have to trigger as many as the number of activated channels.
00057  *  For example with ADTRG trigger, we need press "LEFT CLICK" two times to
00058  *  trigger two channels.
00059  *
00060  *  \section Usage
00061  *
00062  *  -# Build the program and download it inside the evaluation board. Please
00063  *     refer to the
00064  *     <a href="http://www.atmel.com/dyn/resources/prod_documents/doc6224.pdf">
00065  *     SAM-BA User Guide</a>, the
00066  *     <a href="http://www.atmel.com/dyn/resources/prod_documents/doc6310.pdf">
00067  *     GNU-Based Software Development</a>
00068  *     application note or to the
00069  *     <a href="ftp://ftp.iar.se/WWWfiles/arm/Guides/EWARM_UserGuide.ENU.pdf">
00070  *     IAR EWARM User Guide</a>,
00071  *     depending on your chosen solution.
00072  *  -# On the computer, open and configure a terminal application
00073  *     (e.g. HyperTerminal on Microsoft Windows) with these settings:
00074  *    - 115200 bauds
00075  *    - 8 bits of data
00076  *    - No parity
00077  *    - 1 stop bit
00078  *    - No flow control
00079  *  -# In the terminal window, the
00080  *     following text should appear (values depend on the board and chip used):
00081  *     \code
00082  *      -- ADC12 Example xxx --
00083  *      -- xxxxxx-xx
00084  *      -- Compiled: xxx xx xxxx xx:xx:xx --
00085  *      =========================================================
00086  *      Menu: press a key to change the configuration.
00087  *      ---------------------------------------------------------
00088  *      [X] 0: Set ADC trigger mode: Software.
00089  *      [ ] 1: Set ADC trigger mode: ADTRG.
00090  *      [ ] 2: Set ADC trigger mode: Timer TIOA.
00091  *      [ ] 3: Set ADC trigger mode: PWM Event Line.
00092  *      [ ] 4: Set ADC trigger mode: Free run mode.
00093  *      [E] T: Enable/Disable to tranfer with PDC.
00094  *      [D] S: Enable/Disable to use user sequence mode.
00095  *      [D] P: Enable/Disable ADC power save mode.
00096  *      [D] G: Enable/Disable to set gain=2 for potentiometer channel.
00097  *      [D] O: Enable/Disable offset for potentiometer channel.
00098  *          Q: Quit configuration and start ADC.
00099  *      =========================================================
00100  *     \endcode
00101  *  -# The application will output converted value to hyperterminal and display
00102  *     a menu for user to set different mode.
00103  *
00104  *  \section References
00105  *  - adc12/main.c
00106  *  - adc.h
00107  */

00413 /**
00414  * \brief (Re)Sart ADC sample.
00415  * Initialize ADC, set clock and timing, set ADC to given mode.
00416  */
00417 static void _StartAdc( void )
00418 {
00419     uint32_t dw ;
00420     uint32_t dwAdcClock ;
00421 
00422     /* Initialize ADC */
00423     ADC_Initialize( ADC, ID_ADC );
00424     memset( (void*)&gAdcSampleData, 0, sizeof( gAdcSampleData ) );
00425 
00426     /*
00427      * Formula: ADCClock = MCK / ( (PRESCAL+1) * 2 )
00428      * For example, MCK = 84MHZ, PRESCAL = 6, then:
00429      *     ADCClock = 84 / ((6+1) * 2) = 6MHz;
00430      */
00431     /* Set ADC clock */
00432     dwAdcClock = ADC_SetClock( ADC, 6, BOARD_MCK ) ;
00433     printf( "Set ADC clock to %d Hz\n\r", (int)dwAdcClock ) ;
00434 
00435     /* Formula:
00436      *     Startup  Time = startup value / ADCClock
00437      *     Transfer Time = (TRANSFER * 2 + 3) / ADCClock
00438      *     Tracking Time = (TRACKTIM + 1) / ADCClock
00439      *     Settling Time = settling value / ADCClock
00440      * For example, ADC clock = 6MHz (166.7 ns)
00441      *     Startup time = 512 / 6MHz = 85.3 us
00442      *     Transfer Time = (1 * 2 + 3) / 6MHz = 833.3 ns
00443      *     Tracking Time = (0 + 1) / 6MHz = 166.7 ns
00444      *     Settling Time = 3 / 6MHz = 500 ns
00445      */
00446     /* Set ADC timing */
00447     ADC_SetTiming( ADC, ADC_MR_STARTUP_SUT512, ADC_MR_TRANSFER( 1 ), ADC_MR_TRACKTIM( 0 ), ADC_MR_SETTLING_AST17 ) ;
00448 
00449     /* Enable channel number tag */
00450     ADC_SetTagEnable( ADC, 1 ) ;
00451 
00452     /* Enable/disable sequencer */
00453     if ( gAdcTestMode.ucSequenceEn )
00454     {
00455         /* Set user defined channel sequence */
00456         ADC_SetSequence( ADC, ADC_SEQR1_USCH1(ADC_CHANNEL_15) | ADC_SEQR1_USCH2(ADC_CHANNEL_1), 0 ) ;
00457         /* Enable sequencer */
00458         ADC_SetSequenceMode( ADC, 1 ) ;
00459 
00460         /* Enable channels */
00461         for ( dw = 0; dw < NUM_CHANNELS; dw++ )
00462         {
00463             ADC_EnableChannel( ADC, dw ) ;
00464         }
00465         /* Update channel number */
00466         gAdcSampleData.ucChNum[0] = ADC_CHANNEL_15 ;
00467         gAdcSampleData.ucChNum[1] = ADC_CHANNEL_1 ;
00468     }
00469     else
00470     {
00471         /* Disable sequencer */
00472         ADC_SetSequenceMode( ADC, 0 ) ;
00473 
00474         /* Enable  channels */
00475         ADC_EnableChannel( ADC, ADC_CHANNEL_1 ) ;
00476         ADC_EnableChannel( ADC, ADC_CHANNEL_15 ) ;
00477 
00478         /* Update channel number */
00479         gAdcSampleData.ucChNum[0] = ADC_CHANNEL_1 ;
00480         gAdcSampleData.ucChNum[1] = ADC_CHANNEL_15 ;
00481     }
00482     /* Enable the temperature sensor (CH15) */
00483     ADC_EnableTS( ADC ) ;
00484 
00485     /* Set gain and offset (only single ended mode used here) */
00486     ADC_SetAnalogChange( ADC, 0 ) ; /* Disable analog change */
00487     if ( gAdcTestMode.ucGainEn  )
00488     {
00489         ADC_SetAnalogChange( ADC, 1) ;
00490         ADC_SetChannelGain( ADC, ADC_CGR_GAIN1(2) ) ; /* gain = 2 */
00491     }
00492     else
00493     {
00494         ADC_SetChannelGain( ADC, 0 ) ; /* gain = 1 */
00495     }
00496 
00497     if ( gAdcTestMode.ucOffsetEn )
00498     {
00499         ADC_SetAnalogChange( ADC, 1 ) ;
00500         ADC_SetChannelOffset( ADC, ADC_COR_OFF1 ) ;
00501     }
00502     else
00503     {
00504         ADC_SetChannelOffset( ADC, 0 ) ;
00505     }
00506 
00507     /* Set power save */
00508     /* Note: SLEEP mode doesn't work in engineer samples. */
00509     if ( gAdcTestMode.ucPowerSaveEn )
00510     {
00511         ADC_SetSleepMode( ADC, 1 ) ;
00512     }
00513     else
00514     {
00515         ADC_SetSleepMode( ADC, 0 ) ;
00516     }
00517 
00518     /* Transfer with/without PDC */
00519     if ( gAdcTestMode.ucPdcEn )
00520     {
00521         ADC_ReadBuffer( ADC, gAdcSampleData.wValue, BUFFER_SIZE ) ;
00522         /* Enable PDC channel interrupt */
00523         ADC_EnableIt( ADC, ADC_IER_RXBUFF ) ;
00524     }
00525     else
00526     {
00527         /* Enable Data ready interrupt */
00528         ADC_EnableIt( ADC, ADC_IER_DRDY ) ;
00529     }
00530     /* Enable ADC interrupt */
00531     NVIC_EnableIRQ( ADC_IRQn ) ;
00532 
00533     /* Configure trigger mode and start convention */
00534     switch ( gAdcTestMode.ucTriggerMode )
00535     {
00536         case TRIGGER_MODE_SOFTWARE:
00537             ADC_SetTrigger( ADC, 0, 0 ) ; /* Disable hardware trigger */
00538         break ;
00539 
00540         case TRIGGER_MODE_ADTRG:
00541             PIO_PinConfigure( &gPinADTRG, 1 ) ;
00542             ADC_SetTrigger( ADC, 1, ADC_MR_TRGSEL_ADC_TRIG0 ) ;
00543         break ;
00544 
00545         case TRIGGER_MODE_TIMER :
00546             _ConfigureTimerTrigger() ;
00547         break ;
00548 
00549         case TRIGGER_MODE_PWM :
00550             _ConfigurePwmTrigger() ;
00551         break ;
00552 
00553         case TRIGGER_MODE_FREERUN :
00554             ADC_SetFreeRunMode( ADC, 1 ) ;
00555         break ;
00556 
00557         default :
00558         break ;
00559     }
00560 }

00578 {

00580     uint32_t dwStatus;
00581     uint32_t dwTemp;
00582     uint8_t  ucChNum;
00583 
00584     dwStatus = ADC_GetStatus(ADC);
00585 
00586     /* With PDC transfer */
00587     if ( gAdcTestMode.ucPdcEn )
00588     {
00589         if ( (dwStatus & ADC_ISR_RXBUFF) == ADC_ISR_RXBUFF )
00590         {
00591             gAdcSampleData.wDone = ADC_DONE_MASK;
00592             ADC_ReadBuffer( ADC, gAdcSampleData.wValue, BUFFER_SIZE ) ;
00593 
00594             /* Only keep sample value, discard channel number */
00595             for (i = 0; i < NUM_CHANNELS; i ++)
00596             {
00597                 gAdcSampleData.wValue[i] &= ADC_LCDR_LDATA_Msk;
00598             }
00599         }
00600     }
00601 
00602     /* Without PDC transfer */
00603     if ( !gAdcTestMode.ucPdcEn )
00604     {
00605         if ( (dwStatus & ADC_ISR_DRDY) == ADC_ISR_DRDY )
00606         {
00607             dwTemp = ADC_GetLastConvertedData( ADC );
00608 
00609             for (i = 0; i < NUM_CHANNELS; i ++)
00610             {
00611                 ucChNum = (dwTemp & ADC_LCDR_CHNB_Msk) >> ADC_LCDR_CHNB_Pos;
