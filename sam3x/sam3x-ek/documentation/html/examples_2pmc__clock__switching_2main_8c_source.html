00030 /**
00031  * \page pmc_clock_switching PMC Clock Switching Example.
00032  *
00033  * \section Purpose
00034  * This example shows how to switch from a clock to another (PLLA, UPLL, SLCK, MAINCK)
00035  * or change divider.
00036  *
00037  * \section Requirements
00038  *
00039  * This package can be used with SAM3X evaluation kits.
00040  *
00041  * \section Description
00042  *
00043  * Upon startup, the program configure pios for UART,PCK and botton. The baud rate of
00044  * UART is configured as 2400 bps. The application does some printf with the current
00045  * configuration (except 32KHz slow clock ) and waits for button pressed to switch to
00046  * next configuration. PCK Outputs can be selected from the clocks provided by the
00047  * clock (PLLA, UPLL, SLCK, MAINCK) and driven on the pin PCK0 on PB22.
00048  * After the clock switchs, the PCK output signal can be measured by scope compare with
00049  * the clock configuration.
00050  *
00051  * <ul>
00052  * <li> The Clock Generator integrates a 32,768 Hz low-power oscillator.
00053  * In order to use this oscillator, the XIN32 and XOUT32 pins must be connected
00054  * to a 32,768 Hz crystal. The user can select the crystal oscillator to be the source of
00055  * the slow clock, as it provides a more accurate frequency. The command is made by
00056  * function SUPC_SelectExtCrystal32K().</li>
00057  * <li> MAINCK is the output of the Main Clock Oscillator selection: either the Crystal
00058  * Oscillator or 4/8/12 MHz Fast RC Oscillator. The user can select the output frequency
00059  * of the Fast RC Oscillator: either 4 MHz, 8 MHz or 12 MHz are available. It can be done
00060  * through function PMC_EnableIntRC4_8_12MHz(). The user can also select the 3 to 20 MHz Crystal
00061  * oscillator to be the source of MAINCK, as it provides a more accurate frequency.
00062  * The function PMC_EnableExtOsc12MHz() show how to enable the oscillator. </li>
00063  * <li> The PMC features PLL Blocks that permit a wide range of frequencies to be
00064  * selected on either the master clock, the processor clock or the programmable clock outputs.
00065  * Function PMC_ConfigureMckWithPlla() descripts the deatil how to configure PLLA.</li>
00066  * </ul>
00067  * \section Usage
00068  *
00069  * -# Build the program and download it inside the evaluation board. Please
00070  *    refer to the
00071  *    <a href="http://www.atmel.com/dyn/resources/prod_documents/doc6224.pdf">
00072  *    SAM-BA User Guide</a>, the
00073  *    <a href="http://www.atmel.com/dyn/resources/prod_documents/doc6310.pdf">
00074  *    GNU-Based Software Development</a> application note or to the
00075  *    <a href="ftp://ftp.iar.se/WWWfiles/arm/Guides/EWARM_UserGuide.ENU.pdf">
00076  *    IAR EWARM User Guide</a>, depending on your chosen solution.
00077  * -# On the computer, open and configure a terminal application
00078  *    (e.g. HyperTerminal on Microsoft Windows) with these settings:
00079  *   - 2400 bauds
00080  *   - 8 bits of data
00081  *   - No parity
00082  *   - 1 stop bit
00083  *   - No flow control
00084  * -# Start the application.
00085  * -# In the terminal window, the following text should appear:
00086  *    \code
00087  *     -- Pmc clock Switching Example xxx --
00088  *     -- xxxxxx-xx
00089  *     -- Compiled: xxx xx xxxx xx:xx:xx --
00090  *    \endcode
00091  * -# Press the left push button to perform the corresponding action.
00092  *
00093  * \section References
00094  * - pmc_clock_switching/main.c
00095  */

00146  *  Configure the PIO as inputs and generate corresponding interrupt when

00149 static void _ConfigureButton( void )
00150 {
00151     /* Configure pio as inputs. */
00152     PIO_PinConfigure( &pinPB, 1 ) ;
00153     /* Adjust pio debounce filter patameters, uses 10 Hz filter. */

00155     /* Initialize pios interrupt handlers, see PIO definition in board.h. */

00157     /* Enable PIO controller IRQs. */

00159     /* Enable PIO line interrupts. */

00163 /**
00164  * \brief Configure UART with given master clock, and Configure PCK with given
00165  *  divider source of master clock and prescaler.
00166  *
00167  * \param css  The master clock divider source.
00168  * \param pres Master Clock prescaler.
00169  * \param masterClk frequency of the master clock (in Hz).

00171 static void _ConfigureUartAndPck( uint32_t dwClockSource, uint32_t dwPrescaler, uint32_t dwMasterClock )
00172 {
00173     /* Configures an UART Baudrate 2400 bps (except slow clock)*/
00174     if ( dwMasterClock > 32768 )
00175     {
00176         UART_Configure( 2400, dwMasterClock ) ;
00177     }
00178 
00179     /* First, disable programmable clock */
00180     PMC_DisablePck(PMC_SCDR_PCK0);
00181     /* Then, configure PMC Programmable Clock */
00182     PMC_ConfigurePck(0, dwClockSource, dwPrescaler);
00183     /* Finally, enable programmable clock */
00184     PMC_EnablePck(PMC_SCER_PCK0);
00185 }
