00043  * A terminal application, such as hyperterminal, is used to monitor these
00044  * debug information. Open another hyperterminal to connect with
00045  * on-board USART port. Then the program works in ECHO mode, so USART will
00046  * sends back anything it receives from the hyperterminal.  You can send a text
00047  * file from the hyperterminal connected with USART port to the device (without
00048  * any protocol such as X-modem).

00060  *    IAR EWARM User Guide</a>,
00061  *    depending on your chosen solution.
00062  * -# On the computer, open and configure a terminal application
00063  *    (e.g. HyperTerminal on Microsoft Windows) with these settings:
00064  *   - 115200 bauds
00065  *   - 8 bits of data
00066  *   - No parity
00067  *   - 1 stop bit
00068  *   - No flow control
00069  * -# In the terminal window, the
00070  *    following text should appear (values depend on the board and chip used):

00147 /**
00148  * \brief Interrupt handler for USART. Echo the bytes received and starts next
00149  * receive.
00150  */
00151 void USART0_IrqHandler(void)
00152 {
00153     uint32_t dwStatus;
00154 
00155     /* Read USART Status */
00156     dwStatus = USART0->US_CSR;
00157 
00158     /* Receive buffer is full */
00159     if ((dwStatus & US_CSR_ENDRX) == US_CSR_ENDRX
00160         && USART0->US_RNCR == 0 && ( ucTransMode == PDC_TRANSFER ))
00161     {
00162         /* Disable timer */
00163         TC0->TC_CHANNEL[0].TC_CCR = TC_CCR_CLKDIS;
00164 
00165         /* Clear ENDRX interrupt */
00166         USART0->US_RNCR = 0;
00167 
00168         /* Echo back buffer */
00169         USART_WriteBuffer(USART0,
00170                           pBuffer[wBuffer],
00171                           dwSizeBuffer);
00172         wBuffer = 1 - wBuffer;
00173         dwSizeBuffer = BUFFER_SIZE;
00174 
00175         /* Restart timer */
00176         TC0->TC_CHANNEL[0].TC_CCR = TC_CCR_CLKEN | TC_CCR_SWTRG;
00177     }
00178 
00179     /* Buffer has been sent */
00180     if ((dwStatus & US_CSR_ENDTX) == US_CSR_ENDTX && USART0->US_TNCR == 0)
00181     {
00182         if (ucTransMode == BYTE_TRANSFER)
00183         {
00184             /* Clear ENDTX */
00185             USART0->US_TNCR = 0;
00186             USART0->US_IDR = US_IDR_ENDTX;
00187         }
00188         else
00189         {
00190             /* Clear ENDTX */
00191             USART0->US_TNCR = 0;
00192 
00193             /* Restart read on buffer */
00194             USART_ReadBuffer(USART0,
00195                             pBuffer[rBuffer],
00196                             BUFFER_SIZE);
00197             rBuffer = 1 - rBuffer;
00198 
00199             if (rBuffer == wBuffer) 
00200             {
00201                 USART0->US_IDR = US_IDR_ENDTX;
00202             }
00203         }
00204     }
00205 
00206     /* Transfer without PDC */
00207     if ((dwStatus & US_CSR_RXRDY) == US_CSR_RXRDY && ( ucTransMode == BYTE_TRANSFER ))
00208     {
00209         uwReadBuffer = USART0->US_RHR;
00210 
00211         /* Wait untill byte write ready */
00212         while ((USART0->US_CSR & US_CSR_TXEMPTY) == 0);
00213 
00214         USART0->US_THR = uwReadBuffer;
00215     }
00216 }

00218 /**
00219  * \brief Interrupt handler for TC0. Record the number of bytes received,
00220  * then restarts a read transfer on the USART if the transfer was stopped.
00221  */
00222 void TC0_IrqHandler(void)
00223 {
00224     uint32_t dwStatus;
00225     uint32_t dwBytesTotal = 0;
00226 
00227     /* Read TC0 Status */
00228     dwStatus = TC0->TC_CHANNEL[0].TC_SR;
00229 
00230     /* RC compare */
00231     if (((dwStatus & TC_SR_CPCS) == TC_SR_CPCS) && ( ucTransMode == PDC_TRANSFER ))
00232     {
00233         /* Flush PDC buffer */
00234         dwBytesTotal = BUFFER_SIZE - USART0->US_RCR;
00235         if (dwBytesTotal == 0) {
00236

00238             return;

00241         /* Log current size */
00242         dwSizeBuffer = dwBytesTotal;
00243 
00244         /* Trigger USART ENDRX */
00245         USART0->US_RCR = 0;
