00173 /**
00174  * Process timing functions
00175  */
00176 static void timers_update(void)
00177 {
00178     static uint32_t last_time;
00179     uint32_t cur_time, time_diff, idxtimer;
00180     timers_info * ptmr_inf;
00181 
00182     cur_time = sys_get_ms();
00183     if (cur_time >= last_time)
00184     {
00185         time_diff = cur_time - last_time;
00186     }
00187     else
00188     {
00189         time_diff = 0xFFFFFFFF - last_time + cur_time;
00190     }
00191     if (time_diff)
00192     {
00193         last_time = cur_time;
00194         for(idxtimer=0;
00195             idxtimer < (sizeof(timers_table)/sizeof(timers_info));
00196             idxtimer ++)
00197         {
00198             ptmr_inf = &timers_table[idxtimer];
00199             ptmr_inf->timer += time_diff;
00200             if (ptmr_inf->timer > ptmr_inf->timer_interval)
00201             {
00202                 if (ptmr_inf->timer_func)
00203                     ptmr_inf->timer_func();
00204                 ptmr_inf->timer -= ptmr_inf->timer_interval;
00205             }
00206         }
00207     }
00208 }
