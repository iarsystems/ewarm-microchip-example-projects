00067 /**
00068  * Configure OTG settings for USB device
00069  */
00070 static void _ConfigureUotghs(void)
00071 {
00072     /* UTMI parallel mode, High/Full/Low Speed */
00073     /* UOTGCK not used in this configuration (High Speed) */
00074     PMC->PMC_SCDR = PMC_SCDR_UOTGCK;
00075     /* USB clock register: USB Clock Input is UTMI PLL */
00076     PMC->PMC_USB = PMC_USB_USBS;
00077     /* Enable peripheral clock for UOTGHS */
00078     PMC_EnablePeripheral(ID_UOTGHS);
00079     UOTGHS->UOTGHS_CTRL = UOTGHS_CTRL_UIMOD_Device;
00080     /* Enable PLL 480 MHz */
00081     PMC->CKGR_UCKR = CKGR_UCKR_UPLLEN | CKGR_UCKR_UPLLCOUNT(0xF);
00082     /* Wait that PLL is considered locked by the PMC */
00083     while( !(PMC->PMC_SR & PMC_SR_LOCKU) );
00084 
00085     /* UOTGHS pins */
00086     PIO_PinConfigure( pUOTGHSPins, PIO_LISTSIZE( pUOTGHSPins ) );
00087     if ( PIO_PinGet(&pUOTGHS_Fault) == 0 )
00088     {
00089         puts( "UOTGHS_Fault = 0 (active low  ERROR FLAG !"STRING_EOL );
00090         puts( "Undervoltage, Soft Start, Overcurrent, or Overtemperature"STRING_EOL );
00091         while(1);
00092     }
00093 
00094     /* IRQ */
00095     NVIC_EnableIRQ(UOTGHS_IRQn) ;
00096 }
