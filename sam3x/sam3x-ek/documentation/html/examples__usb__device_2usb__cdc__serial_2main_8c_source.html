00082 static void _ConfigureUotghs(void)
00083 {
00084     /* UTMI parallel mode, High/Full/Low Speed */
00085     /* UOTGCK not used in this configuration (High Speed) */
00086     PMC->PMC_SCDR = PMC_SCDR_UOTGCK;
00087     /* USB clock register: USB Clock Input is UTMI PLL */
00088     PMC->PMC_USB = PMC_USB_USBS;
00089     /* Enable peripheral clock for UOTGHS */
00090     PMC_EnablePeripheral(ID_UOTGHS);
00091     UOTGHS->UOTGHS_CTRL = UOTGHS_CTRL_UIMOD_Device;
00092     /* Enable PLL 480 MHz */
00093     PMC->CKGR_UCKR = CKGR_UCKR_UPLLEN | CKGR_UCKR_UPLLCOUNT(0xF);
00094     /* Wait that PLL is considered locked by the PMC */
00095     while( !(PMC->PMC_SR & PMC_SR_LOCKU) );
00096 
00097     /* UOTGHS pins */
00098     PIO_PinConfigure( pUOTGHSPins, PIO_LISTSIZE( pUOTGHSPins ) );
00099     if ( PIO_PinGet(&pUOTGHS_Fault) == 0 )
00100     {
00101         puts( "UOTGHS_Fault = 0 (active low  ERROR FLAG !"STRING_EOL );
00102         puts( "Undervoltage, Soft Start, Overcurrent, or Overtemperature"STRING_EOL );
00103         while(1);
00104     }
00105 
00106     /* IRQ */
00107     NVIC_EnableIRQ(UOTGHS_IRQn) ;

00156             if (main_b_com_startup) {

00160             }
00161             //** The transfer fifo<->CDC is done in the main loop
00162             // to reduce process time in TX/RX UART interrupts
00163             // Transfer CDC RX to UART TX fifo
00164             while (udi_cdc_is_rx_ready()) {
00165                 if (fifo_get_free_size(&fifo_desc_tx) == 0)
00166                     break;  // Fifo full then transmission
00167                 ui_com_rx_start();
00168                 fifo_push_byte(&fifo_desc_tx, udi_cdc_getc());
00169                 usart_enable_tx();
00170             }
00171             // Transfer UART RX fifo to CDC TX
00172             while (udi_cdc_is_tx_ready()) {
00173                 uint8_t value_pull;

00175                         FIFO_OK) {
00176                     // No data then end of loop
00177                     ui_com_tx_stop();
00178                     break;
00179                 }
00180                 udi_cdc_putc(value_pull);

00294  * \section Purpose

00298  * \section Requirements
00299  *
00300  * This package can be used with SAM3X evaluation kits.
00301  *
00302  * \section Description
00303  *
00304  * The Communication Device Class (CDC) is a general-purpose way to enable all
00305  * types of communications on the Universal Serial Bus (USB).
00306  * This class makes it possible to connect communication devices such as 
00307  * digital telephones or analog modems, as well as networking devices 
00308  * like ADSL or Cable modems.
00309  * While a CDC device enables the implementation of quite complex devices,
00310  * it can also be used as a very simple method for communication on the USB.
00311  * For example, a CDC device can appear as a virtual COM port, which greatly 
00312  * simplifies application development on the host side.
00313  *
00314  * The example is a bridge between a USART from the main MCU
00315  * and the USB CDC interface.
00316  *
00317  * For our example, we will only use a PC as a host for the demonstration:
00318  * it will connect to the USB and to the USART board connector.
00319  * - Connect the USART peripheral to the USART interface of the board.
00320  * - Connect the application to a USB host (e.g. a PC) 
00321  *   with a mini-B (embedded side) to A (PC host side) cable.
00322  * The application will behave as a virtual COM (see Windows Hardware Manager).
00323  * - Open a hyperterminal on both COM ports (RS232 and Virtual COM)
00324  * - Select the same configuration for both COM ports up to 115200 baud.
00325  * - Type a character in one hyperterminal and see it in the other.

00327  * \note
00328  * On the first connection of the board on the PC,
00329  * the operating system will detect a new peripheral:
00330  * - This will open a new hardware installation window.
00331  * - Choose "No, not this time" to connect to Windows Update for this installation
00332  * - click "Next"
00333  * - At the request to search the INF file, give the "usb_cdc_serial" folder
