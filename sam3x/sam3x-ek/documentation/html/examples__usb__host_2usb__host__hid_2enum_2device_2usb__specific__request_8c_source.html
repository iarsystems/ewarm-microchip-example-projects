00130 //!
00131 static void hid_get_descriptor(U8 size_of_report, const U8* p_usb_hid_report)
00132 {
00133   Bool  zlp;
00134   U16   wIndex;
00135   U16   wLength;
00136 
00137   zlp = FALSE;                                              /* no zero length packet */
00138 
00139   data_to_transfer = size_of_report;
00140   pbuffer          = p_usb_hid_report;
00141 
00142   wIndex = Usb_read_endpoint_data(EP_CONTROL, 16);
00143   wIndex = usb_format_usb_to_mcu_data(16, wIndex);
00144   wLength = Usb_read_endpoint_data(EP_CONTROL, 16);
00145   wLength = usb_format_usb_to_mcu_data(16, wLength);
00146   Usb_ack_setup_received_free();                          //!< clear the setup received flag
00147 
00148   if (wLength > data_to_transfer)
00149   {
00150     zlp = !(data_to_transfer % EP_CONTROL_LENGTH);  //!< zero length packet condition
00151   }
00152   else
00153   {
00154     data_to_transfer = wLength; //!< send only requested number of data bytes
00155   }
00156 
00157   Usb_ack_nak_out(EP_CONTROL);
00158 
00159   while (data_to_transfer && (!Is_usb_nak_out(EP_CONTROL)))
00160   {
00161     while( !Is_usb_control_in_ready() && !Is_usb_nak_out(EP_CONTROL) );
00162 
00163     if( Is_usb_nak_out(EP_CONTROL) )
00164        break;    // don't clear the flag now, it will be cleared after
00165 
00166     Usb_reset_endpoint_fifo_access(EP_CONTROL);
00167     data_to_transfer = usb_write_ep_txpacket(EP_CONTROL, pbuffer,
00168                                              data_to_transfer, &pbuffer);
00169     if( Is_usb_nak_out(EP_CONTROL) )
00170        break;
00171     else
00172        Usb_ack_control_in_ready_send();  //!< Send data until necessary
00173   }
00174 
00175   if ( zlp && (!Is_usb_nak_out(EP_CONTROL)) )
00176   {
00177     while (!Is_usb_control_in_ready());
00178     Usb_ack_control_in_ready_send();
00179   }
00180 
00181   while (!(Is_usb_nak_out(EP_CONTROL)));
00182   Usb_ack_nak_out(EP_CONTROL);
00183   while (!Is_usb_control_out_received());
00184   Usb_ack_control_out_received_free();
00185 }
