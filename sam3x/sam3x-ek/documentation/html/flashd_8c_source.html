00030 /** \addtogroup flashd_module Flash Memory Interface
00031  * The flash driver manages the programming, erasing, locking and unlocking sequences
00032  * with dedicated commands.
00033  *
00034  * To implement flash programing operation, the user has to follow these few steps :
00035  * <ul>
00036  * <li>Configue flash wait states to initializes the flash. </li>
00037  * <li>Checks whether a region to be programmed is locked. </li>
00038  * <li>Unlocks the user region to be programmed if the region have locked before.</li>
00039  * <li>Erases the user page before program (optional).</li>
00040  * <li>Writes the user page from the page buffer.</li>
00041  * <li>Locks the region of programmed area if any.</li>
00042  * </ul>
00043  *
00044  * Writing 8-bit and 16-bit data is not allowed and may lead to unpredictable data corruption.
00045  * A check of this validity and padding for 32-bit alignment should be done in write algorithm.
00046 
00047  * Lock/unlock range associated with the user address range is automatically translated.
00048  *
00049  * This security bit can be enabled through the command "Set General Purpose NVM Bit 0".
00050  *
00051  * A 128-bit factory programmed unique ID could be read to serve several purposes.
00052  *
00053  * The driver accesses the flash memory by calling the lowlevel module provided in \ref efc_module.
00054  * For more accurate information, please look at the EEFC section of the Datasheet.
00055  *
00056  * Related files :\n
00057  * \ref flashd.c\n
00058  * \ref flashd.h.\n
00059  * \ref efc.c\n
00060  * \ref efc.h.\n
00061 */

00100 /**
00101  * \brief Computes the lock range associated with the given address range.
00102  *
00103  * \param dwStart  Start address of lock range.
00104  * \param dwEnd  End address of lock range.
00105  * \param pdwActualStart  Actual start address of lock range.
00106  * \param pdwActualEnd  Actual end address of lock range.
00107  */
00108 static void ComputeLockRange( uint32_t dwStart, uint32_t dwEnd, uint32_t *pdwActualStart, uint32_t *pdwActualEnd )
00109 {
00110     Efc* pStartEfc ;
00111     Efc* pEndEfc ;
00112     uint16_t wStartPage ;
00113     uint16_t wEndPage ;
00114     uint16_t wNumPagesInRegion ;
00115     uint16_t wActualStartPage ;
00116     uint16_t wActualEndPage ;
00117 
00118     /* If "dwEnd" is at the start address of a lock region, */
00119     /* It actually means the end address of previous lock region  */
00120     /* So, we do adjustment of "dwEnd" here. */
00121     if ((dwEnd != IFLASH0_ADDR) && ((dwEnd & (IFLASH_LOCK_REGION_SIZE-1)) == 0))
00122     {
00123         dwEnd -= 1;
00124     }
00125 
00126     /* Convert start and end address in page numbers */
00127     EFC_TranslateAddress( &pStartEfc, dwStart, &wStartPage, 0 ) ;
00128     EFC_TranslateAddress( &pEndEfc, dwEnd, &wEndPage, 0 ) ;
00129 
00130     /* Find out the first page of the first region to lock */
00131     wNumPagesInRegion = IFLASH_LOCK_REGION_SIZE / IFLASH_PAGE_SIZE;
00132     wActualStartPage = wStartPage - (wStartPage % wNumPagesInRegion) ;
00133     wActualEndPage = wEndPage ;
00134     if ( (wEndPage % wNumPagesInRegion) != (wNumPagesInRegion-1) )
00135     {
00136         wActualEndPage += wNumPagesInRegion - (wEndPage % wNumPagesInRegion) - 1;
00137     }
00138 
00139     /* Store actual page numbers */
00140     EFC_ComputeAddress( pStartEfc, wActualStartPage, 0, pdwActualStart ) ;
00141     EFC_ComputeAddress( pEndEfc, wActualEndPage, 0, pdwActualEnd ) ;
00142     TRACE_DEBUG( "Actual lock range is 0x%06X - 0x%06X\n\r",
00143         *pdwActualStart, (*pdwActualEnd + IFLASH_PAGE_SIZE - 1) ) ;
00144 }

00191 /**
00192  * \brief Writes a data buffer in the internal flash.
00193  *
00194  * \note This function works in polling mode, and thus only returns when the
00195  * data has been effectively written.
00196  * \param address  Write address.
00197  * \param pBuffer  Data buffer.
00198  * \param size  Size of data buffer in bytes.
00199  * \return 0 if successful, otherwise returns an error code.
00200  */
00201 extern uint32_t FLASHD_Write( uint32_t dwAddress, const void *pvBuffer, uint32_t dwSize )
00202 {
00203     Efc* pEfc ;
00204     uint16_t page ;
00205     uint16_t offset ;
00206     uint32_t writeSize ;
00207     uint32_t pageAddress ;
00208     uint16_t padding ;
00209     uint32_t dwError ;
00210     uint32_t sizeTmp ;
00211     uint32_t *pAlignedDestination ;
00212     uint32_t *pAlignedSource ;
00213     uint32_t fwsBak0, fwsBak1;
00214 
00215     /* Set 6 WS for internal Flash writing (refer to errata) */
00216     fwsBak0 = ((EFC0->EEFC_FMR) & EEFC_FMR_FWS_Msk) >> EEFC_FMR_FWS_Pos;
00217     fwsBak1 = ((EFC1->EEFC_FMR) & EEFC_FMR_FWS_Msk) >> EEFC_FMR_FWS_Pos;
00218     EFC_SetWaitState(EFC0, 6);
00219     EFC_SetWaitState(EFC1, 6);
00220 
00221     uint32_t crossWrite = 0;
00222     void     *pvBufferRemain = 0;
00223     uint32_t dwSizeRemain = 0;
00224 
00225     assert( pvBuffer ) ;
00226     assert( dwAddress >= IFLASH0_ADDR ) ;
00227     assert( (dwAddress + dwSize) <= (IFLASH0_ADDR + IFLASH_SIZE) ) ;
00228 
00229     /* Check if the write operation will cross over flash0 and flash1 */
00230     if ( ((dwAddress + dwSize) <= (IFLASH0_ADDR + IFLASH_SIZE/2)) ||
00231          (dwAddress >= IFLASH1_ADDR) )
00232     {
00233         crossWrite = 0;
00234     }
00235     else
00236     {
00237         /* Save information for write operation on flash1 */
00238         crossWrite = 1;
00239         pvBufferRemain = (void *)((uint8_t *)pvBuffer + (IFLASH1_ADDR - dwAddress));
00240         dwSizeRemain = dwSize - (IFLASH1_ADDR - dwAddress);
00241         /* Adjust size for write operation on flash0 */
00242         dwSize = (IFLASH1_ADDR - dwAddress);
00243     }
00244 
00245 fd_write:
00246     /* Translate write address */
00247     EFC_TranslateAddress( &pEfc, dwAddress, &page, &offset ) ;
00248 
00249     /* Write all pages */
00250     while ( dwSize > 0 )
00251     {
00252         /* Copy data in temporary buffer to avoid alignment problems */
00253         writeSize = min((uint32_t)IFLASH_PAGE_SIZE - offset, dwSize ) ;
00254         EFC_ComputeAddress(pEfc, page, 0, &pageAddress ) ;
00255         padding = IFLASH_PAGE_SIZE - offset - writeSize ;
00256 
00257         /* Pre-buffer data */
00258         memcpy( _aucPageBuffer, (void *) pageAddress, offset);
00259 
00260         /* Buffer data */
00261         memcpy( _aucPageBuffer + offset, pvBuffer, writeSize);
00262 
00263         /* Post-buffer data */
00264         memcpy( _aucPageBuffer + offset + writeSize, (void *) (pageAddress + offset + writeSize), padding);
00265 
00266         /* Write page
00267          * Writing 8-bit and 16-bit data is not allowed and may lead to unpredictable data corruption
00268          */
00269         pAlignedDestination = (uint32_t*)pageAddress ;
00270         pAlignedSource = (uint32_t*)_adwPageBuffer ;
00271         sizeTmp = IFLASH_PAGE_SIZE ;
00272 
00273         while ( sizeTmp >= 4 )
00274         {
00275             *pAlignedDestination++ = *pAlignedSource++;
00276             sizeTmp -= 4;
00277         }
00278 
00279         /* Send writing command */
00280         dwError = EFC_PerformCommand( pEfc, EFC_FCMD_EWP, page, _dwUseIAP ) ;
00281         if ( dwError )
00282         {
00283             /* Restore FWS settings */
00284             EFC_SetWaitState(EFC0, fwsBak0);
00285             EFC_SetWaitState(EFC1, fwsBak1);
00286             return dwError ;
00287         }
00288 
00289         /* Progression */
00290         dwAddress += IFLASH_PAGE_SIZE ;
00291         pvBuffer = (void *)((uint32_t) pvBuffer + writeSize) ;
00292         dwSize -= writeSize ;
00293         page++;
00294         offset = 0;
00295     }
00296 
00297     /* Check if there is a writing operation on flash1 */
00298     if (crossWrite == 1)
00299     {
00300         dwAddress = IFLASH1_ADDR;
00301         pvBuffer = pvBufferRemain;
00302         dwSize = dwSizeRemain;
00303         crossWrite = 0;
00304         goto fd_write;
00305     }
00306 
00307     /* Restore FWS settings */
00308     EFC_SetWaitState(EFC0, fwsBak0);
00309     EFC_SetWaitState(EFC1, fwsBak1);
00310     return 0 ;
00311 }

00313  * \brief Locks all the regions in the given address range. The actual lock range is
00314  * reported through two output parameters.

00316  * \param start  Start address of lock range.
00317  * \param end    End address of lock range.

00319  * \param pActualEnd  End address of the actual lock range (optional).
00320  * \return 0 if successful, otherwise returns an error code.

00323 {
00324     Efc *pEfc ;
00325     uint32_t actualStart, actualEnd ;
00326     uint16_t startPage, endPage ;
00327     uint32_t dwError ;
00328     uint16_t numPagesInRegion = IFLASH_LOCK_REGION_SIZE / IFLASH_PAGE_SIZE;
00329 
00330     uint32_t crossLock = 0;
00331     uint32_t endBak = end;
00332 
00333     /* Check if the lock operation will cross over flash0 and flash1 */
00334     if ( (start >= IFLASH1_ADDR) || (end <= IFLASH1_ADDR) )
00335     {
00336         crossLock = 0;
00337     }
00338     else
00339     {
00340         /* Save information for lock operation on flash1 */
00341         crossLock = 1;
00342         endBak = end;
00343         /* Adjust end address for lock operation on flash0 */
00344         end = (IFLASH1_ADDR - 1);
00345     }
00346 
00347 fd_lock:
00348     /* Compute actual lock range and store it */
00349     ComputeLockRange( start, end, &actualStart, &actualEnd ) ;
00350     if ( pActualStart != NULL )
00351     {
00352         *pActualStart = actualStart ;
00353     }
00354     if ( pActualEnd != NULL )
00355     {
00356         *pActualEnd = actualEnd;
00357     }
00358 
00359     /* Compute page numbers */
00360     EFC_TranslateAddress( &pEfc, actualStart, &startPage, 0 ) ;
00361     EFC_TranslateAddress( 0, actualEnd, &endPage, 0 ) ;
00362 
00363     /* Lock all pages */
00364     while ( startPage < endPage )
00365     {
00366         dwError = EFC_PerformCommand( pEfc, EFC_FCMD_SLB, startPage, _dwUseIAP ) ;

00368         {
00369             return dwError ;
00370         }
00371         startPage += numPagesInRegion;
00372     }
00373 
00374     /* Check if there is a lock operation on flash1 */
00375     if (crossLock == 1)
00376     {
00377         start = IFLASH1_ADDR;
00378         end = endBak;
00379         crossLock = 0;
00380         goto fd_lock;
00381     }
00382 
00383     return 0 ;
00384 }

00466 {
00467     Efc *pEfc ;
00468     uint16_t startPage, endPage ;

00470     uint32_t numPagesInRegion ;
00471     uint32_t status ;
00472     uint32_t dwError ;
00473     uint32_t numLockedRegions = 0 ;
00474

00476     assert( (start >=IFLASH0_ADDR) && (end <= IFLASH0_ADDR + IFLASH_SIZE) ) ;
00477 
00478     uint32_t crossLockCheck = 0;
00479     uint32_t endBak = end;
00480 
00481     /* Check if the lock check space will cross over flash0 and flash1 */
00482     if ( (start >= IFLASH1_ADDR) || (end <= IFLASH1_ADDR) )
00483     {
00484         crossLockCheck = 0;
00485     }
00486     else
00487     {
00488         /* Save information for lock check on flash1 */
00489         crossLockCheck = 1;
00490         endBak = end;
00491         /* Adjust end address for lock check on flash0 */
00492         end = (IFLASH1_ADDR - 1);
00493     }

00495 fd_lock_check:
00496     /* Compute page numbers */
00497     EFC_TranslateAddress( &pEfc, start, &startPage, 0 ) ;
00498     EFC_TranslateAddress( 0, end, &endPage, 0 ) ;
00499 
00500     /* Compute region numbers */
00501     numPagesInRegion = IFLASH_LOCK_REGION_SIZE / IFLASH_PAGE_SIZE ;
00502     startRegion = startPage / numPagesInRegion ;
00503     endRegion = endPage / numPagesInRegion ;
00504     if ((endPage % numPagesInRegion) != 0)
00505     {
00506         endRegion++ ;
00507     }
00508 
00509     /* Retrieve lock status */
00510     dwError = EFC_PerformCommand( pEfc, EFC_FCMD_GLB, 0, _dwUseIAP ) ;

00516         while (1);

00518

00520 
00521     /* Check status of each involved region */

00523     {
00524         if ( (status & (1 << startRegion)) != 0 )
00525         {
00526             numLockedRegions++ ;
00527         }

00529     }
00530

00533     {
00534         start = IFLASH1_ADDR;
00535         end = endBak;
00536         crossLockCheck = 0;

00538     }
00539
