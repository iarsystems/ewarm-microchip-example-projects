00217  * HSMCI DMA R/W
00218  * \return 1 if DMA started.
00219  */
00220 static uint32_t _MciDMA(sMcid *pMcid, uint8_t bFByte, uint8_t bRd)
00221 {
00222     Hsmci *pHw = pMcid->pMciHw;
00223     sDmad *pDmad = pMcid->pDmad;
00224     sSdmmcCommand *pCmd = pMcid->pCmd;
00225     sDmaTransferDescriptor td;
00226     uint32_t totalSize = pCmd->wNbBlocks * pCmd->wBlockSize;
00227     uint32_t maxXSize;
00228     uint32_t memAddress;
00229     uint8_t  bMByte;
00230 
00231     /* All transfer finished */
00232     if (pMcid->dwXfrNdx >= totalSize)
00233     {
00234         return 0;
00235     }
00236     /* Memory address and alignment */
00237     memAddress = (uint32_t)&pCmd->pData[pMcid->dwXfrNdx];
00238     bMByte = ((memAddress & 0x3) || (totalSize & 0x3));
00239     /* P to M: Max size is P size */
00240     if (bRd)
00241     {
00242         maxXSize = bFByte ? MAX_DMA_SIZE : (MAX_DMA_SIZE * 4);
00243     }
00244     /* M to P: Max size is M size */
00245     else
00246     {
00247         maxXSize = bMByte ? MAX_DMA_SIZE : (MAX_DMA_SIZE * 4);
00248     }
00249     /* Update index */
00250     pMcid->dwXSize = totalSize - pMcid->dwXfrNdx;
00251     if (pMcid->dwXSize > maxXSize)
00252     {
00253         pMcid->dwXSize = maxXSize;
00254     }
00255     /* Prepare DMA transfer */
00256     if (bRd)
00257     {
00258         /* RX: bytes or words */
00259         uint32_t btR = DMAC_CTRLA_BTSIZE(bFByte ? pMcid->dwXSize :
00260                                                   toWCOUNT(pMcid->dwXSize));
00261         td.dwSrcAddr = (uint32_t)&(pHw->HSMCI_RDR);
00262         td.dwDstAddr = memAddress;
00263         td.dwCtrlA   = btR
00264                        | (bFByte ? DMAC_CTRLA_SRC_WIDTH_BYTE
00265                                  : DMAC_CTRLA_SRC_WIDTH_WORD)
00266                        | (bMByte ? DMAC_CTRLA_DST_WIDTH_BYTE
00267                                  : DMAC_CTRLA_DST_WIDTH_WORD)
00268                        ;
00269         td.dwCtrlB   = DMAC_CTRLB_SRC_DSCR | DMAC_CTRLB_DST_DSCR
00270                        | DMAC_CTRLB_FC_PER2MEM_DMA_FC
00271                        | DMAC_CTRLB_SRC_INCR_FIXED
00272                        | DMAC_CTRLB_DST_INCR_INCREMENTING
00273                        | DMAC_CTRLB_IEN
00274                        ;
00275         td.dwDscAddr = 0;
00276     }
00277     else
00278     {
00279         /* TX: Always words */
00280         uint32_t btW = DMAC_CTRLA_BTSIZE(toWCOUNT(pMcid->dwXSize));
00281         td.dwSrcAddr = memAddress;
00282         td.dwDstAddr = (uint32_t)&(pHw->HSMCI_TDR);
00283         td.dwCtrlA   = btW
00284                        | (bMByte ? DMAC_CTRLA_SRC_WIDTH_BYTE
00285                                  : DMAC_CTRLA_SRC_WIDTH_WORD)
00286                        | (bFByte ? DMAC_CTRLA_DST_WIDTH_BYTE
00287                                  : DMAC_CTRLA_DST_WIDTH_WORD)
00288                        ;
00289         td.dwCtrlB   = DMAC_CTRLB_SRC_DSCR | DMAC_CTRLB_DST_DSCR
00290                        | DMAC_CTRLB_FC_MEM2PER_DMA_FC
00291                        | DMAC_CTRLB_SRC_INCR_INCREMENTING
00292                        | DMAC_CTRLB_DST_INCR_FIXED
00293                        | DMAC_CTRLB_IEN
00294                        ;
00295         td.dwDscAddr = 0;
00296     }
00297     DMAD_PrepareSingleTransfer(pDmad, pMcid->dwDmaCh, &td);
00298     _MciDMAStart(pDmad, pMcid->dwDmaCh);
00299 
00300     return 1;
00301 }

00321 /**
00322  * Configure the  MCI CLKDIV in the MCI_MR register. The max. for MCI clock is
00323  * MCK/2 and corresponds to CLKDIV = 0
00324  * \param pMci  Pointer to the low level MCI driver.
00325  * \param mciSpeed  MCI clock speed in Hz, 0 will not change current speed.
00326  * \param mck       MCK to generate MCI Clock, in Hz
00327  * \return The actual speed used, 0 for fail.
00328  */
00329 static uint32_t MCI_SetSpeed( sMcid* pMci, uint32_t mciSpeed, uint32_t mck )
00330 {
00331     Hsmci *pMciHw = pMci->pMciHw;
00332     uint32_t clkdiv;
00333 
00334     assert(pMciHw);
00335 
00336     /* Multimedia Card Interface clock (MCCK or MCI_CK) is Master Clock (MCK)
00337      * divided by (2*(CLKDIV+1))
00338      * mciSpeed = MCK / (2*(CLKDIV+1)) */
00339     if (mciSpeed > 0) {
00340         clkdiv = (mck / 2 / mciSpeed);
00341         /* Speed should not bigger than expired one */
00342         if (mciSpeed < mck/2/clkdiv)
00343         {
00344             clkdiv ++;
00345         }
00346         if ( clkdiv > 0 )
00347         {
00348             clkdiv -= 1;
00349         }
00350         assert( (clkdiv & 0xFFFFFF00) == 0 ) ; /* "mciSpeed too small" */
00351     }
00352     else
00353     {
00354         clkdiv = 0 ;
00355     }
00356     /* Actual MCI speed */
00357     mciSpeed = mck / 2 / (clkdiv + 1);
00358 
00359     /* Modify MR */
00360     HSMCI_DivCtrl( pMciHw, clkdiv, HSMCI_MR_CLKDIV(0x7) );
00361     return (mciSpeed);
00362 }
