00041 /**
00042  * \brief Configures a list of Pin instances, each of which can either hold a single
00043  * pin or a group of pins, depending on the mask value; all pins are configured
00044  * by this function. The size of the array must also be provided and is easily
00045  * computed using PIO_LISTSIZE whenever its length is not known in advance.
00046  *
00047  * \param list  Pointer to a list of Pin instances.
00048  * \param size  Size of the Pin list (calculated using PIO_LISTSIZE).
00049  *
00050  * \return 1 if the pins have been configured properly; otherwise 0.
00051  */
00052 extern uint32_t PIO_PinConfigure( const Pin* pPins, uint32_t dwSize )
00053 {
00054     uint32_t dwOptions=0 ;
00055 
00056     /* Configure pins */
00057     while ( dwSize > 0 )
00058     {
00059         switch ( pPins->type )
00060         {
00061             case PIO_PERIPH_A :
00062                 if ( pPins->attribute & PIO_PULLUP )
00063                 {
00064                     dwOptions=PIO_PULLUP ;
00065                 }
00066 
00067                 PIO_SetPeripheralA( pPins->pio, pPins->mask, dwOptions ) ;
00068             break ;
00069 
00070             case PIO_PERIPH_B :
00071                 if ( pPins->attribute & PIO_PULLUP )
00072                 {
00073                     dwOptions=PIO_PULLUP ;
00074                 }
00075 
00076                 PIO_SetPeripheralB( pPins->pio, pPins->mask, dwOptions ) ;
00077             break ;
00078 
00079             case PIO_INPUT :
00080                 PMC_EnablePeripheral( pPins->id ) ;
00081                 PIO_SetInput( pPins->pio, pPins->mask, pPins->attribute ) ;
00082             break ;
00083 
00084             case PIO_OUTPUT_0 :
00085             case PIO_OUTPUT_1 :
00086                 if ( pPins->type == PIO_OUTPUT_1 )
00087                 {
00088                     dwOptions|=PIO_OUTPUT_HIGH ;
00089                 }
00090 
00091                 if ( pPins->attribute & PIO_OPENDRAIN )
00092                 {
00093                     dwOptions|=PIO_OPENDRAIN ;
00094                 }
00095 
00096                 if ( pPins->attribute & PIO_PULLUP )
00097                 {
00098                     dwOptions|=PIO_PULLUP ;
00099                 }
00100 
00101                 PIO_SetOutput( pPins->pio, pPins->mask, dwOptions ) ;
00102             break ;
00103 
00104             default :
00105                 return 0 ;
00106         }
00107 
00108         pPins++ ;
00109         dwSize-- ;
00110     }
00111 
00112     return 1 ;
00113 }

00139 /**
00140  * \brief Returns 1 if one or more PIO of the given Pin instance currently have
00141  * a high level; otherwise returns 0. This method returns the actual value that
00142  * is being read on the pin. To return the supposed output value of a pin, use
00143  * PIO_GetOutputDataStatus() instead.
00144  *
00145  * \param pin  Pointer to a Pin instance describing one or more pins.
00146  *
00147  * \return 1 if the Pin instance contains at least one PIO that currently has
00148  * a high level; otherwise 0.
00149  */
00150 extern uint32_t PIO_PinGet( const Pin* pPin )
00151 {
00152     uint32_t dwReg ;
00153 
00154     if ( (pPin->type == PIO_OUTPUT_0) || (pPin->type == PIO_OUTPUT_1) )
00155     {
00156         dwReg = pPin->pio->PIO_ODSR ;
00157     }
00158     else
00159     {
00160         dwReg = pPin->pio->PIO_PDSR ;
00161     }
00162 
00163     if ( (dwReg & pPin->mask) == 0 )
00164     {
00165         return 0 ;
00166     }
00167     else
00168     {
00169         return 1 ;
00170     }
00171 }

00196  * \brief Configures Glitch or Debouncing filter for input.

00198  * \param pPin       Pointer to a Pin instance describing one or more pins.

00201 extern void PIO_PinSetDebounceFilter( const Pin* pPin, uint32_t dwCutOff )
00202 {
00203     Pio* pPio = pPin->pio ;
00204 
00205     pPio->PIO_DIFSR = pPin->mask; /* set Debouncing, 0 bit field no effect */
