00077 /**
00078  * \brief Handles all interrupts on the given PIO controller.
00079  * \param id  PIO controller ID.
00080  * \param pPio  PIO controller base address.
00081  */
00082 extern void PioInterruptHandler( uint32_t id, Pio *pPio )
00083 {
00084     uint32_t status ;
00085     uint32_t i ;
00086 
00087     /* Read PIO controller status */
00088     status = pPio->PIO_ISR ;
00089     status &= pPio->PIO_IMR ;
00090 
00091     /* Check pending events */
00092     if ( status != 0 )
00093     {
00094         TRACE_DEBUG( "PIO interrupt on PIO controller #%d\n\r", id ) ;
00095 
00096         /* Find triggering source */
00097         i = 0 ;
00098         while ( status != 0 )
00099         {
00100             /* There cannot be an unconfigured source enabled. */
00101             assert( i < _dwNumSources ) ;
00102 
00103             /* Source is configured on the same controller */
00104             if ( _aIntSources[i].pPin->id == id )
00105             {
00106                 /* Source has PIOs whose statuses have changed */
00107                 if ( (status & _aIntSources[i].pPin->mask) != 0 )
00108                 {
00109                     TRACE_DEBUG( "Interrupt source #%d triggered\n\r", i ) ;
00110 
00111                     _aIntSources[i].handler(_aIntSources[i].pPin ) ;
00112                     status &= ~(_aIntSources[i].pPin->mask) ;
00113                 }
00114             }
00115             i++ ;
00116         }
00117     }
00118 }
