00387 /**
00388  * \brief Enable external 12MHz oscilator as main clock input.
00389  */
00390 extern void PMC_EnableExtOsc12MHz(void)
00391 {
00392     uint32_t   read_MOR;
00393 
00394     /* Before switching MAIN OSC on external crystal : enable it and don't disable
00395      * at the same time RC OSC in case of if MAIN OSC is still using RC OSC
00396      */
00397 
00398     read_MOR = PMC->CKGR_MOR;
00399 
00400     read_MOR &= ~CKGR_MOR_MOSCRCF_Msk;   /* reset MOSCRCF field in MOR register before select RC 12MHz */
00401 
00402     read_MOR |= (CKGR_MOR_KEY(0x37u) |
00403             CKGR_MOR_MOSCRCF_12MHz   |
00404             CKGR_MOR_MOSCXTEN        |
00405             CKGR_MOR_MOSCRCEN        |
00406             CKGR_MOR_MOSCXTST(PmcMainOscCnt));  /* enable external crystal - enable RC OSC */
00407 
00408     PMC->CKGR_MOR = read_MOR;
00409 
00410     while( !(PMC->PMC_SR & PMC_SR_MOSCRCS ) );  /* wait end of RC oscillator stabilization */
00411     while( !(PMC->PMC_SR & PMC_SR_MCKRDY) );
00412 
00413     read_MOR |= CKGR_MOR_MOSCSEL;               /* select external crystal */
00414 
00415     PMC->CKGR_MOR = read_MOR;
00416 
00417     while( !(PMC->PMC_SR & PMC_SR_MOSCSELS ) ); /* Wait end of Main Oscillator Selection */
00418     while( !(PMC->PMC_SR & PMC_SR_MCKRDY) );
00419 }
