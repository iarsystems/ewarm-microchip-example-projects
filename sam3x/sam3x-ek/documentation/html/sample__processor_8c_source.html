00122 /**
00123  * \brief  Initial the codec for audio play.
00124  */
00125 static void _InitDAC(void)
00126 {
00127     /* First, disable programmable clock */
00128     PMC_DisablePck(PMC_SCDR_PCK0);
00129     /* Then, configure PMC Programmable Clock: 12MHz */
00130     PMC_ConfigurePck(0, PMC_MCKR_CSS_MAIN_CLK, PMC_MCKR_PRES_CLK_1);
00131     /* Finally, enable programmable clock */
00132     PMC_EnablePck(PMC_SCER_PCK0);
00133     /* Initialize the audio DAC */
00134     WM8731_DAC_Init(&twid, WM8731_SLAVE_ADDRESS);
00135 
00136     /* digital audio interface format (R7: default 0x0A? ) */
00137     /* I2S, left justified */
00138     WM8731_Write(&twid, WM8731_SLAVE_ADDRESS, WM8731_REG_DA_INTERFACE_FORMAT, WM8731_FORMAT_I2S|(1<<4));
00139     /* sampling control (R8: default 0x00) */
00140     /* USB mode */
00141     WM8731_Write(&twid, WM8731_SLAVE_ADDRESS, WM8731_REG_SAMPLECTRL, WM8731_USB_MODE);
00142     /* power down control (R6: default 0x9F) */
00143     WM8731_Write(&twid, WM8731_SLAVE_ADDRESS, WM8731_REG_PWDOWN_CTRL, 0x07);
00144 }

00163 void sample_processor_output_samples(uint16_t *samples, uint8_t count)
00164 {
00165     uint8_t i;
00166     SSC_EnableTransmitter(SSC);
00167     // Process each sample in the datastream
00168     for (i = 0; i < count; i += 2)
00169     {
00170         // Two channels (left and right) are encoded together to form a sample pair
00171         int16_t sample_left  = le16_to_cpu(samples[i]);
00172         int16_t sample_right = le16_to_cpu(samples[i + 1]);
00173   
00174         SSC_Write(SSC, sample_left);
00175         SSC_Write(SSC, sample_right);
00176     }
00177     SSC_DisableTransmitter(SSC);
00178 }
