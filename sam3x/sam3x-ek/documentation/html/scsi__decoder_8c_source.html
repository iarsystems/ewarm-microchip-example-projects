00283 Bool sbc_request_sense(void)
00284 {
00285   U8 allocation_length;
00286   U8 request_sense_output[18];  // The maximal size of request is 17.
00287 
00288   allocation_length = min(g_scsi_command[4], sizeof(request_sense_output));
00289 
00290   if( allocation_length != 0 )
00291   {
00292      // Initialize the request sense data.
00293      request_sense_output[ 0] = SBC_RESPONSE_CODE_SENSE; // 0x70.
00294      request_sense_output[ 1] = 0x00;                    // Obsolete.
00295      request_sense_output[ 2] = g_scsi_sense.key;
00296    
00297      request_sense_output[ 3] = 0x00;  // For direct access media, Information field.
00298      request_sense_output[ 4] = 0x00;  // Give the unsigned logical block.
00299      request_sense_output[ 5] = 0x00;  // Address associated with the sense key.
00300      request_sense_output[ 6] = 0x00;
00301    
00302      request_sense_output[ 7] = SBC_ADDITIONAL_SENSE_LENGTH; // Device shall not adjust the additional sense length to reflect truncation.
00303      request_sense_output[ 8] = SBC_COMMAND_SPECIFIC_INFORMATION_3;
00304      request_sense_output[ 9] = SBC_COMMAND_SPECIFIC_INFORMATION_2;
00305      request_sense_output[10] = SBC_COMMAND_SPECIFIC_INFORMATION_1;
00306      request_sense_output[11] = SBC_COMMAND_SPECIFIC_INFORMATION_0;
00307    
00308      request_sense_output[12] = g_scsi_sense.asc;
00309      request_sense_output[13] = g_scsi_sense.ascq;
00310    
00311      request_sense_output[14] = SBC_FIELD_REPLACEABLE_UNIT_CODE;
00312      request_sense_output[15] = SBC_SENSE_KEY_SPECIFIC_2;
00313      request_sense_output[16] = SBC_SENSE_KEY_SPECIFIC_1;
00314      request_sense_output[17] = SBC_SENSE_KEY_SPECIFIC_0;
00315    
00316      // Send the request data.
00317      Usb_reset_endpoint_fifo_access(g_scsi_ep_ms_in);
00318      usb_write_ep_txpacket(g_scsi_ep_ms_in, request_sense_output, allocation_length, NULL);
00319      Sbc_valid_write_usb(allocation_length);
00320      // MSC Compliance - Wait end of all transmitions on USB line, because a stall may be send after data
00321      while( 0 != Usb_nb_busy_bank(EP_MS_IN) );
00322   }
00323   sbc_lun_status_is_good();
00324   return (allocation_length == g_scsi_command[4]);
00325 }

00328 Bool sbc_inquiry(void)
00329 {
00330   U8 allocation_length;
00331 
00332   // CMDT or EPVD bit is not 0 or PAGE or OPERATION CODE fields != 0x00.
00333   if ((g_scsi_command[1] & 0x03) || g_scsi_command[2])
00334   {
00335     sbc_lun_status_is_cdb_field();
00336     return FALSE;
00337   }
00338   
00339   // Send standard INQUIRY data (bytes 0 to (allocation_length - 1)).
00340   allocation_length = min(g_scsi_command[4], sizeof(sbc_st_std_inquiry_data));
00341   if( allocation_length != 0 )
00342   {
00343      while (!Is_usb_in_ready(g_scsi_ep_ms_in))
00344      {
00345        if(!Is_usb_endpoint_enabled(g_scsi_ep_ms_in))
00346          return FALSE; // USB Reset
00347      }         
00348      Usb_reset_endpoint_fifo_access(g_scsi_ep_ms_in);
00349      usb_write_ep_txpacket(g_scsi_ep_ms_in, &sbc_std_inquiry_data,
00350                            allocation_length, NULL);
00351      Sbc_valid_write_usb(allocation_length);
00352      // MSC Compliance - Wait end of all transmitions on USB line, because a stall may be send after data
00353      while( 0 != Usb_nb_busy_bank(EP_MS_IN) );
00354   }
00355   sbc_lun_status_is_good();
00356   return (allocation_length == g_scsi_command[4]);
00357 }

00360 Bool sbc_mode_sense(Bool b_sense_10)
00361 {
00362   U8 allocation_length = (b_sense_10) ? g_scsi_command[8] : g_scsi_command[4];
00363 
00364   // Switch for page code.
00365   switch (g_scsi_command[2] & SBC_MSK_PAGE_CODE)
00366   {
00367   case SBC_PAGE_CODE_READ_WRITE_ERROR_RECOVERY:
00368     sbc_header_mode_sense(b_sense_10, SBC_MODE_DATA_LENGTH_READ_WRITE_ERROR_RECOVERY);
00369     send_read_write_error_recovery_page(allocation_length);
00370     Sbc_valid_write_usb(SBC_MODE_DATA_LENGTH_READ_WRITE_ERROR_RECOVERY + 1);
00371     break;
00372 
00373   // Page Code: Informational Exceptions Control mode page.
00374   case SBC_PAGE_CODE_INFORMATIONAL_EXCEPTIONS:
00375     sbc_header_mode_sense(b_sense_10, SBC_MODE_DATA_LENGTH_INFORMATIONAL_EXCEPTIONS);
00376     send_informational_exceptions_page();
00377     Sbc_valid_write_usb(SBC_MODE_DATA_LENGTH_INFORMATIONAL_EXCEPTIONS + 1);
00378     break;
00379 
00380   case SBC_PAGE_CODE_ALL:
00381     if( b_sense_10 ) {
00382        sbc_header_mode_sense( b_sense_10 , (allocation_length < (SBC_MODE_DATA_LENGTH_CODE_ALL+2))? (allocation_length-2) : SBC_MODE_DATA_LENGTH_CODE_ALL );
00383     }else{
00384        sbc_header_mode_sense( b_sense_10 , (allocation_length < (SBC_MODE_DATA_LENGTH_CODE_ALL+1))? (allocation_length-1) : SBC_MODE_DATA_LENGTH_CODE_ALL );
00385     }
00386     if (allocation_length == ((b_sense_10) ? 8 : 4))
00387     {
00388       Sbc_valid_write_usb(allocation_length);
00389       break;
00390     }
00391     // Send page by ascending order code.
00392     send_read_write_error_recovery_page(allocation_length); // 12 bytes.
00393     if (allocation_length > 12)
00394     {
00395       send_informational_exceptions_page();                 // 12 bytes.
00396       Sbc_valid_write_usb(SBC_MODE_DATA_LENGTH_CODE_ALL + 1);
00397     }
00398     else
00399     {
00400       Sbc_valid_write_usb(allocation_length);
00401     }
00402     break;
00403 
00404   default:
00405     sbc_lun_status_is_cdb_field();
00406     return FALSE;
00407   }
00408 
00409   sbc_lun_status_is_good();
00410 
00411   return TRUE;
00412 }
