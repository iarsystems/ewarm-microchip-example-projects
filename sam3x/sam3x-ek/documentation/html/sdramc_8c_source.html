00044 /**
00045  * \brief Calculate the sdram controller config register value.
00046  * \param pMemory  Pointer to the sdram structure.
00047  * \return Configure register value.
00048  */
00049 static uint32_t SDRAMC_compute_CR_value( SSdramc_Memory* pMemory )
00050 {
00051     uint32_t dw=0 ;
00052 
00053     dw |= pMemory->cfg.dwColumnBits ;
00054     dw |= pMemory->cfg.dwRowBits ;
00055     dw |= pMemory->cfg.dwBanks ;  //NB, number of banks
00056     dw |= pMemory->cfg.dwCAS ;  //CAS, CAS latency
00057     dw |= pMemory->cfg.dwDataBusWidth ;  //DBW, data bus width
00058     dw |= SDRAMC_CR_TWR( pMemory->cfg.dwWriteRecoveryDelay ) ;  //TWR, Write Recovery Delay
00059     dw |= SDRAMC_CR_TRC_TRFC( pMemory->cfg.dwRowCycleDelay_RowRefreshCycle ) ;  //TRC_TRFC,Row Cycle Delay and Row Refresh Cycle
00060     dw |= SDRAMC_CR_TRP( pMemory->cfg.dwRowPrechargeDelay ) ;  //TRP, Row Precharge Delay
00061     dw |= SDRAMC_CR_TRCD( pMemory->cfg.dwRowColumnDelay ) ;  //TRCD, Row to Column Delay
00062     dw |= SDRAMC_CR_TRAS( pMemory->cfg.dwActivePrechargeDelay ) ;  //TRAS, Active to Precharge Delay
00063     dw |= SDRAMC_CR_TXSR( pMemory->cfg.dwExitSelfRefreshActiveDelay ) ;  //TXSR, Exit Self Refresh to Active Delay
00064 
00065     return dw ;
00066 }

00071 /**
00072  * \brief Configure and initialize the SDRAM controller.
00073  * \param pMemory  Pointer to the sdram structure.
00074  * \param dwClockFrequency  SDRAM clock frequency.
00075  */
00076 extern void SDRAMC_Configure( SSdramc_Memory* pMemory, uint32_t dwClockFrequency )
00077 {
00078     volatile uint32_t dw ;
00079 
00080     /* SDRAM hardware init */
00081     /* Enable peripheral clock */
00082     PMC_EnablePeripheral( ID_SMC ) ;
00083 
00084     /* SDRAM device configure */
00085     /* Step 1. */
00086     /* Program the features of SDRAM device into the Configuration Register.*/
00087     SDRAMC->SDRAMC_CR = SDRAMC_compute_CR_value( pMemory ) ;
00088 
00089     /* Step 2. */
00090     /* For low-power SDRAM, temperature-compensated self refresh (TCSR),
00091     drive strength (DS) and partial array self refresh (PASR) must be set
00092     in the Low-power Register.*/
00093     SDRAMC->SDRAMC_LPR = 0;
00094 
00095     /* Step 3. */
00096     /* Program the memory device type into the Memory Device Register */
00097     SDRAMC->SDRAMC_MDR = SDRAMC_MDR_MD_SDRAM;
00098 
00099     /* Step 4 */
00100     /* A minimum pause of 200 ¦Ěs is provided to precede any signal toggle.
00101     (6 core cycles per iteration) */
00102     for ( dw = 0; dw < ((dwClockFrequency/1000000)*200/6) ; dw++ )
00103     {
00104         ;
00105     }
00106 
00107     /* Step 5. */
00108     /* A NOP command is issued to the SDR-SDRAM. Program NOP command into
00109     Mode Register, the application must set Mode to 1 in the Mode Register.
00110     Perform a write access to any SDR-SDRAM address to acknowledge this command.
00111     Now the clock which drives SDR-SDRAM device is enabled.*/
00112     SDRAMC->SDRAMC_MR = SDRAMC_MR_MODE_NOP;
00113     *(uint16_t *)(EBI_SDRAMC_ADDR) = 0;
00114 
00115     /* Step 6. */
00116     /* An all banks precharge command is issued to the SDR-SDRAM. Program all
00117     banks precharge command into Mode Register, the application must set Mode to
00118     2 in the Mode Register . Perform a write access to any SDRSDRAM address to
00119     acknowledge this command. */
00120     SDRAMC->SDRAMC_MR = SDRAMC_MR_MODE_ALLBANKS_PRECHARGE;
00121     *(uint16_t *)(EBI_SDRAMC_ADDR) = 0x0;
00122 
00123     /* add some delays after precharge */
00124     for ( dw = 0; dw < ((dwClockFrequency/1000000)*200/6) ; dw++ )
00125     {
00126         ;
00127     }
00128 
00129     /* Step 7. */
00130     /* Eight auto-refresh (CBR) cycles are provided. Program the auto refresh
00131     command (CBR) into Mode Register, the application must set Mode to 4 in
00132     the Mode Register. Once in the idle state, eight AUTO REFRESH cycles must
00133     be performed. */
00134     SDRAMC->SDRAMC_MR = SDRAMC_MR_MODE_AUTO_REFRESH;
00135     *(uint16_t *)(EBI_SDRAMC_ADDR + 0 ) = 0x1;
00136 
00137     SDRAMC->SDRAMC_MR = SDRAMC_MR_MODE_AUTO_REFRESH;
00138     *(uint16_t *)(EBI_SDRAMC_ADDR + 0) = 0x2;
00139 
00140     SDRAMC->SDRAMC_MR = SDRAMC_MR_MODE_AUTO_REFRESH;
00141     *(uint16_t *)(EBI_SDRAMC_ADDR + 0 ) = 0x3;
00142 
00143     SDRAMC->SDRAMC_MR = SDRAMC_MR_MODE_AUTO_REFRESH;
00144     *(uint16_t *)(EBI_SDRAMC_ADDR + 0) = 0x4;
00145 
00146     SDRAMC->SDRAMC_MR = SDRAMC_MR_MODE_AUTO_REFRESH;
00147     *(uint16_t *)(EBI_SDRAMC_ADDR + 0 ) = 0x5;
00148 
00149     SDRAMC->SDRAMC_MR = SDRAMC_MR_MODE_AUTO_REFRESH;
00150     *(uint16_t *)(EBI_SDRAMC_ADDR + 0) = 0x6;
00151 
00152     SDRAMC->SDRAMC_MR = SDRAMC_MR_MODE_AUTO_REFRESH;
00153     *(uint16_t *)(EBI_SDRAMC_ADDR + 0 ) = 0x7;
00154 
00155     SDRAMC->SDRAMC_MR = SDRAMC_MR_MODE_AUTO_REFRESH;
00156     *(uint16_t *)(EBI_SDRAMC_ADDR + 0) = 0x8;
00157 
00158     /* Step 8. */
00159     /* A Mode Register set (MRS) cycle is issued to program the parameters of
00160     the SDRAM devices, in particular CAS latency and burst length. */
00161     SDRAMC->SDRAMC_MR = SDRAMC_MR_MODE_LOAD_MODEREG;
00162     *(uint16_t *)(EBI_SDRAMC_ADDR + 0x22) = 0xcafe;
00163 
00164     /* Step 9. */
00165     /* For low-power SDR-SDRAM initialization, an Extended Mode Register set
00166     (EMRS) cycle is issued to program the SDR-SDRAM parameters (TCSR, PASR, DS).
00167     The write address must be chosen so that BA[1] is set to 1 and BA[0] is set
00168     to 0 */
00169     SDRAMC->SDRAMC_MR = SDRAMC_MR_MODE_EXT_LOAD_MODEREG;
00170     *((uint16_t *)(EBI_SDRAMC_ADDR + (1 << pMemory->cfg.dwBK1))) = 0;
00171 
00172     /* Step 10. */
00173     /* The application must go into Normal Mode, setting Mode to 0 in the Mode
00174     Register and perform a write access at any location in the SDRAM to
00175     acknowledge this command. */
00176     SDRAMC->SDRAMC_MR = SDRAMC_MR_MODE_NORMAL;
00177     *(uint16_t *)(EBI_SDRAMC_ADDR ) = 0x0;
00178 
00179     /* Step 11. */
00180     /* Write the refresh rate into the count field in the SDRAMC Refresh
00181      Timer register. Set Refresh timer 15.625 us*/
00182     dw=dwClockFrequency/1000u ;
00183     dw*=15625u ;
00184     dw/=1000000u ;
00185     SDRAMC->SDRAMC_TR = SDRAMC_TR_COUNT( dw ) ;
00186 }
