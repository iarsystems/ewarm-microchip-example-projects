00348 /**
00349  * \brief Verifies 4-bytes hsiao codes for a data block whose size is a page Size
00350  * word. Page words block is verified between the given HSIAO code
00351  * generated by hardware and original HSIAO codes store has been previously stored.
00352  * Returns 0 if the data is correct, Hsiao_ERROR_SINGLEBIT if one or more
00353  * block(s) have had a single bit corrected, or either Hsiao_ERROR_ECC
00354  * or Hsiao_ERROR_MULTIPLEBITS.
00355  *
00356  * \param data  Data buffer to verify.
00357  * \param originalCode  Original codes.
00358  * \param verifyCode  codes to be verified.
00359  */
00360 static uint8_t SMC_ECC_VerifyPageOf8bitHsiao(
00361     uint8_t *data,
00362     const uint8_t *originalCode,
00363     const uint8_t *verifyCode)
00364 {
00365     uint8_t correctionCode[4];
00366     uint8_t bitCount;
00367 
00368     /* Xor both codes together */
00369     correctionCode[0] = verifyCode[0] ^ originalCode[0];
00370     correctionCode[1] = verifyCode[1] ^ originalCode[1];
00371     correctionCode[2] = verifyCode[2] ^ originalCode[2];
00372     correctionCode[3] = verifyCode[3] ^ originalCode[3];
00373     TRACE_DEBUG("Correction code = %02X %02X %02X %02X\n\r",
00374                 correctionCode[0], correctionCode[1], correctionCode[2], correctionCode[3]);
00375     /* If all bytes are 0, there is no error */
00376     if ((correctionCode[0] == 0)
00377         && (correctionCode[1] == 0)
00378         && (correctionCode[2] == 0)
00379         && (correctionCode[3] == 0)) {
00380 
00381         return 0;
00382     }
00383      /* If there is a single bit error, there are 15 bits set to 1 */
00384     bitCount = ECC_CountBitsInByte(correctionCode[0]) +
00385                ECC_CountBitsInByte(correctionCode[1]) +
00386                ECC_CountBitsInByte(correctionCode[2]) +
00387                ECC_CountBitsInByte(correctionCode[3]);
00388     if (bitCount == 15) {
00389         /* Get byte and bit indexes */
00390         uint16_t byte = (correctionCode[0] & 0xf0) >> 4;
00391         byte |= (correctionCode[1] & 0xff) << 4;
00392         uint8_t bit = correctionCode[0] & 0x0f;
00393 
00394         /* Correct bit */
00395         TRACE_INFO("Correcting byte #%d at bit %d\n\r", byte, bit);
00396         data[byte] ^= (1 << bit);
00397 
00398         return Hsiao_ERROR_SINGLEBIT;
00399     }
00400 
00401     /* Check if ECC has been corrupted */
00402     if (bitCount == 1) {
00403         return Hsiao_ERROR_ECC;
00404     }
00405     /* Otherwise, this is a multi-bit error */
00406     else {
00407         return Hsiao_ERROR_MULTIPLEBITS;
00408     }
00409 }
