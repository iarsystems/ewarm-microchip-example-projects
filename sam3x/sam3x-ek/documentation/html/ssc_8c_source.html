00078 /**
00079  * \brief Configures a SSC peripheral.If the divided clock is not used, the master
00080  * clock frequency can be set to 0.
00081  * \note The emitter and transmitter are disabled by this function.
00082  * \param bitRate  bit rate.
00083  * \param masterClock  master clock.
00084  */
00085 extern void SSC_Configure( Ssc* pSsc, uint32_t dwBitRate, uint32_t dwMasterClock )
00086 {
00087     /* Enable SSC peripheral clock */
00088     PMC_EnablePeripheral( ID_SSC ) ;
00089 
00090     /* Reset, disable receiver & transmitter */
00091     pSsc->SSC_CR = SSC_CR_RXDIS | SSC_CR_TXDIS | SSC_CR_SWRST ;
00092 
00093 //    pSsc->SSC_PTCR = SSC_PTCR_RXTDIS | SSC_PTCR_TXTDIS ;
00094 
00095     /* Configure clock frequency */
00096     if ( dwBitRate != 0 )
00097     {
00098         pSsc->SSC_CMR = dwMasterClock / (dwBitRate<<1) ;
00099     }
00100     else
00101     {
00102         pSsc->SSC_CMR = 0 ;
00103     }
00104 }
