00094     // Configure OSC0 in crystal mode, external crystal

00096     scif_configure_osc_crystalmode(SCIF_OSC0, 16000000);

00101     pm_set_mclk_source(PM_CLK_SRC_OSC0);

00105     if (!ast_init_counter(&AVR32_AST, AST_OSC_RC, 0, 0)) {
