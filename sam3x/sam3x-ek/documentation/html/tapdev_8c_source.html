00123 /**
00124  * Initialization for EMAC device.
00125  * Should be called at the beginning of the program to set up the
00126  * network interface.
00127  */
00128 void tapdev_init(void)
00129 {
00130     sEmacd    *pEmacd = &gEmacd;
00131     Macb      *pMacb = &gMacb;
00132     uint32_t  dwErrCount = 0 ;
00133     volatile uint32_t  dwDly;
00134 
00135     /* Reset PHY */
00136     RSTC_SetExtResetLength(RSTC, 13); /* (2^(13+1))/32768 */
00137     RSTC_ExtReset(RSTC);
00138     while(RSTC_GetNrstLevel(RSTC) == 0) {}
00139     /* Wait for PHY to be ready (CAT811: Max400ms) */
00140     dwDly = BOARD_MCK / 1000 / 3 * 400;
00141     while(dwDly--);
00142 
00143     /* Init EMAC driver structure */
00144     EMACD_Init(pEmacd, EMAC, ID_EMAC, 0, 0);
00145     EMACD_InitTransfer(pEmacd,
00146       gpRxBuffer, gRxDs, RX_BUFFERS,
00147       gpTxBuffer, gTxDs, gTxCbs, TX_BUFFERS);
00148     EMAC_SetAddress(gEmacd.pHw, 0, gMacAddress);
00149 
00150     /* Enable Interrupt */
00151     NVIC_EnableIRQ( EMAC_IRQn ) ;
00152 
00153     /* Init MACB driver */
00154     MACB_Init(pMacb, pEmacd, BOARD_EMAC_PHY_ADDR);
00155 
00156     /* PHY initialize */
00157     if (!MACB_InitPhy(pMacb, BOARD_MCK,
00158                     gEmacPins, PIO_LISTSIZE(gEmacPins)))
00159     {
00160       printf( "P: PHY Initialize ERROR!\n\r" ) ;
00161       return;
00162     }
00163 
00164     /* Auto Negotiate, work in RMII mode */
00165     if (!MACB_AutoNegotiate(pMacb, BOARD_EMAC_MODE_RMII))
00166     {
00167 
00168       printf( "P: Auto Negotiate ERROR!\n\r" ) ;
00169       return;
00170     }
00171 
00172     while ( MACB_GetLinkSpeed( pMacb, 1 ) == 0 )
00173     {
00174       dwErrCount++;
00175     }
00176     printf( "P: Link detected \n\r" ) ;
00177 }
