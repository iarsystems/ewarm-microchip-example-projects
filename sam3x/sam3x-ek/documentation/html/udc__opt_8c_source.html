00297 static bool udc_req_std_dev_get_descriptor(void) @ "BOOT"

00309         // Configuration descriptor requested

00312             (uint8_t*)&udc_desc_fs,

00317         // String descriptor requested
00318         if (!udc_req_std_dev_get_str_desc()) {
00319             return false;

00323     default:
00324         // Unknown descriptor requested
00325         return false;
00326     }
00327     // if the descriptor is larger than length requested, then reduce it
00328     if (udd_g_ctrlreq.req.wLength < udd_g_ctrlreq.payload_size)
00329         udd_g_ctrlreq.payload_size = udd_g_ctrlreq.req.wLength;
00330     return true;

00417 /**
00418  * \brief Main routine to manage the USB SETUP request.
00419  *
00420  * This function parses a USB SETUP request and submits an appropriate
00421  * response back to the host or, in the case of SETUP OUT requests
00422  * with data, sets up a buffer for receiving the data payload.
00423  *
00424  * The main standard requests defined by the USB 2.0 standard are handled
00425  * internally. The interface requests are sent to UDI, and the specific request
00426  * sent to a specific application callback.
00427  *
00428  * \return true if the request is supported, else the request is stalled by UDD
00429  */
00430 bool udc_process_setup(void) @ "BOOT"
00431 {
00432     // By default no data (receive/send) and no callbacks registered
00433     udd_g_ctrlreq.payload_size = 0;
00434     udd_g_ctrlreq.callback = NULL;
00435     udd_g_ctrlreq.over_under_run = NULL;
00436 
00437     if (Udd_setup_is_in()) {
00438         if (udd_g_ctrlreq.req.wLength == 0)
00439             return false;   // Error from USB host
00440     }
00441     
00442     // If standard request then try to decode it in UDC
00443     if (Udd_setup_type() == USB_REQ_TYPE_STANDARD) {
00444         if (udc_reqstd())
00445             return true;
00446     }
00447 
00448     // If interface request then try to decode it in UDI
00449     if (Udd_setup_recipient() == USB_REQ_RECIP_INTERFACE) {
00450         if (udc_req_iface())
00451             return true;
00452     }
00453     
00454     // Here SETUP request unknown by UDC and UDIs
00455 #ifdef USB_DEVICE_SPECIFIC_REQUEST
00456     // Try to decode it in specific callback
00457     return USB_DEVICE_SPECIFIC_REQUEST();   // Ex: Vendor request,...
00458 #else
00459     return false;
00460 #endif
00461 }
