00398 void udi_cdc_data_recevied(udd_ep_status_t status, iram_size_t n)
00399 {
00400     if (UDD_EP_TRANSFER_OK != status) {
00401         // Abort reception
00402         return;
00403     }
00404     if (0 == n) {
00405         // Empty packet then restart reception on same buffer
00406         udi_cdc_rx_start();
00407         return;
00408     }
00409     // Update the buffer's number
00410     udi_cdc_rx_buf_nb[udi_cdc_rx_trans_sel] = n;
00411 
00412     // Go to next buffer
00413     udi_cdc_rx_trans_sel = (udi_cdc_rx_trans_sel + 1) % 2;
00414     // Check if next buffer is free
00415     if (udi_cdc_rx_trans_sel == udi_cdc_rx_buf_sel) {
00416         // No buffer free; stop reception
00417         udi_cdc_rx_trans_sel = UDI_CDC_TRANS_HALTED;
00418     } else {
00419         // Restart reception on next buffer
00420         udi_cdc_rx_start();
00421     }

00500 int udi_cdc_getc(void)
00501 {
00502     int rx_data = 0;
00503     bool b_databit_9;
00504 
00505     b_databit_9 = (9 == udi_cdc_line_coding.bDataBits);
00506 
00507 udi_cdc_getc_process_one_byte:
00508     // Waiting for data
00509     while (!udi_cdc_is_rx_ready()) {
00510         if (UDI_CDC_TRANS_HALTED == udi_cdc_rx_trans_sel)
00511             return 0;   // Error system
00512     }
00513 
00514     // Read data
00515     rx_data |= udi_cdc_rx_buf[udi_cdc_rx_buf_sel][udi_cdc_rx_pos];
00516     udi_cdc_rx_pos++;
00517 
00518     // Check if buffer empty
00519     if (udi_cdc_rx_pos == udi_cdc_rx_buf_nb[udi_cdc_rx_buf_sel]) {
00520         // Initialize again current buffer
00521         udi_cdc_rx_buf_nb[udi_cdc_rx_buf_sel] = 0;
00522         // Switch to next buffer
00523         udi_cdc_rx_pos = 0;
00524         udi_cdc_rx_buf_sel = (udi_cdc_rx_buf_sel + 1) % 2;
00525         // Check if reception is halted
00526         if (UDI_CDC_TRANS_HALTED == udi_cdc_rx_trans_sel) {
00527             // Restart RX reception
00528             udi_cdc_rx_trans_sel = (udi_cdc_rx_buf_sel + 1) % 2;
00529             udi_cdc_rx_start();
00530         }
00531     }
00532     if (b_databit_9) {
00533         // Receive MSB
00534         b_databit_9 = false;
00535         rx_data = rx_data << 8;
00536         goto udi_cdc_getc_process_one_byte;
00537     }
00538     return rx_data;
00539 }
