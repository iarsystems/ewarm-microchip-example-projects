00203 bool udi_dfu_atmel_setup(void)
00204 {
00205     //** Interface requests
00206     if (Udd_setup_type() != USB_REQ_TYPE_CLASS) {
00207         return false;   // Only class request decoded
00208     }
00209 
00210     if (Udd_setup_is_in()) {
00211         // Requests Class Interface Get
00212         switch (udd_g_ctrlreq.req.bRequest) {
00213         case USB_REQ_DFU_GETSTATUS:
00214 #if (UDI_DFU_ATMEL_PROTOCOL_VERSION == DFU_ATMEL_PROTOCOL_VERSION_2)
00215 #  ifdef UDI_DFU_ATMEL_PROTOCOL_2_SPLIT_ERASE_CHIP
00216             if (udi_dfu_atmel_erase_running) {
00217                 udi_dfu_atmel_erase_chip();
00218             }
00219 #  endif
00220 #endif
00221             Assert(udd_g_ctrlreq.req.wValue==0);
00222             Assert(sizeof(udi_dfu_atmel_status)==udd_g_ctrlreq.req.wLength);
00223             udd_set_setup_payload(
00224                     (uint8_t *) & udi_dfu_atmel_status,
00225                     sizeof(udi_dfu_atmel_status));
00226             return true;
00227 
00228             // Used to send data to the host
00229             // when the previous Atmel command (DNLOAD) request data
00230         case USB_REQ_DFU_UPLOAD:
00231 #if (UDI_DFU_ATMEL_PROTOCOL_VERSION == DFU_ATMEL_PROTOCOL_VERSION_2)
00232 #  ifdef UDI_DFU_ATMEL_PROTOCOL_2_SPLIT_ERASE_CHIP
00233             Assert( !udi_dfu_atmel_erase_running );
00234 #  endif
00235 #endif
00236             Assert(DFU_STATE_DFUERROR != udi_dfu_atmel_status.bState);
00237             if (NULL != udi_dfu_atmel_upload_callback) {
00238                 return udi_dfu_atmel_upload_callback();
00239             }
00240         }
00241     }
00242     
00243 #if (UDI_DFU_ATMEL_PROTOCOL_VERSION == DFU_ATMEL_PROTOCOL_VERSION_2)
00244 #  ifdef UDI_DFU_ATMEL_PROTOCOL_2_SPLIT_ERASE_CHIP
00245     Assert( !udi_dfu_atmel_erase_running );
00246 #  endif
00247 #endif
00248     
00249     if (Udd_setup_is_out()) {
00250         // Requests Class Interface Set
00251         switch (udd_g_ctrlreq.req.bRequest) {
00252         case USB_REQ_DFU_CLRSTATUS:
00253             Assert(udd_g_ctrlreq.req.wValue==0);
00254             Assert(udd_g_ctrlreq.req.wLength==0);
00255             udi_dfu_atmel_reset_protocol();
00256             return true;
00257 
00258             // DNLOAD request including Atmel command fields
00259             // and data for write operation.
00260             // For read operation, the data are sent in the next UPLOAD request
00261         case USB_REQ_DFU_DNLOAD:
00262             // Check if a reset has been requested
00263             if (NULL != udi_dfu_atmel_reset_callback) {
00264                 // Remove force ISP before a reset CPU to start Application
00265                 isp_force_isp(false);
00266                 // Then this DNLOAD DFU request must be empty (0==wLength)
00267                 // and valid the reset application command.
00268                 // Valid SETUP request and reset application via request callback.
00269                 udd_g_ctrlreq.callback = udi_dfu_atmel_reset_cpu;
00270                 return true;
00271             }
00272 
00273             Assert(DFU_STATE_DFUERROR != udi_dfu_atmel_status.bState);
00274             Assert(udd_g_ctrlreq.req.wLength!=0);
00275 
00276             // The first packet contains the command
00277             // after this packet the over_under_run callback can be called 
00278             // if the Host want to send more data to device
00279             udd_set_setup_payload(
00280                 (uint8_t *) & udi_dfu_atmel_cmd,
00281                 sizeof(udi_dfu_atmel_cmd)),
00282             // Called when the first packet is received
00283             // before continuing DATA phase or start ZLP phase
00284             udd_g_ctrlreq.over_under_run = udi_dfu_atmel_cmd_decode;
00285             // Note udd_g_ctrlreq.callback is updated 
00286             // by udi_dfu_atmel_cmd_decode() before ZLP phase
00287             return true;
00288         }
00289     }
00290     // Unknown request
00291     udi_dfu_atmel_status.bStatus = DFU_STATUS_ERRSTALLEDPK;
00292     udi_dfu_atmel_status.bState = DFU_STATE_DFUERROR;
00293     return false;

00326 //! Called by over_under_run when the first data packet
00327 //! from a DNLOAD DFU request is received
00328 static bool udi_dfu_atmel_cmd_decode(void)
00329 {
00330     // By default no callback initialized
00331     // By default request states are success and finish
00332     udi_dfu_atmel_reset_protocol();
00333     udi_dfu_atmel_upload_callback = NULL;
00334 
00335     // To restart ISP in case of USB cable is unplug during program load
00336     isp_force_isp(true);
00337 
00338     // Decode Atmel command ID
00339     switch (CAT_CMD(udi_dfu_atmel_cmd.group, udi_dfu_atmel_cmd.cmd_id)) {
00340 
00341     case CAT_CMD(DFU_ATMEL_V2_CMD_GRP_DNLOAD,
00342             DFU_ATMEL_V2_CMD_PROGRAM_START):
00343         return udi_dfu_atmel_program();
00344 
00345     case CAT_CMD(DFU_ATMEL_V2_CMD_GRP_UPLOAD,
00346                      DFU_ATMEL_V2_CMD_READ_MEMORY):
00347         return udi_dfu_atmel_read();
00348     case CAT_CMD(DFU_ATMEL_V2_CMD_GRP_UPLOAD,
00349                      DFU_ATMEL_V2_CMD_BLANK_CHECK):
00350         return udi_dfu_atmel_blankcheck();
00351 
00352     case CAT_CMD(DFU_ATMEL_V2_CMD_GRP_EXEC,
00353             DFU_ATMEL_V2_CMD_ERASE):
00354         return udi_dfu_atmel_erase_chip();
00355     case CAT_CMD(DFU_ATMEL_V2_CMD_GRP_EXEC,
00356             DFU_ATMEL_V2_CMD_START_APPLI):
00357         udi_dfu_atmel_start();
00358         return true;
00359 
00360     case CAT_CMD(DFU_ATMEL_V2_CMD_GRP_SELECT,
00361             DFU_ATMEL_V2_CMD_SELECT_MEMORY):
00362         return udi_dfu_atmel_select_memory();
00363     }
00364     return false;   // Unknow command
00365 }

00367 static bool udi_dfu_atmel_program(void)
00368 {
00369     // Check list before start memory programmation
00370     if (udi_dfu_atmel_security && udi_dfu_atmel_mem_b_protected) {
00371         // Security enabled
00372         return udi_dfu_atmel_mem_protected();
00373     }
00374     if (!udi_dfu_atmel_mem_sel.fnct_write) {
00375         udi_dfu_atmel_status.bStatus = DFU_STATUS_ERRWRITE;
00376         udi_dfu_atmel_status.bState = DFU_STATE_DFUERROR;
00377         return false;   // Write memory not available
00378     }
00379     if (!udi_dfu_atmel_mem_getaddr(&udi_dfu_atmel_cmd.arg[0]))
00380         return false;   // Bad Range
00381     
00382     // Init buffer to fill during next DATA phase of request
00383     udd_set_setup_payload(
00384         udi_dfu_atmel_buf_trans,
00385         DFU_ATMEL_BUF_TRANS_SIZE);
00386     // Init callback called after buffer filled
00387     udd_g_ctrlreq.over_under_run = udi_dfu_atmel_mem_write;
00388     return true;
00389 }

00746 static void udi_dfu_atmel_mem_check(void)
00747 {
00748     uint8_t *ptr_buf;
00749     uint16_t packet_size;
00750     
00751     while (udi_dfu_atmel_mem_nb_data) {
00752         // Compute buffer to read
00753         packet_size = min(udi_dfu_atmel_mem_nb_data,
00754             DFU_ATMEL_BUF_TRANS_SIZE);
00755         udi_dfu_atmel_mem_nb_data -= packet_size;
00756

00758         udi_dfu_atmel_mem_sel.fnct_read(udi_dfu_atmel_buf_trans,
00759             udi_dfu_atmel_mem_add, packet_size);
00760 
00761         // Check buffer content
00762         ptr_buf = udi_dfu_atmel_buf_trans;
00763         while (packet_size--) {
00764             if (*ptr_buf++ != 0xFF) {
00765                 // Error, don't stall request but:
00766                 // Update DFU status
00767                 udi_dfu_atmel_status.bStatus = DFU_STATUS_ERRCHECK_ERASED;
00768                 // Send last address checked in next Upload command
00769                 udi_dfu_atmel_upload_callback = udi_dfu_atmel_mem_send_last_add;
00770                 return;
00771             }
00772             udi_dfu_atmel_mem_add++;
00773         }
00774     }
