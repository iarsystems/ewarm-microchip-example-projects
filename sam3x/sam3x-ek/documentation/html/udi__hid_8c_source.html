00125 static bool udi_hid_reqstdifaceget_descriptor(uint8_t *report_desc)
00126 {
00127     usb_hid_descriptor_t UDC_DESC_STORAGE *ptr_hid_desc;
00128 
00129     // Get the USB descriptor which is located after the interface descriptor
00130     // This descriptor must be the HID descriptor
00131     ptr_hid_desc = (usb_hid_descriptor_t UDC_DESC_STORAGE *) ((uint8_t *)
00132             udc_get_interface_desc() + sizeof(usb_iface_desc_t));
00133     if (USB_DT_HID != ptr_hid_desc->bDescriptorType)
00134         return false;
00135 
00136     // The SETUP request can ask for:
00137     // - an USB_DT_HID descriptor 
00138     // - or USB_DT_HID_REPORT descriptor
00139     // - or USB_DT_HID_PHYSICAL descriptor
00140     if (USB_DT_HID == (uint8_t) (udd_g_ctrlreq.req.wValue >> 8)) {
00141         // USB_DT_HID descriptor requested then send it
00142         udd_g_ctrlreq.payload = (uint8_t *) ptr_hid_desc;
00143         udd_g_ctrlreq.payload_size =
00144                 min(udd_g_ctrlreq.req.wLength,
00145                 ptr_hid_desc->bLength);
00146         return true;
00147     }
00148     // The HID_X descriptor requested must correspond to report type
00149     // included in the HID descriptor
00150     if (ptr_hid_desc->bRDescriptorType ==
00151             (uint8_t) (udd_g_ctrlreq.req.wValue >> 8)) {
00152         // Send HID Report descriptor given by high level
00153         udd_g_ctrlreq.payload = report_desc;
00154         udd_g_ctrlreq.payload_size =
00155                 min(udd_g_ctrlreq.req.wLength,
00156                 le16_to_cpu(ptr_hid_desc->wDescriptorLength));
00157         return true;
00158     }
00159     return false;
00160 }
