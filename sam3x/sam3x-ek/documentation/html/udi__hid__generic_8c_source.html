00213 static bool udi_hid_generic_setreport(void)
00214 {
00215     if ((USB_HID_REPORT_TYPE_FEATURE == (udd_g_ctrlreq.req.wValue >> 8))
00216             && (0 == (0xFF & udd_g_ctrlreq.req.wValue))
00217             && (sizeof(udi_hid_generic_report_feature) ==
00218                     udd_g_ctrlreq.req.wLength)) {
00219         // Feature type on report ID 0
00220         udd_g_ctrlreq.payload =
00221                 (uint8_t *) & udi_hid_generic_report_feature;
00222         udd_g_ctrlreq.callback = udi_hid_generic_setfeature_valid;
00223         udd_g_ctrlreq.payload_size =
00224                 sizeof(udi_hid_generic_report_feature);
00225         return true;
00226     }
00227     return false;
00228 }
