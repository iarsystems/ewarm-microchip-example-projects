00263 bool udi_hid_kbd_up(uint8_t key_id)
00264 {
00265     uint8_t i;
00266 
00267     irqflags_t flags = cpu_irq_save();
00268 
00269     // Fill report
00270     for (i = 2; i < UDI_HID_KBD_REPORT_SIZE; i++) {
00271         if (0 == udi_hid_kbd_report[i]) {
00272             // Already removed
00273             cpu_irq_restore(flags);
00274             return true;
00275         }
00276         if (key_id == udi_hid_kbd_report[i])
00277             break;
00278     }
00279     if (UDI_HID_KBD_REPORT_SIZE == i) {
00280         // Already removed
00281         cpu_irq_restore(flags);
00282         return true;
00283     }
00284     // Remove key and shift
00285     while (i < (UDI_HID_KBD_REPORT_SIZE - 1)) {
00286         udi_hid_kbd_report[i] = udi_hid_kbd_report[i + 1];
00287         i++;
00288     }
00289     udi_hid_kbd_report[UDI_HID_KBD_REPORT_SIZE - 1] = 0x00;
00290     udi_hid_kbd_b_report_valid = true;
00291 
00292     // Send report
00293     udi_hid_kbd_send_report();
00294 
00295     cpu_irq_restore(flags);
00296     return true;

00340 {

00350                             UDI_HID_KBD_REPORT_SIZE,
