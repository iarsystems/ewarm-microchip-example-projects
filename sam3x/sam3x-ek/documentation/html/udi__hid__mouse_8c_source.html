00264 static bool udi_hid_mouse_move(int8_t pos, uint8_t index_report)
00265 {
00266     int16_t s16_newpos;
00267 
00268     irqflags_t flags = cpu_irq_save();
00269 
00270     // Add position in HID mouse report
00271     s16_newpos = (int8_t) udi_hid_mouse_report[index_report];
00272     s16_newpos += pos;
00273     if ((-127 > s16_newpos) || (127 < s16_newpos)) {
00274         cpu_irq_restore(flags);
00275         return false;   // Overflow of report
00276     }
00277     udi_hid_mouse_report[index_report] = (uint8_t) s16_newpos;
00278 
00279     // Valid and send report
00280     udi_hid_mouse_b_report_valid = true;
00281     udi_hid_mouse_send_report();
00282 
00283     cpu_irq_restore(flags);
00284     return true;
00285 }

00300 static bool udi_hid_mouse_send_report(void)
00301 {
00302     if (udi_hid_mouse_report_trans_ongoing)
00303         return false;   // Transfer on going then send this one after transfer complete
00304 
00305     // Copy report on other array used only for transfer
00306     memcpy(udi_hid_mouse_report_trans, udi_hid_mouse_report,
00307             UDI_HID_MOUSE_REPORT_SIZE);
00308     memset(&udi_hid_mouse_report[1], 0, 3); // Keep status of btn for next report
00309     udi_hid_mouse_b_report_valid = false;
00310 
00311     // Send report
00312     udi_hid_mouse_report_trans_ongoing =
00313             udd_ep_run( UDI_HID_MOUSE_EP_IN,
00314                             false,
00315                             udi_hid_mouse_report_trans,
00316                             UDI_HID_MOUSE_REPORT_SIZE,
00317                             udi_hid_mouse_report_sent);
00318     return udi_hid_mouse_report_trans_ongoing;
00319 }
