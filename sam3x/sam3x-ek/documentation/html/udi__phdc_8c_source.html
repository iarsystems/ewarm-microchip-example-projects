00316 bool udi_phdc_senddata(udi_phdc_metadata_t * metadata,
00317         void (*callback) (uint16_t))
00318 {
00319     bool b_status;
00320     irqflags_t flags;
00321     uint8_t qos = metadata->qos;
00322 
00323     // Check qos: Only one bit must be set in qos
00324     if ((0 == qos) || (0 != (qos & (qos - 1))))
00325         return false;   // bad qos value
00326 
00327     // Check if qos is supported
00328     if (0 == (qos & UDI_PHDC_QOS_IN))
00329         return false;   // qos not supported
00330 
00331     if (udi_phdc_in_trans.b_run)
00332         return false;   // transfer already running
00333 
00334     // Init transfer
00335     udi_phdc_in_trans.metadata = metadata;
00336     udi_phdc_in_trans.metadata_pos = 0;
00337     udi_phdc_in_trans.callback = callback;
00338     udi_phdc_in_trans.b_preample_run = false;
00339 
00340     flags = cpu_irq_save();
00341 #if (UDI_PHDC_PREAMBLE_FEATURE == TRUE)
00342     // It is a bulk then check preample feature
00343     if ((USB_PHDC_QOS_LOW_GOOD != qos)
00344             && (udi_phdc_preample_feature)) {
00345         b_status = udi_phdc_send_preamplemsg();
00346     } else
00347 #endif
00348         b_status = udi_phdc_send_metadata();
00349     cpu_irq_restore(flags);
00350     return b_status;

00397 static bool udi_phdc_send_preamplemsg(void)
00398 {
00399     if ((UDI_PHDC_EP_SIZE_BULK_IN - 1 - sizeof(usb_phdc_metadata_msg_t))
00400             < udi_phdc_in_trans.metadata->opaque_size)
00401         return false;   // Opaque data too large
00402     // Fill preample message
00403     udi_phdc_in_trans.preample_header.bNumTransfers =
00404             1 + ((udi_phdc_in_trans.metadata->metadata_size -
00405                     1) / UDI_PHDC_EP_SIZE_BULK_IN);
00406     udi_phdc_in_trans.preample_header.bmLatencyReliability =
00407             udi_phdc_in_trans.metadata->qos;
00408     udi_phdc_in_trans.preample_header.bOpaqueDataSize =
00409             udi_phdc_in_trans.metadata->opaque_size;
00410     memcpy(udi_phdc_in_trans.preample_opaque_data,
00411             udi_phdc_in_trans.metadata->opaquedata,
00412             udi_phdc_in_trans.metadata->opaque_size);
00413     // Send preample message
00414     if (!udd_ep_run(UDI_PHDC_EP_BULK_IN,
00415                     false,
00416                     (uint8_t *) &
00417                     udi_phdc_in_trans.preample_header,
00418                     udi_phdc_in_trans.
00419                     preample_header.bOpaqueDataSize +
00420                     sizeof(usb_phdc_metadata_msg_t),
00421                     udi_phdc_preamplemsg_ack)) {
00422         return false;
00423     }
00424     udi_phdc_in_trans.b_run = true;
00425     udi_phdc_in_trans.b_preample_run = true;
00426     udi_phdc_holding_data |=
00427             1 << (UDI_PHDC_EP_SIZE_BULK_IN & USB_EP_ADDR_MASK);
00428     return true;
00429 }
