00057 /**
00058  * Starts a Usart master transfer. This is a non blocking function. It will
00059  * return as soon as the transfer is started.
00060  * Returns 0 if the transfer has been started successfully; otherwise returns
00061  * 1 is the driver is in use
00062  * \param pUsartd  Pointer to a usart instance.
00063  * \param pCommand Pointer to the uSART command to execute.
00064  */
00065 uint8_t USART_SPIDSendCommand(Usartd *pUsartd, SpidCmd *pCommand)
00066 {
00067     Usart *pUsart = pUsartd->pUsartHw;
00068 
00069     // Try to get the dataflash semaphore
00070     if (pUsartd->ucSemaphore == 0)
00071     {
00072         return 1;
00073     }
00074 
00075     pUsartd->ucSemaphore--;
00076 
00077     // Setup Force CS signal as dataflash required
00078     pUsart->US_CR |= US_CR_FCS;
00079 
00080     if (pUsartd->eTransferType == USART_PDC_TRANS_TYPE)
00081     {
00082         pUsartd->ucUsartTransEndSig = 0;
00083 
00084         pUsart->US_TPR = (uint32_t) pCommand->pCmd;
00085         pUsart->US_TCR = pCommand->cmdSize;
00086 
00087         pUsart->US_RPR = (uint32_t) pCommand->pCmd;
00088         pUsart->US_RCR = pCommand->cmdSize;
00089 
00090         pUsart->US_TNPR = (uint32_t) pCommand->pData;
00091         pUsart->US_TNCR = pCommand->dataSize;
00092 
00093         pUsart->US_RNPR = (uint32_t) pCommand->pData;
00094         pUsart->US_RNCR = pCommand->dataSize;
00095 
00096         // Enable PDC transfer
00097         pUsart->US_PTCR = US_PTCR_RXTEN | US_PTCR_TXTEN;
00098 
00099         // Enable buffer complete interrupt
00100         USART_EnableIt(pUsart, US_IER_RXBUFF);
00101 
00102         // Wait till command done
00103         while (!pUsartd->ucUsartTransEndSig);
00104     }
00105     else
00106     {
00107         uint32_t dw;
00108         /* Send dataflash command */
00109         for (dw=0; dw < pCommand->cmdSize; dw++ )
00110         {
00111             USART_Write(pUsart, pCommand->pCmd[dw], DELAY_TIME);
00112             USART_Read(pUsart, DELAY_TIME);
00113         }
00114 
00115         /* Send/receive the data */
00116         for (dw=0; dw < pCommand->dataSize; dw++ )
00117         {
00118             USART_Write(pUsart, (uint16_t)pCommand->pData[dw], DELAY_TIME);
00119             pCommand->pData[dw] = (uint8_t)USART_Read(pUsart, DELAY_TIME);
00120         }
00121     }
00122 
00123     return 0;
00124 }

00127  * The USART_SPIDHandler must be called by the Usart Interrupt Service Routine with the
00128  * corresponding Spi instance.
00129  * The USART_SPIDHandler will unlock the Spi semaphore and invoke the upper application

00131  * \param pUsartd  Pointer to a pUsartd instance.

00133 void USART_SPIDHandler(Usartd *pUsartd)
00134 {
00135     SpidCmd *pUsartCmd = pUsartd->pCurrentCommand;
00136     Usart *pUsart = pUsartd->pUsartHw;
00137     volatile uint32_t dwStatus;
00138 
00139     dwStatus = USART_GetStatus(pUsart);
00140 
00141     // Check the status register
00142     if ((dwStatus & US_CSR_RXBUFF) || (pUsartd->eTransferType == USART_FIFO_TRANS_TYPE))
00143     {
00144         // Disable buffer complete interrupt
00145         USART_DisableIt(pUsart, US_IDR_RXBUFF);
00146 
00147         // Release the chip select signal
00148         pUsart->US_CR |= US_CR_RCS;
00149 
00150         // Release the dataflash semaphore
00151         pUsartd->ucSemaphore++;
00152 
00153         // Invoke the callback associated with the current command
00154         if (pUsartCmd && pUsartCmd->callback) {
00155 
00156             pUsartCmd->callback(0, pUsartCmd->pArgument);
00157         }
00158     }
