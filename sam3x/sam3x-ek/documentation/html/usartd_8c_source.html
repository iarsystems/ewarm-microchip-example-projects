00068 ISR(usart_interrupt, AVR32_USART1_IRQ_GROUP, 3)
00069 {
00070     uint32_t dwCsr = USART_GetStatus(USART);
00071     if (dwCsr & US_CSR_RXRDY) {
00072         // Data received
00073         int value;
00074         bool b_error = (dwCsr & (US_CSR_FRAME | US_CSR_TIMEOUT | US_CSR_PARE));
00075         value = USART->US_RHR;
00076         if (b_error) {
00077             USART->US_CR = US_CR_RSTRX;
00078         }
00079         main_usart_rx_occur(b_error, value);
00080         return;
00081     }
00082 
00083     if (dwCsr & US_CSR_TXRDY) {
00084         uint8_t value_tx;
00085         if (main_usart_tx_free(&value_tx)) {
00086             // Transmit next data
00087             USART->US_THR = value_tx;
00088         } else {
00089             // Fifo empty then Stop UART transmission
00090             USART_DisableIt(USART, US_IDR_TXRDY);
00091         }
00092     }
00093 }
