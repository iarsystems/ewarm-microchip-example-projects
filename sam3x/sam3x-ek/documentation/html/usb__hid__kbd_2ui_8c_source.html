00175 void ui_process(uint16_t framenumber)
00176 {
00177     bool b_btn_states[2], sucess;
00178     static bool sequence_running = false;
00179     static uint8_t u8_sequence_pos = 0;
00180     uint8_t u8_value;
00181     static uint16_t cpt_sof = 0;
00182 
00183     if ((framenumber % 1000) == 0) {
00184         LED_On(LED_HID);
00185     }
00186     if ((framenumber % 1000) == 500) {
00187         LED_Off(LED_HID);
00188     }
00189 
00190     // Scan process running each 2ms
00191     cpt_sof++;
00192     if ((cpt_sof%2)==0)
00193         return;
00194 
00195     // Scan buttons on switch 0 to send keys sequence
00196     b_btn_states[PUSHBUTTON_LEFT] = (0 == PIO_PinGet(&pinsButtons[PUSHBUTTON_LEFT])) ? true : false;
00197     if (b_btn_states[PUSHBUTTON_LEFT] != btn_last_states[PUSHBUTTON_LEFT]) {
00198         btn_last_states[PUSHBUTTON_LEFT] = b_btn_states[PUSHBUTTON_LEFT];
00199         sequence_running = true;
00200     }
00201 
00202     // Sequence process running each 150ms
00203     if (150 > cpt_sof)
00204         return;
00205     cpt_sof = 0;
00206 
00207     if (sequence_running) {
00208         // Send next key
00209         u8_value = ui_sequence[u8_sequence_pos].u8_value;
00210         if (ui_sequence[u8_sequence_pos].b_modifier) {
00211             if (ui_sequence[u8_sequence_pos].b_down) {
00212                 sucess = udi_hid_kbd_modifier_down(u8_value);
00213             }else{
00214                 sucess = udi_hid_kbd_modifier_up(u8_value);
00215             }
00216         }else{
00217             if (ui_sequence[u8_sequence_pos].b_down) {
00218                 sucess = udi_hid_kbd_down(u8_value);
00219             }else{
00220                 sucess = udi_hid_kbd_up(u8_value);
00221             }
00222         }
00223         if (!sucess)
00224             return; // Retry it on next schedule
00225   
00226         // Valid sequence position
00227         u8_sequence_pos++;
00228         if (u8_sequence_pos>=sizeof(ui_sequence)/sizeof(ui_sequence[0])) {
00229             u8_sequence_pos=0;
00230             sequence_running=false;
00231         }
00232     }
00233     else {
00234         b_btn_states[PUSHBUTTON_RIGHT] = (0 == PIO_PinGet(&pinsButtons[PUSHBUTTON_RIGHT]));
00235         if (b_btn_states[PUSHBUTTON_RIGHT] != btn_last_states[PUSHBUTTON_RIGHT]) {
00236             btn_last_states[PUSHBUTTON_RIGHT] = b_btn_states[PUSHBUTTON_RIGHT];
00237             if (b_btn_states[PUSHBUTTON_RIGHT])
00238                 udi_hid_kbd_down(HID_CAPS_LOCK);
00239             else
00240                 udi_hid_kbd_up(HID_CAPS_LOCK);
00241         }
00242     }
00243 }
