00866         {
00867           status = PIPE_DELAY_TIMEOUT;
00868           Host_reset_pipe(pipe);
00869           goto host_get_data_end;

00871       }

00873       {
00874         if ((sav_glob_int_en = Is_global_interrupt_enabled())) Disable_global_interrupt();

01050   U16 n, i;

01052

01054   TRACE_OTG("pipe= %d\n\r", pipe);

01056   {
01057     TRACE_OTG("pipe_error\n\r");

01059     it_pipe_str[pipe].enable = FALSE;

01065 
01066   if (Is_host_stall(pipe))          // STALL management
01067   {
01068     TRACE_OTG("pipe_stall\n\r");
01069     it_pipe_str[pipe].status = PIPE_STALL;
01070     it_pipe_str[pipe].enable = FALSE;
01071     Host_reset_pipe(pipe);
01072     callback = TRUE;
01073     goto usb_pipe_interrupt_end;
01074   }
01075 
01076   #if NAK_TIMEOUT_ENABLE == ENABLE
01077   if (Is_host_nak_received(pipe))   // NAK received
01078   {
01079     TRACE_OTG("pipe_NAK\n\r");
01080     Host_ack_nak_received(pipe);
01081     // Check if NAK time-out error occurs (not for interrupt pipes)
01082     if (!--it_pipe_str[pipe].nak_timeout && Host_get_pipe_type(pipe) != TYPE_INTERRUPT)
01083     {
01084       it_pipe_str[pipe].status = PIPE_NAK_TIMEOUT;
01085       it_pipe_str[pipe].enable = FALSE;
01086       Host_reset_pipe(pipe);
01087       callback = TRUE;
01088       goto usb_pipe_interrupt_end;
01089     }
01090   }
01091   #endif
01092 
01093   if (Is_host_in_received(pipe))    // Pipe IN reception?
01094   {
01095     TRACE_OTG("pipe_IN\n\r");
01096     ptr_buf = (U8 *)it_pipe_str[pipe].ptr_buf + it_pipe_str[pipe].nb_byte_processed;  // Build pointer to data buffer
01097     n = it_pipe_str[pipe].nb_byte_to_process - it_pipe_str[pipe].nb_byte_processed; // Remaining data bytes
01098     Host_freeze_pipe(pipe);
01099     Host_reset_pipe_fifo_access(pipe);
01100     i = Host_get_pipe_size(pipe) - Host_byte_count(pipe);
01101     n = host_read_p_rxpacket(pipe, ptr_buf, n, NULL);
01102     it_pipe_str[pipe].nb_byte_processed = it_pipe_str[pipe].nb_byte_to_process - n;
01103     if (Host_byte_count(pipe))      // More bytes received than expected
01104     {
01105       //! @todo Error code management
01106     }
01107     else if (i)                     // Short packet with nb bytes received <= expected
01108     {
01109       n = 0;
01110     }
01111     Host_ack_in_received(pipe);
01112     if (n)                          // Still data to process
01113     {
01114       Host_free_in(pipe);
01115       Host_unfreeze_pipe(pipe);     // Request another IN transfer
01116       private_sof_counter = 0;      // Reset the counter in SOF detection subroutine
01117       it_pipe_str[pipe].timeout = 0;  // Reset time-out
01118       it_pipe_str[pipe].nak_timeout = NAK_RECEIVE_TIMEOUT;
01119     }
01120     else                            // End of transfer
01121     {
01122       it_pipe_str[pipe].enable = FALSE;
01123       it_pipe_str[pipe].status = PIPE_GOOD;
01124       Host_reset_pipe(pipe);
01125       callback = TRUE;
01126     }
01127   }
01128 
01129   if (Is_host_out_ready(pipe))      // Pipe OUT sent?
01130   {
01131     TRACE_OTG("pipe_OUT\n\r");
01132     Host_ack_out_ready(pipe);
01133     it_pipe_str[pipe].nb_byte_processed += it_pipe_str[pipe].nb_byte_on_going;
01134     it_pipe_str[pipe].nb_byte_on_going = 0;
01135     ptr_buf = (U8 *)it_pipe_str[pipe].ptr_buf + it_pipe_str[pipe].nb_byte_processed;  // Build pointer to data buffer
01136     n = it_pipe_str[pipe].nb_byte_to_process - it_pipe_str[pipe].nb_byte_processed; // Remaining data bytes
01137     if (n)                          // Still data to process
01138     {
01139       Host_unfreeze_pipe(pipe);
01140       // Prepare data to be sent
01141       Host_reset_pipe_fifo_access(pipe);
01142       it_pipe_str[pipe].nb_byte_on_going = n - host_write_p_txpacket(pipe, ptr_buf, n, NULL);
01143       private_sof_counter = 0;      // Reset the counter in SOF detection subroutine
01144       it_pipe_str[pipe].timeout = 0;  // Refresh time-out counter
01145       it_pipe_str[pipe].nak_timeout = NAK_SEND_TIMEOUT;
01146       Host_send_out(pipe);          // Send the USB frame
01147     }
01148     else                            // End of transfer
01149     {
01150       it_pipe_str[pipe].enable = FALSE; // Tranfer end
01151       it_pipe_str[pipe].status = PIPE_GOOD; // Status OK
01152       Host_reset_pipe(pipe);
01153       callback = TRUE;
01154     }
01155   }
01156 
01157 usb_pipe_interrupt_end:
01158   if (!is_any_interrupt_pipe_active() && !g_sav_int_sof_enable) // If no more transfer is armed
01159   {
01160     Host_disable_sof_interrupt();
01161   }
01162   if (callback)                     // Any call-back function to perform?
01163   {
