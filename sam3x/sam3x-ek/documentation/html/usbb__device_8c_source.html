00059 /**
00060  * \ingroup usb_device_group
00061  * \defgroup udd_group USB Device Driver (UDD)
00062  *  
00063  * \section USBB_CONF USBB Custom configuration
00064  * The following USBB driver configuration must be included in the conf_usb.h
00065  * file of the application.
00066  *
00067  * UDD_USB_INT_LEVEL<br>
00068  * Option to change the interrupt priority (0 to 3) by default 0 (recommended).
00069  * 
00070  * UDD_ISOCHRONOUS_NB_BANK<br>
00071  * Feature to reduce or increase isochronous endpoints buffering (1 to 2).
00072  * Default value 2.
00073  * 
00074  * UDD_BULK_NB_BANK<br>
00075  * Feature to reduce or increase bulk endpoints buffering (1 to 2).
00076  * Default value 2.
00077  * 
00078  * UDD_INTERRUPT_NB_BANK<br>
00079  * Feature to reduce or increase interrupt endpoints buffering (1 to 2).
00080  * Default value 1.
00081  *  
00082  * \section Callbacks management
00083  * The USB driver is fully managed by interrupt and does not request periodique
00084  * task. Thereby, the USB events use callbacks to transfer the information.
00085  * The callbacks are declared in static during compilation or in variable during
00086  * code execution.
00087  *  
00088  * Static declarations defined in conf_usb.h:
00089  * - UDC_VBUS_EVENT(bool b_present)<br>
00090  *   To signal Vbus level change
00091  * - UDC_SUSPEND_EVENT()<br>
00092  *   Called when USB bus enter in suspend mode
00093  * - UDC_RESUME_EVENT()<br>
00094  *   Called when USB bus is wakeup
00095  * - UDC_SOF_EVENT()<br>
00096  *   Called for each received SOF, Note: Each 1ms in HS/FS mode only.
00097  *   
00098  * Dynamic callbacks, called "endpoint job" , are registered
00099  * in udd_ep_job_t structure via the following functions:
00100  * - udd_ep_run()<br>
00101  *   To call it when a transfer is finish
00102  * - udd_ep_wait_stall_clear()<br>
00103  *   To call it when a endpoint halt is disabled
00104  *   
00105  * \section Power mode management
00106  * The Sleep modes authorized :
00107  * - in USB IDLE state, the USBB needs of USB clock and authorizes up to IDLE mode 
00108  * - in USB SUSPEND state, the USBB no needs USB clock but requests a minimum 
00109  *   clock restart timing. Thus, it authorizes up to STATIC or STANDBY mode.
00110  * - VBUS monitoring used in USB Self-Power mode authorizes up to STOP mode
00111  * 
00112  * The USBB_SLEEP_MODE_USB_IDLE equals SLEEPMGR_IDLE.
00113  *
00114  * The USBB_SLEEP_MODE_USB_SUSPEND depends on USB Power mode, 
00115  * USB clock startup timing and USB Speed mode:
00116  * | Power Mode | Speed mode | Clock Startup | Sleep mode authorized |
00117  * | Self-Power | LS, FS, HS | X             | SLEEPMGR_STOP        |
00118  * | Bus-Power  | LS, FS     | >10ms         | SLEEPMGR_STDBY       |
00119  * | Bus-Power  | LS, FS     | <=10ms        | SLEEPMGR_STATIC      |
00120  * | Bus-Power  | HS         | >3ms          | SLEEPMGR_STDBY       |
00121  * | Bus-Power  | HS         | <=3ms         | SLEEPMGR_STATIC      |
00122  *
00123  * @{
00124  */

00821 bool udd_ep_run(udd_ep_id_t ep, bool b_shortpacket,
00822         uint8_t * buf, iram_size_t buf_size,
00823         udd_callback_trans_t callback)
00824 {
00825     bool b_dir_in;
00826     uint32_t udd_dma_ctrl = 0;
00827     udd_ep_job_t *ptr_job;
00828     irqflags_t flags;
00829 
00830     b_dir_in = (USB_EP_DIR_IN == (ep & USB_EP_DIR_IN));
00831     ep &= USB_EP_ADDR_MASK;
00832     if (USB_DEVICE_MAX_EP < ep)
00833         return false;
00834 
00835     // Get job about endpoint
00836     ptr_job = &udd_ep_job[ep - 1];
00837 
00838     if ((!Is_udd_endpoint_enabled(ep))
00839             || Is_udd_endpoint_stall_requested(ep)
00840             || ptr_job->stall_requested)
00841         return false;   // Endpoint is halted
00842 
00843     flags = cpu_irq_save();
00844     if (ptr_job->busy == true) {
00845         cpu_irq_restore(flags);
00846         return false;   // Job already on going
00847     }
00848     ptr_job->busy = true;
00849     cpu_irq_restore(flags);
00850 
00851     // The USBB supports a maximum transfer size of 64KB
00852     if (0x10000 <= buf_size) {
00853         // Transfer size = 64KB
00854         ptr_job->buf_size = 0x10000;
00855         buf_size = 0;
00856     } else {
00857         ptr_job->buf_size = buf_size;
00858         if (b_dir_in && (0 != buf_size % udd_get_endpoint_size(ep))) {
00859             // Force short packet option to send a shortpacket on IN,
00860             // else the DMA transfer is accepted and interrupt DMA valid but nothing is sent.
00861             b_shortpacket = true;
00862         }
00863     }
00864     ptr_job->buf = buf;
00865     ptr_job->call_trans = callback;
00866 
00867   #ifdef UDD_EP_FIFO_SUPPORTED
00868     // No DMA support
00869     if (!Is_udd_endpoint_dma_supported(ep)) {
00870         //printf("fifoEP%d.%c\n\r", ep, b_dir_in ? 'i':'o');
00871         ptr_job->buf_cnt = 0;
00872         //ptr_job->need_zlp = b_shortpacket;
00873         if (b_dir_in && buf_size == 0 && buf == NULL)
00874             ptr_job->need_zlp = true; // Only accept a forced ZLP
00875         flags = cpu_irq_save();
00876         udd_enable_endpoint_interrupt(ep);
00877         if (b_dir_in) {
00878             udd_ep_in_sent(ep);
00879             udd_enable_in_send_interrupt(ep);
00880         }
00881         else {
00882             udd_enable_out_received_interrupt(ep);
00883         }
00884         cpu_irq_restore(flags);
00885         return true;
00886     }
00887   #endif
00888 
00889     // Start USB DMA to fill or read fifo of the selected endpoint
00890     udd_endpoint_dma_set_addr(ep, (U32) buf);
00891     if (b_shortpacket) {
00892         if (b_dir_in) {
00893             udd_dma_ctrl = UOTGHS_DEVDMACONTROL_END_B_EN;
00894         } else {
00895             udd_dma_ctrl = UOTGHS_DEVDMACONTROL_END_TR_IT
00896                     |
00897                     UOTGHS_DEVDMACONTROL_END_TR_EN;
00898         }
00899     }
00900     udd_dma_ctrl |= (buf_size <<
00901             UOTGHS_DEVDMACONTROL_BUFF_LENGTH_Pos)
00902             & UOTGHS_DEVDMACONTROL_BUFF_LENGTH_Msk;
00903     udd_dma_ctrl |= UOTGHS_DEVDMACONTROL_END_BUFFIT |
00904             UOTGHS_DEVDMACONTROL_CHANN_ENB;
00905     udd_enable_endpoint_bank_autoswitch(ep);
00906     udd_endpoint_dma_set_control(ep, udd_dma_ctrl);
00907     flags = cpu_irq_save();
00908     udd_enable_endpoint_dma_interrupt(ep);
00909     cpu_irq_restore(flags);
00910 
00911     return true;
00912 }

01152 static void udd_ctrl_in_sent(void)
01153 {
01154     uint16_t nb_remain;
01155     uint8_t i;
01156     uint8_t *ptr_dest, *ptr_src;
01157     irqflags_t flags;
01158 
01159     flags = cpu_irq_save();
01160     udd_disable_in_send_interrupt(0);
01161     cpu_irq_restore(flags);
01162 
01163     if (UDD_EPCTRL_HANDSHAKE_WAIT_IN_ZLP == udd_ep_control_state) {
01164         // ZLP on IN is sent, then valid end of setup request
01165         udd_ctrl_endofrequest();
01166         // Reinitializes control endpoint management
01167         udd_ctrl_init();
01168         return;
01169     }
01170     Assert(udd_ep_control_state == UDD_EPCTRL_DATA_IN);
01171 
01172     nb_remain = udd_g_ctrlreq.payload_size - udd_ctrl_payload_nb_trans;
01173     if (0 == nb_remain) {
01174         // All content of current buffer payload are sent
01175         if (!udd_ctrl_payload_need_in_zlp) {
01176             // It is the end of data phase, because the last data packet is a short packet
01177             // then generate an OUT ZLP for handshake phase.
01178             udd_ctrl_send_zlp_out();
01179             return;
01180         }
01181         if ((udd_g_ctrlreq.req.wLength > (udd_ctrl_prev_payload_nb_trans
01182                                 +
01183                                 udd_g_ctrlreq.
01184                                 payload_size))
01185                 || (!udd_g_ctrlreq.over_under_run)
01186                 || (!udd_g_ctrlreq.over_under_run())) {
01187             // Underrun or data packet complette than send zlp on IN (note don't change DataToggle)
01188             udd_ctrl_payload_need_in_zlp = false;
01189             // nb_remain==0 allows to send a IN ZLP
01190         } else {
01191             // A new payload buffer is given
01192             // Update number of total data sending by previous playlaod buffer
01193             udd_ctrl_prev_payload_nb_trans +=
01194                     udd_ctrl_payload_nb_trans;
01195             // Update maangement of current playoad transfer
01196             udd_ctrl_payload_nb_trans = 0;
01197             nb_remain = udd_g_ctrlreq.payload_size;
01198             // Compute if an IN ZLP must be send after IN data
01199             udd_ctrl_payload_need_in_zlp =
01200                     ((udd_g_ctrlreq.payload_size %
01201                             USB_DEVICE_EP_CTRL_SIZE)
01202                     == 0);
01203         }
01204     }
01205     // Continue transfer and send next data
01206     if (nb_remain > USB_DEVICE_EP_CTRL_SIZE) {
01207         nb_remain = USB_DEVICE_EP_CTRL_SIZE;
01208     }
01209     // Fill buffer of endpoint control
01210     ptr_dest = (uint8_t *) & udd_get_endpoint_fifo_access(0, 8);
01211     ptr_src = udd_g_ctrlreq.payload + udd_ctrl_payload_nb_trans;
01212     //** Critical section
01213     // Only in case of DATA IN phase abort without USB Reset signal after.
01214     // The IN data don't must be writed in endpoint 0 DPRAM during
01215     // a next setup reception in same endpoint 0 DPRAM.
