00030 /**
00031    \mainpage CMSIS DSP Software Library
00032    *
00033    * <b>Introduction</b>
00034    *
00035    * This user manual describes the CMSIS DSP software library, 
00036    * a suite of common signal processing functions for use on Cortex-M processor based devices.
00037    *
00038    * The library is divided into a number of modules each covering a specific category:
00039    * - Basic math functions
00040    * - Fast math functions
00041    * - Complex math functions
00042    * - Filters
00043    * - Matrix functions
00044    * - Transforms
00045    * - Motor control functions
00046    * - Statistical functions
00047    * - Support functions
00048    * - Interpolation functions
00049    *
00050    * The library has separate functions for operating on 8-bit integers, 16-bit integers,
00051    * 32-bit integer and 32-bit floating-point values. 
00052    *
00053    * <b>Processor Support</b>
00054    *
00055    * The library is completely written in C and is fully CMSIS compliant. 
00056    * High performance is achieved through maximum use of Cortex-M4 intrinsics. 
00057    *
00058    * The supplied library source code also builds and runs on the Cortex-M3 and Cortex-M0 processor,
00059    * with the DSP intrinsics being emulated through software. 
00060    *
00061    *
00062    * <b>Toolchain Support</b>
00063    *
00064    * The library has been developed and tested with MDK-ARM version 4.21. 
00065    * The library is being tested in GCC and IAR toolchains and updates on this activity will be made available shortly.
00066    *
00067    * <b>Using the Library</b>
00068    *
00069    * The library installer contains prebuilt versions of the libraries in the <code>Lib</code> folder.
00070    * - arm_cortexM4lf_math.lib (Little endian and Floating Point Unit on Cortex-M4)
00071    * - arm_cortexM4bf_math.lib (Big endian and Floating Point Unit on Cortex-M4)
00072    * - arm_cortexM4l_math.lib (Little endian on Cortex-M4)
00073    * - arm_cortexM4b_math.lib (Big endian on Cortex-M4)
00074    * - arm_cortexM3l_math.lib (Little endian on Cortex-M3)
00075    * - arm_cortexM3b_math.lib (Big endian on Cortex-M3)
00076    * - arm_cortexM0l_math.lib (Little endian on Cortex-M0)
00077    * - arm_cortexM0b_math.lib (Big endian on Cortex-M3)
00078    *
00079    * The library functions are declared in the public file <code>arm_math.h</code> which is placed in the <code>Include</code> folder.
00080    * Simply include this file and link the appropriate library in the application and begin calling the library functions. The Library supports single 
00081    * public header file <code> arm_math.h</code> for Cortex-M4/M3/M0 with little endian and big endian. Same header file will be used for floating point unit(FPU) variants. 
00082    * Define the appropriate pre processor MACRO ARM_MATH_CM4 or  ARM_MATH_CM3 or 
00083    * ARM_MATH_CM0 depending on the target processor in the application.
00084    *

00155 /**
00156  * @defgroup groupMatrix Matrix Functions
00157  *
00158  * This set of functions provides basic matrix math operations.
00159  * The functions operate on matrix data structures.  For example,
00160  * the type
00161  * definition for the floating-point matrix structure is shown
00162  * below:
00163  * <pre>
00164  *     typedef struct
00165  *     {
00166  *       uint16_t numRows;     // number of rows of the matrix.
00167  *       uint16_t numCols;     // number of columns of the matrix.
00168  *       float32_t *pData;     // points to the data of the matrix.
00169  *     } arm_matrix_instance_f32;
00170  * </pre>
00171  * There are similar definitions for Q15 and Q31 data types.
00172  *
00173  * The structure specifies the size of the matrix and then points to
00174  * an array of data.  The array is of size <code>numRows X numCols</code>
00175  * and the values are arranged in row order.  That is, the
00176  * matrix element (i, j) is stored at:
00177  * <pre>
00178  *     pData[i*numCols + j]
00179  * </pre>
00180  *
00181  * \par Init Functions
00182  * There is an associated initialization function for each type of matrix
00183  * data structure.
00184  * The initialization function sets the values of the internal structure fields.
00185  * Refer to the function <code>arm_mat_init_f32()</code>, <code>arm_mat_init_q31()</code>
00186  * and <code>arm_mat_init_q15()</code> for floating-point, Q31 and Q15 types,  respectively.
00187  *
00188  * \par
00189  * Use of the initialization function is optional. However, if initialization function is used
00190  * then the instance structure cannot be placed into a const data section.
00191  * To place the instance structure in a const data
00192  * section, manually initialize the data structure.  For example:
00193  * <pre>
00194  * <code>arm_matrix_instance_f32 S = {nRows, nColumns, pData};</code>
00195  * <code>arm_matrix_instance_q31 S = {nRows, nColumns, pData};</code>
00196  * <code>arm_matrix_instance_q15 S = {nRows, nColumns, pData};</code>
00197  * </pre>
00198  * where <code>nRows</code> specifies the number of rows, <code>nColumns</code>
00199  * specifies the number of columns, and <code>pData</code> points to the
00200  * data array.
00201  *
00202  * \par Size Checking
00203  * By default all of the matrix functions perform size checking on the input and
00204  * output matrices.  For example, the matrix addition function verifies that the
00205  * two input matrices and the output matrix all have the same number of rows and
00206  * columns.  If the size check fails the functions return:
00207  * <pre>
00208  *     ARM_MATH_SIZE_MISMATCH
00209  * </pre>
00210  * Otherwise the functions return
00211  * <pre>
00212  *     ARM_MATH_SUCCESS
00213  * </pre>
00214  * There is some overhead associated with this matrix size checking.
00215  * The matrix size checking is enabled via the #define
00216  * <pre>
00217  *     ARM_MATH_MATRIX_CHECK
00218  * </pre>
00219  * within the library project settings.  By default this macro is defined
00220  * and size checking is enabled.  By changing the project settings and
00221  * undefining this macro size checking is eliminated and the functions
00222  * run a bit faster.  With size checking disabled the functions always
00223  * return <code>ARM_MATH_SUCCESS</code>.
00224  */

00463   static __INLINE uint32_t arm_recip_q31(
00464                      q31_t in,
00465                      q31_t * dst,
00466                      q31_t * pRecipTable)
00467   {
00468 
00469     uint32_t out, tempVal;
00470     uint32_t index, i;
00471     uint32_t signBits;
00472 
00473     if(in > 0)
00474       {
00475     signBits = __CLZ(in) - 1;
00476       }
00477     else
00478       {
00479     signBits = __CLZ(-in) - 1;
00480       }
00481 
00482     /* Convert input sample to 1.31 format */
00483     in = in << signBits;
00484 
00485     /* calculation of index for initial approximated Val */
00486     index = (uint32_t) (in >> 24u);
00487     index = (index & INDEX_MASK);
00488 
00489     /* 1.31 with exp 1 */
00490     out = pRecipTable[index];
00491 
00492     /* calculation of reciprocal value */
00493     /* running approximation for two iterations */
00494     for (i = 0u; i < 2u; i++)
00495       {
00496     tempVal = (q31_t) (((q63_t) in * out) >> 31u);
00497     tempVal = 0x7FFFFFFF - tempVal;
00498     /*      1.31 with exp 1 */
00499     //out = (q31_t) (((q63_t) out * tempVal) >> 30u);
00500     out = (q31_t) clip_q63_to_q31(((q63_t) out * tempVal) >> 30u);
00501       }
00502 
00503     /* write output */
00504     *dst = out;
00505 
00506     /* return num of signbits of out = 1/in value */
00507     return (signBits + 1u);

00509   }

04641   /**
04642    * @defgroup PID PID Motor Control
04643    *
04644    * A Proportional Integral Derivative (PID) controller is a generic feedback control 
04645    * loop mechanism widely used in industrial control systems.
04646    * A PID controller is the most commonly used type of feedback controller.
04647    *
04648    * This set of functions implements (PID) controllers
04649    * for Q15, Q31, and floating-point data types.  The functions operate on a single sample
04650    * of data and each call to the function returns a single processed value.
04651    * <code>S</code> points to an instance of the PID control data structure.  <code>in</code>
04652    * is the input sample value. The functions return the output value.
04653    *
04654    * \par Algorithm:
04655    * <pre>
04656    *    y[n] = y[n-1] + A0 * x[n] + A1 * x[n-1] + A2 * x[n-2]
04657    *    A0 = Kp + Ki + Kd
04658    *    A1 = (-Kp ) - (2 * Kd )
04659    *    A2 = Kd  </pre>
04660    *
04661    * \par
04662    * where \c Kp is proportional constant, \c Ki is Integral constant and \c Kd is Derivative constant
04663    * 
04664    * \par 
04665    * \image html PID.gif "Proportional Integral Derivative Controller" 
04666    *
04667    * \par
04668    * The PID controller calculates an "error" value as the difference between
04669    * the measured output and the reference input.
04670    * The controller attempts to minimize the error by adjusting the process control inputs.  
04671    * The proportional value determines the reaction to the current error, 
04672    * the integral value determines the reaction based on the sum of recent errors, 
04673    * and the derivative value determines the reaction based on the rate at which the error has been changing.
04674    *
04675    * \par Instance Structure 
04676    * The Gains A0, A1, A2 and state variables for a PID controller are stored together in an instance data structure. 
04677    * A separate instance structure must be defined for each PID Controller. 
04678    * There are separate instance structure declarations for each of the 3 supported data types. 
04679    * 
04680    * \par Reset Functions 
04681    * There is also an associated reset function for each data type which clears the state array. 
04682    *
04683    * \par Initialization Functions 
04684    * There is also an associated initialization function for each data type. 
04685    * The initialization function performs the following operations: 
04686    * - Initializes the Gains A0, A1, A2 from Kp,Ki, Kd gains.
04687    * - Zeros out the values in the state buffer.   
04688    * 
04689    * \par 
04690    * Instance structure cannot be placed into a const data section and it is recommended to use the initialization function. 
04691    *
04692    * \par Fixed-Point Behavior 
04693    * Care must be taken when using the fixed-point versions of the PID Controller functions. 
04694    * In particular, the overflow and saturation behavior of the accumulator used in each function must be considered. 
04695    * Refer to the function specific documentation below for usage guidelines. 
04696    */
04697

04704    * @brief  Process function for the floating-point PID Control.

04707    * @return out processed output sample.
04708    */

04711   static __INLINE float32_t arm_pid_f32(
04712                     arm_pid_instance_f32 * S,

04718     out = (S->A0 * in) +
04719       (S->A1 * S->state[0]) + (S->A2 * S->state[1]) + (S->state[2]);
04720

04723     S->state[0] = in;

04731   /**
04732    * @brief  Process function for the Q31 PID Control.
04733    * @param[in,out] *S points to an instance of the Q31 PID Control structure
04734    * @param[in] in input sample to process
04735    * @return out processed output sample.
04736    *
04737    * <b>Scaling and Overflow Behavior:</b> 
04738    * \par 
04739    * The function is implemented using an internal 64-bit accumulator. 
04740    * The accumulator has a 2.62 format and maintains full precision of the intermediate multiplication results but provides only a single guard bit. 
04741    * Thus, if the accumulator result overflows it wraps around rather than clip. 
04742    * In order to avoid overflows completely the input signal must be scaled down by 2 bits as there are four additions. 
04743    * After all multiply-accumulates are performed, the 2.62 accumulator is truncated to 1.32 format and then saturated to 1.31 format. 
04744    */
04745 
04746   static __INLINE q31_t arm_pid_q31(
04747                     arm_pid_instance_q31 * S,
04748                     q31_t in)
04749   {
04750     q63_t acc;
04751     q31_t out;
04752 
04753     /* acc = A0 * x[n]  */
04754     acc = (q63_t) S->A0 * in;
04755 
04756     /* acc += A1 * x[n-1] */
04757     acc += (q63_t) S->A1 * S->state[0];
04758 
04759     /* acc += A2 * x[n-2]  */
04760     acc += (q63_t) S->A2 * S->state[1];
04761 
04762     /* convert output to 1.31 format to add y[n-1] */
04763     out = (q31_t) (acc >> 31u);
04764 
04765     /* out += y[n-1] */
04766     out += S->state[2];
04767 
04768     /* Update state */
04769     S->state[1] = S->state[0];
04770     S->state[0] = in;
04771     S->state[2] = out;
04772 
04773     /* return to application */
04774     return (out);
04775 
04776   }

04778   /**
04779    * @brief  Process function for the Q15 PID Control.
04780    * @param[in,out] *S points to an instance of the Q15 PID Control structure
04781    * @param[in] in input sample to process
04782    * @return out processed output sample.
04783    *
04784    * <b>Scaling and Overflow Behavior:</b> 
04785    * \par 
04786    * The function is implemented using a 64-bit internal accumulator. 
04787    * Both Gains and state variables are represented in 1.15 format and multiplications yield a 2.30 result. 
04788    * The 2.30 intermediate results are accumulated in a 64-bit accumulator in 34.30 format. 
04789    * There is no risk of internal overflow with this approach and the full precision of intermediate multiplications is preserved. 
04790    * After all additions have been performed, the accumulator is truncated to 34.15 format by discarding low 15 bits. 
04791    * Lastly, the accumulator is saturated to yield a result in 1.15 format.
04792    */
04793 
04794   static __INLINE q15_t arm_pid_q15(
04795                     arm_pid_instance_q15 * S,
04796                     q15_t in)
04797   {
04798     q63_t acc;
04799     q15_t out;
04800 
04801     /* Implementation of PID controller */
04802 
04803     #ifdef ARM_MATH_CM0
04804 
04805     /* acc = A0 * x[n]  */
04806     acc = ((q31_t) S->A0 )* in ;
04807 
04808     #else
04809                 
04810     /* acc = A0 * x[n]  */
04811     acc = (q31_t) __SMUAD(S->A0, in);
04812     
04813     #endif
04814 
04815     #ifdef ARM_MATH_CM0
04816                            
04817     /* acc += A1 * x[n-1] + A2 * x[n-2]  */
04818     acc += (q31_t) S->A1  *  S->state[0] ;
04819     acc += (q31_t) S->A2  *  S->state[1] ;
04820 
04821     #else
04822 
04823     /* acc += A1 * x[n-1] + A2 * x[n-2]  */
04824     acc = __SMLALD(S->A1, (q31_t)__SIMD32(S->state), acc);
04825 
04826     #endif
04827 
04828     /* acc += y[n-1] */
04829     acc += (q31_t) S->state[2] << 15;
04830 
04831     /* saturate the output */
04832     out = (q15_t) (__SSAT((acc >> 15), 16));
04833 
04834     /* Update state */
04835     S->state[1] = S->state[0];
04836     S->state[0] = in;
04837     S->state[2] = out;
04838 
04839     /* return to application */
04840     return (out);
04841 
04842   }

04930    * The accumulator maintains 1.31 format by truncating lower 31 bits of the intermediate multiplication in 2.62 format.
04931    * There is saturation on the addition, hence there is no risk of overflow.
04932    */

05336    * Linear interpolation works by effectively drawing a straight line between two neighboring samples and returning the appropriate point along that line
05337    *
05338    * \par 
05339    * \image html LinearInterp.gif "Linear interpolation"

05344    *

05346    * <pre>
05347    *       y = y0 + (x - x0) * ((y1 - y0)/(x1-x0))

05356    * <code>S</code> points to an instance of the Linear Interpolate function data structure.
05357    * <code>x</code> is the input sample value. The functions returns the output value.
05358    * 
05359    * \par
05360    * if x is outside of the table boundary, Linear interpolation returns first value of the table 
05361    * if x is below input range and returns last value of table if x is above range.  
05362    */

05410           y1 = pYData[i + 1];

05583         y1 = pYData[index + 1u];

05588         /* Calculation of y1 * fract + y0 * (1-fract) and y is in 13.27(q27) format */

06673   /**
06674    * @defgroup BilinearInterpolate Bilinear Interpolation
06675    *
06676    * Bilinear interpolation is an extension of linear interpolation applied to a two dimensional grid.
06677    * The underlying function <code>f(x, y)</code> is sampled on a regular grid and the interpolation process
06678    * determines values between the grid points.
06679    * Bilinear interpolation is equivalent to two step linear interpolation, first in the x-dimension and then in the y-dimension.
06680    * Bilinear interpolation is often used in image processing to rescale images.
06681    * The CMSIS DSP library provides bilinear interpolation functions for Q7, Q15, Q31, and floating-point data types.
06682    *
06683    * <b>Algorithm</b>
06684    * \par
06685    * The instance structure used by the bilinear interpolation functions describes a two dimensional data table.
06686    * For floating-point, the instance structure is defined as:
06687    * <pre>
06688    *   typedef struct
06689    *   {
06690    *     uint16_t numRows;
06691    *     uint16_t numCols;
06692    *     float32_t *pData;
06693    * } arm_bilinear_interp_instance_f32;
06694    * </pre>
06695    *
06696    * \par
06697    * where <code>numRows</code> specifies the number of rows in the table;
06698    * <code>numCols</code> specifies the number of columns in the table;
06699    * and <code>pData</code> points to an array of size <code>numRows*numCols</code> values.
06700    * The data table <code>pTable</code> is organized in row order and the supplied data values fall on integer indexes.
06701    * That is, table element (x,y) is located at <code>pTable[x + y*numCols]</code> where x and y are integers.
06702    *
06703    * \par
06704    * Let <code>(x, y)</code> specify the desired interpolation point.  Then define:
06705    * <pre>
06706    *     XF = floor(x)
06707    *     YF = floor(y)
06708    * </pre>
06709    * \par
06710    * The interpolated output point is computed as:
06711    * <pre>
06712    *  f(x, y) = f(XF, YF) * (1-(x-XF)) * (1-(y-YF))
06713    *           + f(XF+1, YF) * (x-XF)*(1-(y-YF))
06714    *           + f(XF, YF+1) * (1-(x-XF))*(y-YF)
06715    *           + f(XF+1, YF+1) * (x-XF)*(y-YF)
06716    * </pre>
06717    * Note that the coordinates (x, y) contain integer and fractional components.  
06718    * The integer components specify which portion of the table to use while the
06719    * fractional components control the interpolation processor.
06720    *
06721    * \par
06722    * if (x,y) are outside of the table boundary, Bilinear interpolation returns zero output. 
06723    */
06724

06726    * @addtogroup BilinearInterpolate
06727    * @{
06728    */

06730   /**
06731   *
06732   * @brief  Floating-point bilinear interpolation.
06733   * @param[in,out] *S points to an instance of the interpolation structure.
06734   * @param[in] X interpolation coordinate.
06735   * @param[in] Y interpolation coordinate.
06736   * @return out interpolated value.
06737   */
06738 
06739   
06740   static __INLINE float32_t arm_bilinear_interp_f32(
06741                             const arm_bilinear_interp_instance_f32 * S,
06742                             float32_t X,
06743                             float32_t Y)
06744   {
06745     float32_t out;
06746     float32_t f00, f01, f10, f11;
06747     float32_t *pData = S->pData;
06748     int32_t xIndex, yIndex, index;
06749     float32_t xdiff, ydiff;
06750     float32_t b1, b2, b3, b4;
06751 
06752     xIndex = (int32_t) X;
06753     yIndex = (int32_t) Y;
06754 
06755     /* Care taken for table outside boundary */
06756     /* Returns zero output when values are outside table boundary */
06757     if(xIndex < 0 || xIndex > (S->numRows-1) || yIndex < 0  || yIndex > ( S->numCols-1))
06758     {
06759         return(0);
06760     }
06761     
06762     /* Calculation of index for two nearest points in X-direction */
06763     index = (xIndex - 1) + (yIndex-1) *  S->numCols ;
06764 
06765 
06766     /* Read two nearest points in X-direction */
06767     f00 = pData[index];
06768     f01 = pData[index + 1];
06769 
06770     /* Calculation of index for two nearest points in Y-direction */
06771     index = (xIndex-1) + (yIndex) * S->numCols;
06772 
06773 
06774     /* Read two nearest points in Y-direction */
06775     f10 = pData[index];
06776     f11 = pData[index + 1];
06777 
06778     /* Calculation of intermediate values */
06779     b1 = f00;
06780     b2 = f01 - f00;
06781     b3 = f10 - f00;
06782     b4 = f00 - f01 - f10 + f11;
06783 
06784     /* Calculation of fractional part in X */
06785     xdiff = X - xIndex;
06786 
06787     /* Calculation of fractional part in Y */
06788     ydiff = Y - yIndex;
06789 
06790     /* Calculation of bi-linear interpolated output */
06791      out = b1 + b2 * xdiff + b3 * ydiff + b4 * xdiff * ydiff;
06792 
06793    /* return to application */
06794     return (out);
06795 
06796   }

06798   /**
06799   *
06800   * @brief  Q31 bilinear interpolation.
06801   * @param[in,out] *S points to an instance of the interpolation structure.
06802   * @param[in] X interpolation coordinate in 12.20 format.
06803   * @param[in] Y interpolation coordinate in 12.20 format.
06804   * @return out interpolated value.
06805   */
06806 
06807   static __INLINE q31_t arm_bilinear_interp_q31(
06808                         arm_bilinear_interp_instance_q31 * S,
06809                         q31_t X,
06810                         q31_t Y)
06811   {
06812     q31_t out;                                   /* Temporary output */
06813     q31_t acc = 0;                               /* output */
06814     q31_t xfract, yfract;                        /* X, Y fractional parts */
06815     q31_t x1, x2, y1, y2;                        /* Nearest output values */
06816     int32_t rI, cI;                             /* Row and column indices */
06817     q31_t *pYData = S->pData;                    /* pointer to output table values */
06818     uint32_t nCols = S->numCols;                 /* num of rows */
06819 
06820 
06821     /* Input is in 12.20 format */
06822     /* 12 bits for the table index */
06823     /* Index value calculation */
06824     rI = ((X & 0xFFF00000) >> 20u);
06825 
06826     /* Input is in 12.20 format */
06827     /* 12 bits for the table index */
06828     /* Index value calculation */
06829     cI = ((Y & 0xFFF00000) >> 20u);
06830 
06831     /* Care taken for table outside boundary */
06832     /* Returns zero output when values are outside table boundary */
06833     if(rI < 0 || rI > (S->numRows-1) || cI < 0  || cI > ( S->numCols-1))
06834     {
06835         return(0);
06836     }
06837 
06838     /* 20 bits for the fractional part */
06839     /* shift left xfract by 11 to keep 1.31 format */
06840     xfract = (X & 0x000FFFFF) << 11u;
06841 
06842     /* Read two nearest output values from the index */
06843     x1 = pYData[(rI) + nCols * (cI)];
06844     x2 = pYData[(rI) + nCols * (cI) + 1u];
06845 
06846     /* 20 bits for the fractional part */
06847     /* shift left yfract by 11 to keep 1.31 format */
06848     yfract = (Y & 0x000FFFFF) << 11u;
06849 
06850     /* Read two nearest output values from the index */
06851     y1 = pYData[(rI) + nCols * (cI + 1)];
06852     y2 = pYData[(rI) + nCols * (cI + 1) + 1u];
06853 
06854     /* Calculation of x1 * (1-xfract ) * (1-yfract) and acc is in 3.29(q29) format */
06855     out = ((q31_t) (((q63_t) x1 * (0x7FFFFFFF - xfract)) >> 32));
06856     acc = ((q31_t) (((q63_t) out * (0x7FFFFFFF - yfract)) >> 32));
06857 
06858     /* x2 * (xfract) * (1-yfract)  in 3.29(q29) and adding to acc */
06859     out = ((q31_t) ((q63_t) x2 * (0x7FFFFFFF - yfract) >> 32));
06860     acc += ((q31_t) ((q63_t) out * (xfract) >> 32));
06861 
06862     /* y1 * (1 - xfract) * (yfract)  in 3.29(q29) and adding to acc */
06863     out = ((q31_t) ((q63_t) y1 * (0x7FFFFFFF - xfract) >> 32));
06864     acc += ((q31_t) ((q63_t) out * (yfract) >> 32));
06865 
06866     /* y2 * (xfract) * (yfract)  in 3.29(q29) and adding to acc */
06867     out = ((q31_t) ((q63_t) y2 * (xfract) >> 32));
06868     acc += ((q31_t) ((q63_t) out * (yfract) >> 32));
06869 
06870     /* Convert acc to 1.31(q31) format */
06871     return (acc << 2u);
06872 
06873   }
