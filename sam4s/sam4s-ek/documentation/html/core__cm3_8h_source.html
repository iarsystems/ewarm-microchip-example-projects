00886 /** \brief  Set Priority Grouping
00887 
00888   This function sets the priority grouping field using the required unlock sequence.
00889   The parameter PriorityGroup is assigned to the field SCB->AIRCR [10:8] PRIGROUP field.
00890   Only values from 0..7 are used.
00891   In case of a conflict between priority grouping and available
00892   priority bits (__NVIC_PRIO_BITS) the smallest possible priority group is set.
00893 
00894     \param [in]      PriorityGroup  Priority grouping field
00895  */
00896 static __INLINE void NVIC_SetPriorityGrouping(uint32_t PriorityGroup)
00897 {
00898   uint32_t reg_value;
00899   uint32_t PriorityGroupTmp = (PriorityGroup & (uint32_t)0x07);               /* only values 0..7 are used          */
00900 
00901   reg_value  =  SCB->AIRCR;                                                   /* read old register configuration    */
00902   reg_value &= ~(SCB_AIRCR_VECTKEY_Msk | SCB_AIRCR_PRIGROUP_Msk);             /* clear bits to change               */
00903   reg_value  =  (reg_value                                 |
00904                 ((uint32_t)0x5FA << SCB_AIRCR_VECTKEY_Pos) |
00905                 (PriorityGroupTmp << 8));                                     /* Insert write key and priorty group */
00906   SCB->AIRCR =  reg_value;
00907 }

01045 /** \brief  Encode Priority
01046 
01047     This function encodes the priority for an interrupt with the given priority group,
01048     preemptive priority value and sub priority value.
01049     In case of a conflict between priority grouping and available
01050     priority bits (__NVIC_PRIO_BITS) the samllest possible priority group is set.
01051 
01052     The returned priority value can be used for NVIC_SetPriority(...) function
01053 
01054     \param [in]     PriorityGroup  Used priority group
01055     \param [in]   PreemptPriority  Preemptive priority value (starting from 0)
01056     \param [in]       SubPriority  Sub priority value (starting from 0)
01057     \return                        Encoded priority for the interrupt
01058  */
01059 static __INLINE uint32_t NVIC_EncodePriority (uint32_t PriorityGroup, uint32_t PreemptPriority, uint32_t SubPriority)
01060 {
01061   uint32_t PriorityGroupTmp = (PriorityGroup & 0x07);          /* only values 0..7 are used          */
01062   uint32_t PreemptPriorityBits;
01063   uint32_t SubPriorityBits;
01064 
01065   PreemptPriorityBits = ((7 - PriorityGroupTmp) > __NVIC_PRIO_BITS) ? __NVIC_PRIO_BITS : 7 - PriorityGroupTmp;
01066   SubPriorityBits     = ((PriorityGroupTmp + __NVIC_PRIO_BITS) < 7) ? 0 : PriorityGroupTmp - 7 + __NVIC_PRIO_BITS;
01067 
01068   return (
01069            ((PreemptPriority & ((1 << (PreemptPriorityBits)) - 1)) << SubPriorityBits) |
01070            ((SubPriority     & ((1 << (SubPriorityBits    )) - 1)))
01071          );
01072 }

01075 /** \brief  Decode Priority
01076 
01077     This function decodes an interrupt priority value with the given priority group to
01078     preemptive priority value and sub priority value.
01079     In case of a conflict between priority grouping and available
01080     priority bits (__NVIC_PRIO_BITS) the samllest possible priority group is set.
01081 
01082     The priority value can be retrieved with NVIC_GetPriority(...) function
01083 
01084     \param [in]         Priority   Priority value
01085     \param [in]     PriorityGroup  Used priority group
01086     \param [out] pPreemptPriority  Preemptive priority value (starting from 0)
01087     \param [out]     pSubPriority  Sub priority value (starting from 0)
01088  */
01089 static __INLINE void NVIC_DecodePriority (uint32_t Priority, uint32_t PriorityGroup, uint32_t* pPreemptPriority, uint32_t* pSubPriority)
01090 {
01091   uint32_t PriorityGroupTmp = (PriorityGroup & 0x07);          /* only values 0..7 are used          */
01092   uint32_t PreemptPriorityBits;
01093   uint32_t SubPriorityBits;
01094 
01095   PreemptPriorityBits = ((7 - PriorityGroupTmp) > __NVIC_PRIO_BITS) ? __NVIC_PRIO_BITS : 7 - PriorityGroupTmp;
01096   SubPriorityBits     = ((PriorityGroupTmp + __NVIC_PRIO_BITS) < 7) ? 0 : PriorityGroupTmp - 7 + __NVIC_PRIO_BITS;
01097 
01098   *pPreemptPriority = (Priority >> SubPriorityBits) & ((1 << (PreemptPriorityBits)) - 1);
01099   *pSubPriority     = (Priority                   ) & ((1 << (SubPriorityBits    )) - 1);
01100 }
