00030 /** \addtogroup efc_module Working with EEFC
00031  * \ingroup peripherals_module
00032  * The EEFC driver provides the interface to configure and use the EEFC
00033  * peripheral.
00034  *
00035  * The user needs to set the number of wait states depending on the frequency used.\n
00036  * Configure number of cycles for flash read/write operations in the FWS field of EEFC_FMR.
00037  *
00038  * It offers a function to send flash command to EEFC and waits for the
00039  * flash to be ready.
00040  *
00041  * To send flash command, the user could do in either of following way:
00042  * <ul>
00043  * <li>Write a correct key, command and argument in EEFC_FCR. </li>
00044  * <li>Or, Use IAP (In Application Programming) function which is executed from
00045  * ROM directly, this allows flash programming to be done by code running in flash.</li>
00046  * <li>Once the command is achieved, it can be detected even by polling EEFC_FSR or interrupt.
00047  * </ul>
00048  *
00049  * The command argument could be a page number,GPNVM number or nothing, it depends on
00050  * the command itself. Some useful functions in this driver could help user tranlate physical
00051  * flash address into a page number and vice verse.
00052  *
00053  * For more accurate information, please look at the EEFC section of the
00054  * Datasheet.
00055  *
00056  * Related files :\n
00057  * \ref efc.c\n
00058  * \ref efc.h.\n
00059 */

00230 /**
00231  * \brief Performs the given command and wait until its completion (or an error).
00232  *
00233  * \param efc  Pointer to a Efc instance
00234  * \param command  Command to perform.
00235  * \param argument  Optional command argument.
00236  *
00237  * \return 0 if successful, otherwise returns an error code.
00238  */
00239 
00240 extern uint32_t EFC_PerformCommand( Efc* efc, uint32_t dwCommand, uint32_t dwArgument, uint32_t dwUseIAP )
00241 {
00242     if ( dwUseIAP != 0 )
00243     {
00244         /* Pointer on IAP function in ROM */
00245         static uint32_t (*IAP_PerformCommand)( uint32_t, uint32_t ) ;
00246 
00247         IAP_PerformCommand = (uint32_t (*)( uint32_t, uint32_t )) *((uint32_t*)CHIP_FLASH_IAP_ADDRESS ) ;
00248         IAP_PerformCommand( 0, EEFC_FCR_FKEY(0x5Au) | EEFC_FCR_FARG(dwArgument) | EEFC_FCR_FCMD(dwCommand) ) ;
00249 
00250         return (efc->EEFC_FSR & (EEFC_FSR_FLOCKE | EEFC_FSR_FCMDE | EEFC_FSR_FLERR)) ;
00251     }
00252     else
00253     {
00254         uint32_t dwStatus ;
00255 
00256         efc->EEFC_FCR = EEFC_FCR_FKEY(0x5Au) | EEFC_FCR_FARG(dwArgument) | EEFC_FCR_FCMD(dwCommand) ;
00257         do
00258         {
00259             dwStatus = efc->EEFC_FSR ;
00260         }
00261         while ( (dwStatus & EEFC_FSR_FRDY) != EEFC_FSR_FRDY ) ;
00262 
00263         return ( dwStatus & (EEFC_FSR_FLOCKE | EEFC_FSR_FCMDE | EEFC_FSR_FLERR) ) ;
00264     }
00265 }
