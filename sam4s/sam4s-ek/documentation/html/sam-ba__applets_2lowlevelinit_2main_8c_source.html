00143     switch (crystalFreq) {

00159         mul = USER_MUL(24);

00170         /* PLL Clock = crystalFreq * (mul + 1) / div = 18.432MHz * (37 + 1) / 7 = 100MHz */
00171         mul = USER_MUL(37);
00172         div = USER_DIV(7);

00179     /* other master clock configurations */
00180 
00181     /* 12MHz */
00182     default:
00183         /* PLL Clock = crystalFreq * (mul + 1) / div = 12MHz * (24 + 1) / 3 = 100MHz */
00184         mul = USER_MUL(24);

00187         prescaler = USER_PRESCALER(0);

00207     timeout = 0;

00210     PMC->CKGR_PLLBR = mul | div | pllcount;
00211     timeout = 0;
00212     while (!(PMC->PMC_SR & PMC_SR_LOCKB) && (timeout++ < CLOCK_TIMEOUT));
00213     for ( timeout = 0; !(PMC->PMC_SR & PMC_SR_MCKRDY) && (timeout++ < CLOCK_TIMEOUT) ; );

00216     PMC->PMC_MCKR = (mck & ~PMC_MCKR_CSS_Msk) | PMC_MCKR_CSS_MAIN_CLK;
00217     for ( timeout = 0; !(PMC->PMC_SR & PMC_SR_MCKRDY) && (timeout++ < CLOCK_TIMEOUT) ; );
00218 
00219     /* Switch to main clock */
00220     PMC->PMC_MCKR = PMC_MCKR_PRES_CLK | PMC_MCKR_CSS_MAIN_CLK;
00221     for ( timeout = 0; !(PMC->PMC_SR & PMC_SR_MCKRDY) && (timeout++ < CLOCK_TIMEOUT) ; );
00222 
00223     /* Switch to pllb clock */
00224     PMC->PMC_MCKR = mck;
